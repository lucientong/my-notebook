### 2.2 QPS 估算模板

```go
// QPS 估算
func EstimateQPS(dau int64, avgActionsPerDay int64) int64 {
    // 日活 * 平均每日操作次数 / 86400秒
    avgQPS := dau * avgActionsPerDay / 86400
    // 峰值一般取 2-5 倍
    peakQPS := avgQPS * 3
    return peakQPS
}

// 示例：DAU=1000万，每人每天发5条消息
// avgQPS = 10_000_000 * 5 / 86400 ≈ 578 QPS
// peakQPS ≈ 1734 QPS
```

```go
// 存储估算
func EstimateStorage() {
```

### 2.4 带宽估算

```go
// 带宽估算
func EstimateBandwidth() {
    peakQPS := 1734        // 峰值 QPS
    avgResponseSize := 10  // KB per response
    
    // 出带宽 = QPS * 响应大小
    outboundBandwidth := peakQPS * avgResponseSize // ≈ 17 MB/s ≈ 136 Mbps
    
    // 入带宽一般是出带宽的 1/10 到 1/3

}
```

### 3.5 核心算法 - Base62 编码

```go
package shorturl

import (
    "math"
    "strings"
)

const base62Chars = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ"

// Base62Encode 将数字ID转为短链字符串
func Base62Encode(id uint64) string {
    if id == 0 {
        return string(base62Chars[0])
    }
    
    var result strings.Builder
    for id > 0 {
        remainder := id % 62
        result.WriteByte(base62Chars[remainder])
        id /= 62
    }
    
    // 反转字符串
    encoded := result.String()
    runes := []rune(encoded)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}

// Base62Decode 将短链字符串转回数字ID
func Base62Decode(s string) uint64 {
    var result uint64
    for _, c := range s {
        result *= 62
        idx := strings.IndexRune(base62Chars, c)
        result += uint64(idx)
    }
    return result
}

// 7位 Base62 可表示: 62^7 = 3.5万亿 个短链，足够用
// 6位 Base62 可表示: 62^6 = 568亿
```

### 3.6 哈希冲突解决

```go
package shorturl

import (
    "crypto/md5"
    "encoding/hex"
    "fmt"
)

// 方案1: 全局自增ID + Base62（推荐）
// 优点: 无冲突
// 缺点: ID可预测，需要ID生成器

// 方案2: Hash + 冲突检测
func GenerateShortKey(longURL string) (string, error) {
    hash := md5.Sum([]byte(longURL))
    hexStr := hex.EncodeToString(hash[:])
    
    // 取前7个字符作为短链
    shortKey := hexStr[:7]
    
    // 检查冲突
    existing, err := db.Get(shortKey)
    if err != nil {
        return "", err
    }
    
    if existing != "" && existing != longURL {
        // 冲突：追加随机字符重试
        shortKey = hexStr[:7] + randomChar()
        // 递归或循环重试...
    }
    
    return shortKey, nil
}

// 方案3: 预生成Key池（推荐用于高并发）
type KeyPool struct {
    usedKeys   map[string]bool  // ZooKeeper/Redis 管理
    unusedKeys chan string       // 预生成的 key 队列
}

func (p *KeyPool) GetKey() string {
    return <-p.unusedKeys  // 从预生成池取，无冲突
}
```

### 3.7 缓存策略

```go
package shorturl

import (
    "context"
    "time"
    
    "github.com/go-redis/redis/v8"
)

type URLService struct {
    cache *redis.Client
    db    *MySQLClient
}

func (s *URLService) Redirect(ctx context.Context, shortKey string) (string, error) {
    // 1. 先查缓存
    longURL, err := s.cache.Get(ctx, "url:"+shortKey).Result()
    if err == nil {
        return longURL, nil
    }
    
    // 2. 缓存未命中，查DB
    longURL, err = s.db.GetLongURL(shortKey)
    if err != nil {
        return "", fmt.Errorf("short key not found: %s", shortKey)
    }
    
    // 3. 回写缓存，TTL=24h
    s.cache.Set(ctx, "url:"+shortKey, longURL, 24*time.Hour)
    
    return longURL, nil
}

// 缓存策略：
// - 热点URL常驻缓存（LRU淘汰）
// - 新创建URL立即写入缓存
// - 缓存命中率目标 > 80%（因为大部分短链访问集中在少量热点URL）
```

## 4. 设计秒杀系统 ⭐⭐⭐

### 4.1 需求分析

**功能需求：**
- 在指定时间开放抢购
- 库存有限（如1000件），先到先得
- 一人一单限制
- 下单成功后限时支付

**非功能需求：**
- 超高并发（百万级QPS瞬时流量）
- 数据一致性（不超卖）
- 高可用（核心链路不宕机）
- 低延迟（<200ms 响应）

**估算：**
- 100万人同时抢1000件商品
- 峰值 QPS: ~100万/s（持续约10s）
- 有效下单 QPS: ~1000/s

### 4.6 核心代码实现

```go
package seckill

import (
    "context"
    "fmt"
    "sync"
    "time"
    
    "github.com/go-redis/redis/v8"
    "github.com/segmentio/kafka-go"
)

// SeckillService 秒杀服务
type SeckillService struct {
    rdb         *redis.Client
    kafkaWriter *kafka.Writer
    localStock  sync.Map  // 本地库存标记 productID -> soldOut(bool)
}

// 预热：活动开始前将库存加载到 Redis
func (s *SeckillService) Preheat(ctx context.Context, productID int64, stock int) error {
    key := fmt.Sprintf("seckill:stock:%d", productID)
    return s.rdb.Set(ctx, key, stock, 2*time.Hour).Err()
}

// 秒杀核心逻辑
func (s *SeckillService) DoSeckill(ctx context.Context, userID, productID int64) (string, error) {
    // 1. 本地缓存判断（避免无效请求打到Redis）
    if soldOut, ok := s.localStock.Load(productID); ok && soldOut.(bool) {
        return "", fmt.Errorf("sold out")
    }
    
    // 2. Redis 判断是否已抢购（一人一单）
    dedupKey := fmt.Sprintf("seckill:dedup:%d:%d", productID, userID)
    added, err := s.rdb.SetNX(ctx, dedupKey, 1, 30*time.Minute).Result()
    if err != nil {
        return "", err
    }
    if !added {
        return "", fmt.Errorf("duplicate purchase")
    }
    
    // 3. Redis 库存预减（原子操作）
    stockKey := fmt.Sprintf("seckill:stock:%d", productID)
    stock, err := s.rdb.Decr(ctx, stockKey).Result()
    if err != nil {
        return "", err
    }
    
    if stock < 0 {
        // 库存不足，回滚
        s.rdb.Incr(ctx, stockKey)
        s.rdb.Del(ctx, dedupKey)
        // 设置本地售罄标记
        s.localStock.Store(productID, true)
        return "", fmt.Errorf("sold out")
    }
    
    // 4. 发送消息到 Kafka 异步下单
    orderID := generateOrderID(userID, productID)
    msg := kafka.Message{
        Key:   []byte(fmt.Sprintf("%d", productID)),
        Value: []byte(fmt.Sprintf(`{"order_id":"%s","user_id":%d,"product_id":%d}`, 
            orderID, userID, productID)),
    }
    
    if err := s.kafkaWriter.WriteMessages(ctx, msg); err != nil {
        // MQ 发送失败，回滚Redis
        s.rdb.Incr(ctx, stockKey)
        s.rdb.Del(ctx, dedupKey)
        return "", err
    }
    
    return orderID, nil
}

// Lua脚本实现原子性检查+扣减（更优方案）
const seckillLuaScript = `
    -- KEYS[1]: 库存key, KEYS[2]: 去重key
    -- ARGV[1]: userID
    
    -- 检查是否已购买

        return -1
    end
    
    -- 检查库存

    if stock <= 0 then
        return 0
    end
    
    -- 扣减库存 + 标记已购买
```

```go
package feed

import (
    "context"
    "fmt"
    "sort"
    "time"
    
    "github.com/go-redis/redis/v8"
)

const (
    BigVThreshold = 10000 // 粉丝超过1万视为大V
    FeedCacheSize = 200   // 每人缓存最近200条
)

type FeedService struct {
    rdb          *redis.Client
    postDB       *PostRepository
    followDB     *FollowRepository
    fanoutWorker *FanoutWorker
}

// 发布动态
func (s *FeedService) PublishPost(ctx context.Context, post *Post) error {
    // 1. 存储动态
    if err := s.postDB.Create(ctx, post); err != nil {
        return err
    }
    
    // 2. 获取粉丝列表
    followers, err := s.followDB.GetFollowers(ctx, post.UserID)
    if err != nil {
        return err
    }
    
    // 3. 判断是否大V
    if len(followers) > BigVThreshold {
        // 大V：不推送，读的时候拉取
        // 仅更新大V自己的发件箱
        return s.updateOutbox(ctx, post)
    }
    
    // 4. 普通用户：推送到所有粉丝的收件箱
    return s.fanoutWorker.Fanout(ctx, post, followers)
}

// 更新大V发件箱
func (s *FeedService) updateOutbox(ctx context.Context, post *Post) error {
    key := fmt.Sprintf("outbox:%d", post.UserID)
    member := &redis.Z{
        Score:  float64(post.CreatedAt.UnixMilli()),
        Member: post.ID,
    }
    s.rdb.ZAdd(ctx, key, member)
    s.rdb.ZRemRangeByRank(ctx, key, 0, -FeedCacheSize-1) // 只保留最近N条
    return nil
}

// 获取 Feed（混合模式）
func (s *FeedService) GetFeed(ctx context.Context, userID int64, cursor int64, limit int) ([]*Post, error) {
    // 1. 从收件箱获取（推过来的普通用户动态）
    inboxKey := fmt.Sprintf("inbox:%d", userID)
    inboxPostIDs, err := s.rdb.ZRevRangeByScore(ctx, inboxKey, &redis.ZRangeBy{
        Min:    "-inf",
        Max:    fmt.Sprintf("%d", cursor),
        Offset: 0,
        Count:  int64(limit),
    }).Result()
    if err != nil {
        return nil, err
    }
    
    // 2. 获取关注的大V列表
    bigVList, err := s.followDB.GetFollowedBigVs(ctx, userID)
    if err != nil {
        return nil, err
    }
    
    // 3. 从大V发件箱拉取（拉模式）
    var bigVPostIDs []string
    for _, bigV := range bigVList {
        outboxKey := fmt.Sprintf("outbox:%d", bigV)
        posts, _ := s.rdb.ZRevRangeByScore(ctx, outboxKey, &redis.ZRangeBy{
            Min:    "-inf",
            Max:    fmt.Sprintf("%d", cursor),
            Offset: 0,
            Count:  5, // 每个大V取最近5条
        }).Result()
        bigVPostIDs = append(bigVPostIDs, posts...)
    }
    
    // 4. 合并 + 排序
    allPostIDs := append(inboxPostIDs, bigVPostIDs...)
    posts, err := s.postDB.BatchGet(ctx, allPostIDs)
    if err != nil {
        return nil, err
    }
    
    // 按时间倒序排列
    sort.Slice(posts, func(i, j int) bool {
        return posts[i].CreatedAt.After(posts[j].CreatedAt)
    })
    
    // 取前 limit 条
    if len(posts) > limit {
        posts = posts[:limit]
    }
    
    return posts, nil
}

// Fanout Worker - 异步推送
type FanoutWorker struct {
    rdb *redis.Client
}

func (w *FanoutWorker) Fanout(ctx context.Context, post *Post, followers []int64) error {
    pipe := w.rdb.Pipeline()
    
    for _, followerID := range followers {
        inboxKey := fmt.Sprintf("inbox:%d", followerID)
        member := &redis.Z{
            Score:  float64(post.CreatedAt.UnixMilli()),
            Member: post.ID,
        }
        pipe.ZAdd(ctx, inboxKey, member)
        pipe.ZRemRangeByRank(ctx, inboxKey, 0, -FeedCacheSize-1)
    }
    
    _, err := pipe.Exec(ctx)
    return err
}
```

### 6.3 一致性哈希实现

```go
package cache

import (
    "crypto/md5"
    "encoding/binary"
    "fmt"
    "sort"
    "sync"
)

// ConsistentHash 一致性哈希
type ConsistentHash struct {
    mu       sync.RWMutex
    ring     []uint32          // 排序后的哈希值
    nodes    map[uint32]string // 哈希值 -> 节点名
    replicas int               // 虚拟节点数
}

func NewConsistentHash(replicas int) *ConsistentHash {
    return &ConsistentHash{
        nodes:    make(map[uint32]string),
        replicas: replicas, // 建议150-200个虚拟节点
    }
}

// AddNode 添加节点
func (ch *ConsistentHash) AddNode(node string) {
    ch.mu.Lock()
    defer ch.mu.Unlock()
    
    for i := 0; i < ch.replicas; i++ {
        key := fmt.Sprintf("%s#%d", node, i)
        hash := ch.hash(key)
        ch.ring = append(ch.ring, hash)
        ch.nodes[hash] = node
    }
    sort.Slice(ch.ring, func(i, j int) bool {
        return ch.ring[i] < ch.ring[j]
    })
}

// RemoveNode 移除节点
func (ch *ConsistentHash) RemoveNode(node string) {
    ch.mu.Lock()
    defer ch.mu.Unlock()
    
    newRing := make([]uint32, 0, len(ch.ring)-ch.replicas)
    for _, hash := range ch.ring {
        if ch.nodes[hash] != node {
            newRing = append(newRing, hash)
        } else {
            delete(ch.nodes, hash)
        }
    }
    ch.ring = newRing
}

// GetNode 根据key定位节点
func (ch *ConsistentHash) GetNode(key string) string {
    ch.mu.RLock()
    defer ch.mu.RUnlock()
    
    if len(ch.ring) == 0 {
        return ""
    }
    
    hash := ch.hash(key)
    // 二分查找第一个大于等于hash的位置
    idx := sort.Search(len(ch.ring), func(i int) bool {
        return ch.ring[i] >= hash
    })
    
    // 环形结构
    if idx == len(ch.ring) {
        idx = 0
    }
    
    return ch.nodes[ch.ring[idx]]
}

func (ch *ConsistentHash) hash(key string) uint32 {
    h := md5.Sum([]byte(key))
    return binary.BigEndian.Uint32(h[:4])
}
```

### 6.4 热点Key解决方案

```go
package cache

import (
    "context"
    "fmt"
    "math/rand"
    "sync"
    "time"
)

// 方案1: 本地缓存 + 随机过期
type HotKeyCache struct {
    localCache sync.Map // 本地L1缓存
    remote     *RedisCluster
}

func (c *HotKeyCache) Get(ctx context.Context, key string) (string, error) {
    // 1. 先查本地缓存
    if val, ok := c.localCache.Load(key); ok {
        entry := val.(*CacheEntry)
        if time.Now().Before(entry.ExpireAt) {
            return entry.Value, nil
        }
        c.localCache.Delete(key) // 过期删除
    }
    
    // 2. 查远程缓存
    val, err := c.remote.Get(ctx, key)
    if err != nil {
        return "", err
    }
    
    // 3. 如果是热点key，写入本地缓存（加随机jitter防雪崩）
    jitter := time.Duration(rand.Intn(5)) * time.Second
    c.localCache.Store(key, &CacheEntry{
        Value:    val,
        ExpireAt: time.Now().Add(10*time.Second + jitter),
    })
    
    return val, nil
}

type CacheEntry struct {
    Value    string
    ExpireAt time.Time
}

// 方案2: Key分片（将热点key拆分为多个子key）
func (c *HotKeyCache) GetWithSharding(ctx context.Context, key string, shards int) (string, error) {
    // 随机选择一个分片
    shardIdx := rand.Intn(shards)
    shardKey := fmt.Sprintf("%s_shard_%d", key, shardIdx)
    return c.remote.Get(ctx, shardKey)
}

// 写入时需要写所有分片
func (c *HotKeyCache) SetWithSharding(ctx context.Context, key, value string, shards int, ttl time.Duration) error {
    pipe := c.remote.Pipeline()
    for i := 0; i < shards; i++ {
        shardKey := fmt.Sprintf("%s_shard_%d", key, i)
        pipe.Set(ctx, shardKey, value, ttl)
    }
    _, err := pipe.Exec(ctx)
    return err
}

// 方案3: 热点key发现（基于访问频率统计）
type HotKeyDetector struct {
    mu      sync.Mutex
    counter map[string]int64
    window  time.Duration
}

func (d *HotKeyDetector) RecordAccess(key string) bool {
    d.mu.Lock()
    defer d.mu.Unlock()
    d.counter[key]++
    return d.counter[key] > 1000 // 阈值：1秒内超过1000次认为是热点
}
```

### 6.5 缓存问题与对策

| 问题 | 描述 | 解决方案 |
|------|------|----------|
| 缓存穿透 | 查询不存在的数据 | 布隆过滤器 + 空值缓存(TTL短) |
| 缓存击穿 | 热key过期瞬间大量请求 | 互斥锁(singleflight) + 永不过期 |
| 缓存雪崩 | 大量key同时过期 | 过期时间加随机jitter + 多级缓存 |
| 数据一致性 | 缓存与DB不一致 | Cache-Aside + 延时双删 |

```go
type Gateway struct {

    upgrader    websocket.Upgrader
    registry    SessionRegistry // 记录用户在哪台网关
    router      MessageRouter   // 跨网关消息路由

type Connection struct {
    UserID int64
    Conn   *websocket.Conn
    Send   chan []byte
    mu     sync.Mutex
}

func (g *Gateway) HandleWebSocket(w http.ResponseWriter, r *http.Request) {
    conn, err := g.upgrader.Upgrade(w, r, nil)
    if err != nil {
        return
    }
    
    userID := getUserIDFromToken(r) // JWT验证
    
    c := &Connection{
        UserID: userID,
        Conn:   conn,
        Send:   make(chan []byte, 256),
    }
    
    // 注册连接
    g.connections.Store(userID, c)
    g.registry.Register(userID, g.serverID) // 记录到Redis
    
    // 启动读写协程
    go c.readPump(g)
    go c.writePump()
    
    // 拉取离线消息
    go g.syncOfflineMessages(userID, c)
}

func (c *Connection) readPump(g *Gateway) {
    defer func() {
        g.connections.Delete(c.UserID)
        g.registry.Unregister(c.UserID)
        c.Conn.Close()
    }()
    
    c.Conn.SetReadDeadline(time.Now().Add(60 * time.Second))
    c.Conn.SetPongHandler(func(string) error {
        c.Conn.SetReadDeadline(time.Now().Add(60 * time.Second))
        return nil
    })
    
    for {
        _, message, err := c.Conn.ReadMessage()
        if err != nil {
            break
        }
        
        var msg WSMessage
        if err := json.Unmarshal(message, &msg); err != nil {
            continue
        }
        
        switch msg.Type {
        case "msg":
            g.handleMessage(context.Background(), &msg)
        case "ack":
            g.handleAck(context.Background(), &msg)
        case "heartbeat":
            // 心跳保活
        }
    }
}

func (c *Connection) writePump() {
    ticker := time.NewTicker(30 * time.Second) // 心跳间隔
    defer func() {
        ticker.Stop()
        c.Conn.Close()
    }()
    
    for {
        select {
        case message, ok := <-c.Send:
            if !ok {
                c.Conn.WriteMessage(websocket.CloseMessage, []byte{})
                return
            }
            c.mu.Lock()
            c.Conn.WriteMessage(websocket.TextMessage, message)
            c.mu.Unlock()
            
        case <-ticker.C:
            c.mu.Lock()
            c.Conn.WriteMessage(websocket.PingMessage, nil)
            c.mu.Unlock()
        }
    }
}
```

### 7.6 消息发送与路由

```go
// 消息处理核心逻辑
func (g *Gateway) handleMessage(ctx context.Context, msg *WSMessage) error {
    // 1. 生成服务端消息ID和序列号
    msg.SeqID = g.seqGenerator.Next(msg.getConversationID())
    msg.Timestamp = time.Now().UnixMilli()
    
    // 2. 持久化消息
    if err := g.messageStore.Save(ctx, msg); err != nil {
        return err
    }
    
    // 3. 路由消息到接收方
    if msg.GroupID > 0 {
        return g.routeGroupMessage(ctx, msg)
    }
    return g.routeDirectMessage(ctx, msg)
}

// 单聊路由
func (g *Gateway) routeDirectMessage(ctx context.Context, msg *WSMessage) error {
    // 查找接收方所在网关
    gatewayID, online := g.registry.Lookup(msg.To)
    
    if !online {
        // 离线：写入离线消息队列，等上线时同步
        return g.offlineStore.Push(ctx, msg.To, msg)
    }
    
    if gatewayID == g.serverID {
        // 本机：直接推送
        if conn, ok := g.connections.Load(msg.To); ok {
            data, _ := json.Marshal(msg)
            conn.(*Connection).Send <- data
        }
    } else {
        // 跨网关：通过Kafka/内部RPC转发
        return g.router.Route(ctx, gatewayID, msg)
    }
    
    return nil
}

// 群聊路由
func (g *Gateway) routeGroupMessage(ctx context.Context, msg *WSMessage) error {
    // 获取群成员列表
    members, err := g.groupService.GetMembers(ctx, msg.GroupID)
    if err != nil {
        return err
    }
    
    // 扇出到每个在线成员
    for _, memberID := range members {
        if memberID == msg.From {
            continue // 不发给自己
        }
        
        copyMsg := *msg
        copyMsg.To = memberID
        g.routeDirectMessage(ctx, &copyMsg)
    }
    
    return nil
}
```

```go
package search

import (
    "sort"
    "strings"
    "sync"
)

// 倒排索引结构
type InvertedIndex struct {
    mu    sync.RWMutex
    index map[string]*PostingList // term -> posting list
}

// PostingList 倒排列表
type PostingList struct {
    DocFreq  int        // 包含该词的文档数
    Postings []Posting  // 倒排项列表
}

// Posting 倒排项
type Posting struct {
    DocID     uint64   // 文档ID
    TermFreq  int      // 词在文档中出现次数
    Positions []int    // 词在文档中的位置（用于短语查询）
}

func NewInvertedIndex() *InvertedIndex {
    return &InvertedIndex{
        index: make(map[string]*PostingList),
    }
}

// IndexDocument 索引一个文档
func (idx *InvertedIndex) IndexDocument(docID uint64, content string) {
    // 1. 分词（简化版，实际需要更复杂的分词器）
    tokens := tokenize(content)
    
    // 2. 统计词频和位置
    termPositions := make(map[string][]int)
    for pos, token := range tokens {
        termPositions[token] = append(termPositions[token], pos)
    }
    
    // 3. 写入倒排索引
    idx.mu.Lock()
    defer idx.mu.Unlock()
    
    for term, positions := range termPositions {
        if _, exists := idx.index[term]; !exists {
            idx.index[term] = &PostingList{}
        }
        
        pl := idx.index[term]
        pl.DocFreq++
        pl.Postings = append(pl.Postings, Posting{
            DocID:     docID,
            TermFreq:  len(positions),
            Positions: positions,
        })
    }
}

// Search 搜索
func (idx *InvertedIndex) Search(query string) []SearchResult {
    idx.mu.RLock()
    defer idx.mu.RUnlock()
    
    queryTerms := tokenize(query)
    
    // 收集匹配文档
    docScores := make(map[uint64]float64)
    totalDocs := idx.getTotalDocs()
    
    for _, term := range queryTerms {
        pl, exists := idx.index[term]
        if !exists {
            continue
        }
        
        // 计算 IDF
        idf := calcIDF(totalDocs, pl.DocFreq)
        
        for _, posting := range pl.Postings {
            // TF-IDF 评分
            tf := calcTF(posting.TermFreq)
            docScores[posting.DocID] += tf * idf
        }
    }
    
    // 排序结果
    results := make([]SearchResult, 0, len(docScores))
    for docID, score := range docScores {
        results = append(results, SearchResult{DocID: docID, Score: score})
    }
    sort.Slice(results, func(i, j int) bool {
        return results[i].Score > results[j].Score
    })
    
    return results
}

type SearchResult struct {
    DocID uint64
    Score float64
}

// TF: 1 + log(termFreq)
func calcTF(termFreq int) float64 {
    if termFreq == 0 {
        return 0
    }
    return 1.0 + math.Log(float64(termFreq))
}

// IDF: log(totalDocs / docFreq)
func calcIDF(totalDocs, docFreq int) float64 {
    if docFreq == 0 {
        return 0
    }
    return math.Log(float64(totalDocs) / float64(docFreq))
}

func tokenize(text string) []string {
    // 简化版分词：按空格分割 + 小写
    text = strings.ToLower(text)
    return strings.Fields(text)
}
```

### 8.4 排序算法

```go
// BM25 评分（比TF-IDF更优）
func BM25Score(tf, docLen, avgDocLen float64, docFreq, totalDocs int) float64 {
    k1 := 1.2
    b := 0.75
    
    idf := math.Log((float64(totalDocs)-float64(docFreq)+0.5) / (float64(docFreq) + 0.5))
    tfNorm := (tf * (k1 + 1)) / (tf + k1*(1-b+b*docLen/avgDocLen))
    
    return idf * tfNorm
}

// 综合排序因子
type RankingFactors struct {
    BM25Score    float64 // 文本相关性
    PageRank     float64 // 页面权威度
    Freshness    float64 // 时效性
    ClickRate    float64 // 点击率
    UserPersonal float64 // 个性化因子
}

func FinalScore(f RankingFactors) float64 {
    return f.BM25Score*0.4 + f.PageRank*0.2 + f.Freshness*0.15 + 
           f.ClickRate*0.15 + f.UserPersonal*0.1
}
```

```go
// TokenBucket 令牌桶（单机版）
type TokenBucket struct {
    mu         sync.Mutex
    capacity   int64         // 桶容量
    tokens     int64         // 当前令牌数
    rate       int64         // 每秒生成令牌数
    lastRefill time.Time     // 上次填充时间
}

func NewTokenBucket(capacity, rate int64) *TokenBucket {
    return &TokenBucket{
        capacity:   capacity,
        tokens:     capacity,
        rate:       rate,
        lastRefill: time.Now(),
    }
}

func (tb *TokenBucket) Allow() bool {
    return tb.AllowN(1)
}

func (tb *TokenBucket) AllowN(n int64) bool {
    tb.mu.Lock()
    defer tb.mu.Unlock()
    
    // 补充令牌
    now := time.Now()
    elapsed := now.Sub(tb.lastRefill).Seconds()
    tb.tokens += int64(elapsed * float64(tb.rate))
    if tb.tokens > tb.capacity {
        tb.tokens = tb.capacity
    }
    tb.lastRefill = now
    
    // 消耗令牌
    if tb.tokens >= n {
        tb.tokens -= n
        return true
    }
    
    return false
}
```

```go
package ratelimit

import (
    "context"
    "fmt"
    "time"
    
    "github.com/go-redis/redis/v8"
)

// SlidingWindowLimiter 滑动窗口限流器（分布式版本）
type SlidingWindowLimiter struct {
    rdb       *redis.Client
    limit     int64         // 窗口内最大请求数
    window    time.Duration // 窗口大小
}

func NewSlidingWindowLimiter(rdb *redis.Client, limit int64, window time.Duration) *SlidingWindowLimiter {
    return &SlidingWindowLimiter{
        rdb:    rdb,
        limit:  limit,
        window: window,
    }
}

// IsAllowed 判断是否允许通过
func (l *SlidingWindowLimiter) IsAllowed(ctx context.Context, key string) (bool, error) {
    now := time.Now().UnixMilli()
    windowStart := now - l.window.Milliseconds()
    
    pipe := l.rdb.Pipeline()
    
    // 移除窗口外的记录
    pipe.ZRemRangeByScore(ctx, key, "0", fmt.Sprintf("%d", windowStart))
    
    // 统计窗口内的请求数
    countCmd := pipe.ZCard(ctx, key)
    
    // 添加当前请求
    pipe.ZAdd(ctx, key, &redis.Z{Score: float64(now), Member: now})
    
    // 设置key过期时间（防止内存泄漏）
    pipe.Expire(ctx, key, l.window)
    
    _, err := pipe.Exec(ctx)
    if err != nil {
        return true, err // 限流服务异常时降级放行
    }
    
    count := countCmd.Val()
    if count >= l.limit {
        // 超限，移除刚添加的记录
        l.rdb.ZRem(ctx, key, now)
        return false, nil
    }
    
    return true, nil
}

// Lua脚本实现（原子性更好）
const slidingWindowLua = `
    local key = KEYS[1]
    local now = tonumber(ARGV[1])
    local window = tonumber(ARGV[2])
    local limit = tonumber(ARGV[3])
    
    -- 清除过期记录
    redis.call('ZREMRANGEBYSCORE', key, 0, now - window)
    
    -- 获取当前窗口计数
    local count = redis.call('ZCARD', key)
    
    if count < limit then
        -- 允许通过，添加记录
        redis.call('ZADD', key, now, now .. '-' .. math.random(1000000))
        redis.call('PEXPIRE', key, window)
        return 1
    else
        return 0
    end
`

func (l *SlidingWindowLimiter) IsAllowedLua(ctx context.Context, key string) (bool, error) {
    result, err := l.rdb.Eval(ctx, slidingWindowLua, []string{key},
        time.Now().UnixMilli(),
        l.window.Milliseconds(),
        l.limit,

    if err != nil {

    return result == 1, nil
}
```

```go
// 分布式令牌桶 Lua 脚本
const tokenBucketLua = `
    local key = KEYS[1]
    local capacity = tonumber(ARGV[1])
    local rate = tonumber(ARGV[2])
    local now = tonumber(ARGV[3])
    local requested = tonumber(ARGV[4])
    
    -- 获取上次信息
    local info = redis.call('HMGET', key, 'tokens', 'last_time')
    local tokens = tonumber(info[1])
    local last_time = tonumber(info[2])
    
    -- 首次访问初始化
    if tokens == nil then
        tokens = capacity
        last_time = now
    end
    
    -- 计算补充的令牌
    local elapsed = (now - last_time) / 1000.0
    tokens = math.min(capacity, tokens + elapsed * rate)
    
    local allowed = 0
    if tokens >= requested then
        tokens = tokens - requested
        allowed = 1
    end
    
    -- 更新状态
    redis.call('HMSET', key, 'tokens', tokens, 'last_time', now)
    redis.call('PEXPIRE', key, 60000)
    
    return allowed
`

type DistributedTokenBucket struct {
    rdb      *redis.Client
    capacity int64
    rate     int64 // tokens per second
}

func (tb *DistributedTokenBucket) Allow(ctx context.Context, key string) (bool, error) {
    result, err := tb.rdb.Eval(ctx, tokenBucketLua, []string{key},
        tb.capacity,
        tb.rate,
        time.Now().UnixMilli(),
        1, // 请求1个令牌
    ).Int64()
    
    if err != nil {
        return true, err
    }
    return result == 1, nil
}
```

```go
func RateLimitMiddleware(limiter *SlidingWindowLimiter) func(http.Handler) http.Handler {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            // 限流维度：用户ID + 接口路径
            userID := r.Header.Get("X-User-ID")
            key := fmt.Sprintf("ratelimit:%s:%s", userID, r.URL.Path)
            
            allowed, err := limiter.IsAllowed(r.Context(), key)
            if err != nil {
                // 限流服务异常，降级放行
                next.ServeHTTP(w, r)
                return
            }
            
            if !allowed {
                w.Header().Set("X-RateLimit-Limit", "100")
                w.Header().Set("X-RateLimit-Remaining", "0")
                w.Header().Set("Retry-After", "1")
                http.Error(w, "Too Many Requests", http.StatusTooManyRequests)
                return
            }
            
            next.ServeHTTP(w, r)
        })
    }
}
```

```go
package notification

import (
    "context"
    "fmt"
    "strings"
    "time"

// NotificationService 通知服务
type NotificationService struct {
    templateStore *TemplateStore
    prefStore     *PreferenceStore
    router        *ChannelRouter
    deduplicator  *Deduplicator
    rateLimiter   *RateLimiter
}

// SendNotification 发送通知
func (s *NotificationService) SendNotification(ctx context.Context, req *NotifyRequest) error {
    // 1. 去重检查
    if s.deduplicator.IsDuplicate(ctx, req.DeduplicationKey) {
        return nil
    }
    
    // 2. 用户偏好检查
    pref, _ := s.prefStore.GetPreference(ctx, req.UserID, req.Channel)
    if pref != nil && !pref.Enabled {
        return nil // 用户关闭了该渠道
    }
    if pref != nil && isQuietTime(pref.QuietStart, pref.QuietEnd) {
        if req.Priority < PriorityHigh {
            return s.scheduleForLater(ctx, req) // 非紧急延后发送
        }
    }
    
    // 3. 频率控制
    if !s.rateLimiter.Allow(ctx, req.UserID, req.Channel) {
        return fmt.Errorf("rate limited for user %d", req.UserID)
    }
    
    // 4. 模板渲染
    content, err := s.renderTemplate(ctx, req.TemplateID, req.Variables)
    if err != nil {
        return err
    }
    
    // 5. 路由到对应渠道
    return s.router.Route(ctx, req.Channel, &RenderedNotification{
        UserID:   req.UserID,
        Title:    content.Title,
        Body:     content.Body,
        Priority: req.Priority,
    })
}

// 模板渲染
func (s *NotificationService) renderTemplate(ctx context.Context, templateID int64, vars map[string]string) (*Content, error) {
    tpl, err := s.templateStore.Get(ctx, templateID)
    if err != nil {
        return nil, err
    }
    
    title := tpl.TitleTpl
    body := tpl.BodyTpl
    
    for k, v := range vars {
        placeholder := fmt.Sprintf("{{%s}}", k)
        title = strings.ReplaceAll(title, placeholder, v)
        body = strings.ReplaceAll(body, placeholder, v)
    }
    
    return &Content{Title: title, Body: body}, nil
}

// ChannelRouter 渠道路由器
type ChannelRouter struct {
    workers map[string]ChannelWorker
}

type ChannelWorker interface {
    Send(ctx context.Context, notification *RenderedNotification) error
}

func (r *ChannelRouter) Route(ctx context.Context, channel string, notif *RenderedNotification) error {
    worker, ok := r.workers[channel]
    if !ok {
        return fmt.Errorf("unknown channel: %s", channel)
    }
    return worker.Send(ctx, notif)
}
```

### 11.3 Snowflake 实现

```
Snowflake ID 结构 (64 bit):

┌─────────┬───────────────────────────────┬──────────┬──────────────┐
│  符号位 │        时间戳(41bit)           │机器ID    │  序列号      │
│  1bit   │     毫秒级,可用69年           │ (10bit)  │  (12bit)     │
│    0    │                               │ 0-1023   │  0-4095      │
└─────────┴───────────────────────────────┴──────────┴──────────────┘
```

```go
package idgen

import (
    "fmt"
    "sync"
    "time"
)

const (
    epoch         = int64(1640995200000) // 2022-01-01 00:00:00 UTC 毫秒
    workerIDBits  = 10
    sequenceBits  = 12
    maxWorkerID   = -1 ^ (-1 << workerIDBits) // 1023
    maxSequence   = -1 ^ (-1 << sequenceBits)  // 4095
    workerIDShift = sequenceBits                // 12
    timestampShift = sequenceBits + workerIDBits // 22
)

// Snowflake ID生成器
type Snowflake struct {
    mu        sync.Mutex
    workerID  int64
    sequence  int64
    lastTime  int64
}

func NewSnowflake(workerID int64) (*Snowflake, error) {
    if workerID < 0 || workerID > maxWorkerID {
        return nil, fmt.Errorf("worker ID must be between 0 and %d", maxWorkerID)
    }
    return &Snowflake{workerID: workerID}, nil
}

func (s *Snowflake) NextID() (int64, error) {
    s.mu.Lock()
    defer s.mu.Unlock()
    
    now := time.Now().UnixMilli()
    
    if now < s.lastTime {
        // 时钟回拨处理
        diff := s.lastTime - now
        if diff < 5 {
            // 小幅回拨：等待追上
            time.Sleep(time.Duration(diff) * time.Millisecond)
            now = time.Now().UnixMilli()
        } else {
            return 0, fmt.Errorf("clock moved backwards: %d ms", diff)
        }
    }
    
    if now == s.lastTime {
        // 同一毫秒内，递增序列号
        s.sequence = (s.sequence + 1) & maxSequence
        if s.sequence == 0 {
            // 序列号溢出，等待下一毫秒
            now = s.waitNextMillis(now)
        }
    } else {
        // 新的毫秒，重置序列号
        s.sequence = 0
    }
    
    s.lastTime = now
    
    id := ((now - epoch) << timestampShift) |
        (s.workerID << workerIDShift) |
        s.sequence
    
    return id, nil
}

func (s *Snowflake) waitNextMillis(lastTime int64) int64 {
    now := time.Now().UnixMilli()
    for now <= lastTime {
        now = time.Now().UnixMilli()
    }
    return now
}

// 解析 Snowflake ID
func ParseSnowflakeID(id int64) (timestamp int64, workerID int64, sequence int64) {
    timestamp = (id >> timestampShift) + epoch
    workerID = (id >> workerIDShift) & maxWorkerID
    sequence = id & maxSequence
    return
}
```

### 11.4 号段模式实现

```go
package idgen

import (
    "database/sql"
    "fmt"
    "sync"
)

// 号段模式：批量从DB获取一段ID，本地分配
// 优点：减少DB访问频率，性能高
// 缺点：服务重启会浪费部分ID

type SegmentIDGenerator struct {
    mu       sync.Mutex
    db       *sql.DB
    bizTag   string   // 业务标签
    step     int64    // 每次获取的号段大小
    current  int64    // 当前分配位置
    max      int64    // 当前号段最大值
    
    // 双Buffer，异步加载下一个号段
    nextSeg  *Segment
    loading  bool
}

type Segment struct {
    Start int64
    End   int64
}

func NewSegmentIDGenerator(db *sql.DB, bizTag string, step int64) *SegmentIDGenerator {
    return &SegmentIDGenerator{
        db:     db,
        bizTag: bizTag,
        step:   step,
    }
}

func (g *SegmentIDGenerator) NextID() (int64, error) {
    g.mu.Lock()
    defer g.mu.Unlock()
    
    // 当前号段耗尽
    if g.current >= g.max {
        if g.nextSeg != nil {
            // 使用预加载的下一个号段
            g.current = g.nextSeg.Start
            g.max = g.nextSeg.End
            g.nextSeg = nil
        } else {
            // 同步加载新号段
            seg, err := g.loadSegment()
            if err != nil {
                return 0, err
            }
            g.current = seg.Start
            g.max = seg.End
        }
    }
    
    id := g.current
    g.current++
    
    // 使用率达到 70% 时异步加载下一个号段
    usage := float64(g.current-g.max+g.step) / float64(g.step)
    if usage > 0.7 && g.nextSeg == nil && !g.loading {
        g.loading = true
        go func() {
            seg, _ := g.loadSegment()
            g.mu.Lock()
            g.nextSeg = seg
            g.loading = false
            g.mu.Unlock()
        }()
    }
    
    return id, nil
}

// 从DB获取新号段（乐观锁）
func (g *SegmentIDGenerator) loadSegment() (*Segment, error) {
    for retry := 0; retry < 3; retry++ {
        // 读取当前最大值
        var maxID int64
        var version int64
        err := g.db.QueryRow(
            "SELECT max_id, version FROM id_alloc WHERE biz_tag = ?",
            g.bizTag,
        ).Scan(&maxID, &version)
        if err != nil {
            return nil, err
        }
        
        // CAS 更新
        newMax := maxID + g.step
        result, err := g.db.Exec(
            "UPDATE id_alloc SET max_id = ?, version = version + 1 WHERE biz_tag = ? AND version = ?",
            newMax, g.bizTag, version,
        )
        if err != nil {
            return nil, err
        }
        
        affected, _ := result.RowsAffected()
        if affected == 1 {
            return &Segment{Start: maxID + 1, End: newMax}, nil
        }
        // 乐观锁冲突，重试
    }
    
    return nil, fmt.Errorf("failed to allocate segment after retries")
}

/*
CREATE TABLE id_alloc (
    biz_tag   VARCHAR(64) PRIMARY KEY,
    max_id    BIGINT NOT NULL DEFAULT 0,
    step      INT NOT NULL DEFAULT 1000,
    version   BIGINT NOT NULL DEFAULT 0,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
*/
```

```sql
    INDEX idx_video (video_id)
);

-- 上传分片记录
CREATE TABLE upload_chunks (
    upload_id   VARCHAR(64) NOT NULL,
    chunk_index INT NOT NULL,
    chunk_size  BIGINT,
    etag        VARCHAR(64),
    status      TINYINT DEFAULT 0,
    PRIMARY KEY (upload_id, chunk_index)
);
```

### 12.4 分片上传实现

```go
package video

import (
    "context"
    "fmt"
    "io"
    "os"
)

const ChunkSize = 5 * 1024 * 1024 // 5MB per chunk

// UploadService 视频上传服务
type UploadService struct {
    oss       ObjectStorage
    db        *Repository
    transcode *TranscodeService
}

// InitiateUpload 初始化分片上传
func (s *UploadService) InitiateUpload(ctx context.Context, req *InitUploadRequest) (*InitUploadResponse, error) {
    // 1. 创建视频记录
    video := &Video{
        ID:     generateID(),
        UserID: req.UserID,
        Title:  req.Title,
        Status: StatusUploading,
    }
    if err := s.db.CreateVideo(ctx, video); err != nil {
        return nil, err
    }
    
    // 2. 计算分片数
    totalChunks := (req.FileSize + ChunkSize - 1) / ChunkSize
    
    // 3. 初始化OSS分片上传
    uploadID, err := s.oss.InitMultipartUpload(ctx, req.FileName)
    if err != nil {
        return nil, err
    }
    
    // 4. 为每个分片生成预签名URL
    uploadURLs := make([]string, totalChunks)
    for i := int64(0); i < totalChunks; i++ {
        url, _ := s.oss.GeneratePresignedURL(ctx, uploadID, int(i))
        uploadURLs[i] = url
    }
    
    return &InitUploadResponse{
        VideoID:     video.ID,
        UploadID:    uploadID,
        TotalChunks: int(totalChunks),
        UploadURLs:  uploadURLs,
    }, nil
}

// CompleteUpload 完成上传，触发转码
func (s *UploadService) CompleteUpload(ctx context.Context, videoID int64, uploadID string) error {
    // 1. 合并分片
    fileURL, err := s.oss.CompleteMultipartUpload(ctx, uploadID)
    if err != nil {
        return err
    }
    
    // 2. 更新视频状态
    s.db.UpdateVideo(ctx, videoID, map[string]interface{}{
        "original_url": fileURL,
        "status":       StatusTranscoding,
    })
    
    // 3. 触发转码任务
    return s.transcode.SubmitJob(ctx, &TranscodeJob{
        VideoID:     videoID,
        SourceURL:   fileURL,
        Resolutions: []string{"360p", "720p", "1080p"},
    })
    
    return nil
}
```

### 12.5 转码服务

```go
package video

import (
    "context"
    "fmt"
)

type TranscodeService struct {
    queue   MessageQueue
    workers int
}

type TranscodeJob struct {
    VideoID     int64
    SourceURL   string
    Resolutions []string
}

// SubmitJob 提交转码任务
func (s *TranscodeService) SubmitJob(ctx context.Context, job *TranscodeJob) error {
    // 为每个分辨率创建子任务
    for _, resolution := range job.Resolutions {
        subJob := &TranscodeSubJob{
            VideoID:    job.VideoID,
            SourceURL:  job.SourceURL,
            Resolution: resolution,
            Format:     "hls", // HLS切片格式
        }
        if err := s.queue.Publish(ctx, "transcode", subJob); err != nil {
            return err
        }
    }
    return nil
}

// Worker 转码工作协程
func (s *TranscodeService) Worker(ctx context.Context) {
    for {
        job, err := s.queue.Consume(ctx, "transcode")
        if err != nil {
            continue
        }
        
        subJob := job.(*TranscodeSubJob)
        
        // 调用 FFmpeg 转码
        outputURL, err := s.transcode(ctx, subJob)
        if err != nil {
            // 重试 or 标记失败
            s.handleFailure(ctx, subJob, err)
            continue
        }
        
        // 更新转码记录
        s.db.CreateTranscode(ctx, &VideoTranscode{
            VideoID:    subJob.VideoID,
            Resolution: subJob.Resolution,
            Format:     subJob.Format,
            FileURL:    outputURL,
            Status:     StatusCompleted,
        })
        
        // 检查是否所有分辨率都转码完成
        s.checkAllCompleted(ctx, subJob.VideoID)
    }
}

// FFmpeg 转码参数映射
func getFFmpegParams(resolution string) string {
    params := map[string]string{
        "360p":  "-vf scale=640:360 -b:v 800k -b:a 96k",
        "720p":  "-vf scale=1280:720 -b:v 2500k -b:a 128k",
        "1080p": "-vf scale=1920:1080 -b:v 5000k -b:a 192k",
    }
    return params[resolution]
}
```

### 治理三条（面试心态管理）

1. **不要追求完美设计** —— 面试考察的是思考过程和权衡能力，不存在唯一正确答案。展示你如何分析tradeoff比给出"标准答案"更重要。

3. **控制节奏，先广后深** —— 先用5分钟画出全局架构，再由面试官引导深入某个模块。避免一开始就陷入某个细节出不来。

### 初级题（5题）

**Q1: 系统设计面试的四步法是什么？每步的时间分配？**

答：① 需求澄清(3-5min) → ② 概要设计(10-15min) → ③ 详细设计(15-20min) → ④ 扩展讨论(5-8min)。

**Q2: 如何估算一个DAU=1000万的社交App的消息QPS？**

答：假设每人每天发50条消息，avg QPS = 1000万×50/86400 ≈ 5800 QPS，峰值取3倍 ≈ 17K QPS。

**Q3: 短链接服务为什么用Base62而不用Base64？**

答：Base62只包含0-9、a-z、A-Z，不含+/=等特殊字符，URL友好，不需要额外编码。7位Base62可表示62^7≈3.5万亿个短链。

**Q4: 秒杀系统为什么要多级过滤？**

答：层层削减流量——CDN过滤静态请求→网关限流→本地缓存售罄判断→Redis预减→最终只有少量有效请求到达DB，保护后端不被压垮。

**Q5: Feed流的推模式和拉模式各有什么优缺点？**

答：推模式写放大（发布时推送给所有粉丝），但读快；拉模式读放大（刷Feed时聚合所有关注人），但写轻。工业级系统用混合模式解决大V写放大问题。

### 中级题（5题）

**Q1: 一致性哈希为什么需要虚拟节点？建议多少个？**

答：虚拟节点解决数据倾斜问题。少量物理节点时，哈希环上分布不均导致某些节点负载过高。建议每个物理节点150-200个虚拟节点，实测数据分布可达到较好的均匀性。

**Q2: 聊天系统如何保证消息不丢？**

答：三重保障：① 发送方重试（未收到Server ACK时） ② 服务端持久化后再ACK ③ 接收方离线时存入离线队列，上线后按seq_id增量同步。每个会话维护递增序列号，客户端对比可发现缺失消息。

**Q3: 分布式限流如何平衡精确性和性能？**

答：方案A：中心化Redis精确计数（一致但有网络开销）；方案B：全局配额预分配到本地（性能好但不精确）；方案C：本地预扣+定期同步（折中方案，允许短时超限10%）。根据业务容忍度选择。

**Q4: Snowflake算法遇到时钟回拨怎么办？**

答：① 小幅回拨(<5ms)：等待追上 ② 中幅回拨：使用扩展位/备用workerID ③ 大幅回拨(>1s)：报警+拒绝服务。根本方案：用单调递增的逻辑时钟替代物理时钟。

**Q5: 视频平台为什么用HLS而不直接提供MP4？**

答：HLS(HTTP Live Streaming)将视频切成小片段(如10s)，支持自适应码率（根据网络状况切换分辨率），利于CDN缓存（小文件），支持加密(DRM)，且用HTTP协议穿透防火墙。MP4需要完整下载或Range请求，不如HLS灵活。

### 深入题（15题 Q&A）

**Q1: 短链接服务如何处理热点短链导致的Redis单key瓶颈？**

A: 三层方案：① 应用层本地缓存(sync.Map/BigCache)缓存top-K热点URL，TTL 10s+随机jitter ② Redis层做key分片(key_shard_0~key_shard_9)，读时随机选择一个分片 ③ 监控热点key自动发现(如Redis 4.0的hotkeys命令)，触发动态缓存预热。根本思路是减少对单key的并发访问。

**Q2: 秒杀系统如何防止Redis库存预减后Kafka发送失败导致的库存不一致？**

A: 采用"先减后补"策略：Redis扣减成功后发Kafka，如果Kafka发送失败则立即Redis INCR回补。同时设置补偿定时任务：对比Redis库存和实际订单数，发现不一致时修正。极端情况下Redis宕机丢数据，用DB订单数作为最终基准重建Redis库存。

**Q3: Feed流系统中，用户取关一个大V后如何处理已经在收件箱中的旧动态？**

A: 混合模式下大V动态不推送到收件箱(拉取方式)，取关后读Feed时自然不再拉取该大V的发件箱。如果是推模式产生的旧数据，有两种策略：① 惰性删除：读取时过滤(检查关注关系) ② 异步清理：后台任务批量删除(不影响主流程性能)。推荐惰性删除+定期后台清理结合。

**Q4: 分布式缓存在线扩容时如何保证数据不丢失？**

A: 渐进式迁移方案：① 新节点加入哈希环 ② 计算需要迁移的key范围(一致性哈希只需迁移相邻节点的部分数据) ③ 开启双写：新数据同时写新旧节点 ④ 后台迁移存量数据 ⑤ 迁移完成后切换读路由到新节点 ⑥ 确认无误后删除旧节点冗余数据。整个过程对客户端透明，参考Redis Cluster的slot迁移机制。

**Q5: 聊天系统的群聊(500人)如何避免消息扇出导致的网关压力？**

A: 分层处理：① 小群(<100人)直接推送到每个在线成员 ② 大群(>100人)改为"写扩散"变"读扩散"：消息只存一份到群timeline，成员拉取 ③ 网关层做合并推送：同一个网关上的多个群成员只投递一次，由网关本地扇出 ④ 消息推送限流：单群消息推送QPS上限，超限走增量同步。

**Q6: 搜索引擎如何实现近实时索引更新？**

A: 采用段(Segment)机制(参考Lucene/ES)：① 新文档先写入内存buffer ② 每隔1秒执行refresh，将buffer内容生成新的不可变segment(此时可搜索) ③ 后台定期merge多个小segment为大segment(减少文件数) ④ 使用事务日志(translog)保证refresh前的数据不丢。新数据从写入到可搜索的延迟约1秒。

**Q7: 限流服务自身不可用时怎么办？**

A: 分级降级策略：① 第一层：Redis不可用时切换到本地令牌桶(预分配配额) ② 第二层：本地限流也异常时降级为放行(允许超限) ③ 配合熔断：限流服务连续失败N次后熔断，定期探测恢复 ④ 关键原则：限流是保护措施，限流本身不应成为故障点，宁可短时超限也不能阻断正常业务。

**Q8: 通知系统如何避免对用户过度打扰？**

A: 多维度频控：① 时间维度：每用户每小时/每天通知上限 ② 类型维度：同类通知合并(如"张三等5人点赞了你") ③ 优先级维度：低优先级通知在免打扰时段静默 ④ 智能退避：用户连续忽略某类通知后降低推送频率 ⑤ 全局限流：系统级限制总推送量避免供应商封禁。这些规则在Notification Processor中统一执行。

**Q9: 分布式ID的号段模式中，双Buffer是如何工作的？**

A: 双Buffer机制：当前号段使用率达到70%时，异步预加载下一个号段到内存(nextSegment)。当前号段耗尽时无需等待DB IO，直接切换到预加载的号段。好处：消除了DB IO对ID分配延迟的影响，实现了毫秒级ID生成。美团Leaf开源实现就采用此方案，实测99.9%延迟<1ms。

**Q10: 视频平台如何实现秒传功能？**

A: 基于文件指纹去重：① 客户端上传前先计算文件MD5/SHA256 ② 发送指纹到服务端查询 ③ 如果指纹已存在，直接关联到已有视频文件(无需重新上传) ④ 注意：同一个视频文件可被多个用户"拥有"(引用计数)，删除时只减引用不删文件 ⑤ 安全考虑：仅对完全相同的文件做去重，不同用户可见性通过权限控制。

**Q11: 如何设计一个高可用的WebSocket网关集群？**

A: 关键设计：① 无状态网关+有状态会话注册(Redis/etcd记录用户在哪个网关) ② 连接迁移：网关滚动重启时通知客户端重连到其他节点 ③ 跨网关通信：用户A在GW-1，用户B在GW-2，消息通过内部MQ(Kafka)转发 ④ 负载均衡：新连接按当前连接数最少分配(非简单轮询) ⑤ 心跳机制：30s一次ping/pong，超时断开释放资源 ⑥ 容量规划：单机10万连接(Go协程优势)，50台网关支撑500万在线。

**Q12: 秒杀系统中，如何解决Redis集群模式下Lua脚本不支持跨slot的问题？**

A: 方案：① 使用HashTag：将相关key放到同一slot，如{seckill_product_001}:stock和{seckill_product_001}:dedup，大括号内容决定slot ② 单key设计：将库存和去重信息存在同一个Hash结构中(HSET seckill:001 stock 1000 / HSET seckill:001 user:123 1) ③ 业务层保证：一个商品的所有key路由到同一节点。核心是让Lua脚本涉及的所有key在同一个slot。

**Q13: Feed流系统如何处理冷启动用户（新注册，没有关注任何人）？**

A: 分级策略：① 系统推荐：新用户Feed由推荐系统填充(热门内容/编辑精选) ② 引导关注：注册流程中推荐感兴趣的用户/话题关注 ③ 兜底内容：维护各分类的优质内容池，新用户从中获取 ④ 行为追踪：用户的点击/停留行为快速建立兴趣画像，3天内切换到个性化推荐 ⑤ 混合策略：Feed = 关注内容(有则展示) + 推荐内容(补充填满一屏)。

**Q14: 搜索引擎如何处理中文分词的歧义问题？**

A: 多策略组合：① 正向最大匹配+反向最大匹配，取分词数最少的结果 ② 基于统计模型(HMM/CRF)处理未登录词 ③ 用户词典：维护业务专有词库(品牌名/人名) ④ 索引时多粒度分词：同时索引"中华人民共和国"和"中华/人民/共和国" ⑤ 查询时使用细粒度分词，提高召回率 ⑥ 结合上下文：利用深度学习模型(BERT)理解查询意图。

**Q15: 整体系统设计中，如何选择CP还是AP？请举例说明。**

A: 根据业务场景决定：

| 场景 | 选择 | 原因 |
|------|------|------|
| 支付/库存 | CP(强一致性) | 不能超卖/多扣，数据正确性最重要 |
| Feed流/推荐 | AP(高可用) | 少看一条动态可接受，不能服务不可用 |
| 消息系统 | 折中(最终一致) | 消息可延迟到达，但不能丢，不能长时间不可用 |
| 配置中心 | CP | 配置错误可能导致全局故障 |
| 缓存 | AP | 缓存不一致只是暂时的，DB是最终一致来源 |

核心原则：涉及"钱"和"核心状态"的选CP，涉及"体验"和"展示"的选AP。大多数互联网系统选择AP+最终一致性，通过补偿机制保证数据最终正确。

**D2：系统设计面试中，面试官追问"如果QPS从1万涨到100万怎么办"，如何回答？**

**参考思路**：
- 不要直接说"加机器"，展示系统化思考
- 识别瓶颈：是计算瓶颈（CPU）还是存储瓶颈（DB/磁盘）还是网络瓶颈
- 逐层优化：缓存（减少DB压力）→ 读写分离（DB扩展读能力）→ 分库分表（DB扩展写能力）→ 异步化（MQ削峰）→ 水平扩展（无状态服务多副本）
- 架构演进：单体→微服务→CQRS→事件驱动，每一步解决什么问题说清楚
- 关键：给出具体数字（"如果单机能扛5000QPS，100万QPS需要200台+冗余50%=300台"）
