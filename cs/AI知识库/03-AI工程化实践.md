```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import openai
import asyncio

app = FastAPI()

class LLMRequest(BaseModel):
    query: str
    user_id: str
    model: str = "gpt-4"

class LLMService:
    def __init__(self):
        self.router = ModelRouter()
        self.cache = CacheLayer()
        self.rate_limiter = RateLimiter()
        self.monitor = Monitor()
    
    async def generate(self, request: LLMRequest) -> str:
        """生成回答"""
        # 1. 监控：记录请求开始
        span = self.monitor.start_span("llm_generate")
        
        try:
            # 2. 策略：限流检查
            if not await self.rate_limiter.allow(request.user_id):
                raise HTTPException(429, "请求过于频繁")
            
            # 3. 策略：缓存检查
            cache_key = f"{request.model}:{request.query}"
            cached = await self.cache.get(cache_key)
            if cached:
                self.monitor.record_cache_hit()
                return cached
            
            # 4. 路由：选择模型
            model = self.router.select_model(request)
            
            # 5. LLM：生成回答
            response = await self.call_llm(model, request.query)
            
            # 6. 策略：写入缓存
            await self.cache.set(cache_key, response, ttl=3600)
            
            # 7. 监控：记录成功
            self.monitor.record_success(span)
            
            return response
            
        except Exception as e:
            # 8. 监控：记录失败
            self.monitor.record_error(span, e)
            
            # 9. 策略：降级处理
            return await self.fallback(request.query)
    
    async def call_llm(self, model: str, query: str) -> str:
        """调用LLM"""
        response = await openai.ChatCompletion.acreate(
            model=model,
            messages=[{"role": "user", "content": query}],
            temperature=0.7
        )
        return response.choices[0].message.content
    
    async def fallback(self, query: str) -> str:
        """降级处理"""
        return "服务暂时不可用，请稍后再试。"

# API端点
service = LLMService()

@app.post("/chat")
async def chat(request: LLMRequest):
    response = await service.generate(request)
    return {"response": response}
```

### 1.2 模型路由

**策略**：
```python
class ModelRouter:
    def __init__(self):
        self.models = {
            "simple": "gpt-3.5-turbo",    # 简单问答
            "complex": "gpt-4",            # 复杂推理
            "code": "claude-3-opus",       # 代码生成
            "long_context": "claude-3"     # 长文本
        }
    
    def select_model(self, request: LLMRequest) -> str:
        """根据请求选择模型"""
        query = request.query
        
        # 1. 用户指定模型
        if request.model:
            return request.model
        
        # 2. 代码相关 → Claude
        if any(kw in query.lower() for kw in ["代码", "code", "编程", "bug"]):
            return self.models["code"]
        
        # 3. 长文本 → Claude (200K上下文)
        if len(query) > 5000:
            return self.models["long_context"]
        
        # 4. 复杂推理 → GPT-4
        if self.is_complex(query):
            return self.models["complex"]
        
        # 5. 默认 → GPT-3.5 (便宜快速)
        return self.models["simple"]
    
    def is_complex(self, query: str) -> bool:
        """判断是否复杂问题"""
        # 启发式规则
        complex_keywords = ["分析", "推理", "设计", "规划", "比较"]
        return any(kw in query for kw in complex_keywords)
```

```python
class GrayReleaseRouter:
    def __init__(self):
        self.config = {
            "model_a": "gpt-4-0613",  # 旧版本
            "model_b": "gpt-4-1106",  # 新版本
            "gray_percentage": 10      # 10%流量到新版本
        }
    
    def select_model(self, user_id: str) -> str:
        """灰度路由"""
        # 根据user_id哈希决定路由
        hash_value = int(hashlib.md5(user_id.encode()).hexdigest(), 16)
        percentage = hash_value % 100
        
        if percentage < self.config["gray_percentage"]:
            return self.config["model_b"]  # 新版本
        else:
            return self.config["model_a"]  # 旧版本
```

```python
class PromptManager:
    def __init__(self, base_dir="prompts"):
        self.base_dir = base_dir
        self.cache = {}
    
    def get_prompt(self, name: str, version: str = "current") -> str:
        """获取Prompt模板"""
        cache_key = f"{version}:{name}"
        
        if cache_key in self.cache:
            return self.cache[cache_key]
        
        # 读取文件
        path = os.path.join(self.base_dir, version, f"{name}.txt")
        with open(path, 'r', encoding='utf-8') as f:
            template = f.read()
        
        self.cache[cache_key] = template
        return template
    
    def render(self, name: str, **kwargs) -> str:
        """渲染Prompt"""
        template = self.get_prompt(name)
        return template.format(**kwargs)

# 使用

    language="Python",
```

## 3. 评估体系

### 3.1 评估指标

**分类**：
| 类别 | 指标 | 描述 | 计算方法 |
|------|------|------|----------|
| **准确性** | Accuracy | 答案正确率 | 正确数 / 总数 |
| **相关性** | Relevance | 与问题的相关程度 | 人工评分 / 自动评分 |
| **完整性** | Completeness | 信息覆盖度 | 关键点覆盖率 |
| **幻觉率** | Hallucination Rate | 编造信息的比例 | 幻觉数 / 总回答数 |
| **文本质量** | BLEU/ROUGE | 与参考答案的相似度 | N-gram重叠 |
| **语义相似度** | Embedding Similarity | 语义层面相似度 | Cosine相似度 |

**Python实现**：
```python
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np

class LLMEvaluator:
    def __init__(self, embedding_model):
        self.embedding_model = embedding_model
    
    def evaluate_accuracy(self, predictions, references):
        """准确性"""
        correct = sum(p == r for p, r in zip(predictions, references))
        return correct / len(predictions)
    
    def evaluate_relevance(self, response: str, query: str) -> float:
        """相关性（基于Embedding）"""
        # 向量化
        query_emb = self.embedding_model.encode(query)
        response_emb = self.embedding_model.encode(response)
        
        # Cosine相似度
        similarity = cosine_similarity(
            [query_emb], [response_emb]
        )[0][0]
        
        return similarity
    
    def evaluate_hallucination(self, response: str, context: str) -> bool:
        """幻觉检测"""
        # 简单策略：检查回答中的事实是否出现在上下文中
        facts = self.extract_facts(response)
        
        for fact in facts:
            if fact not in context:
                return True  # 存在幻觉
        
        return False
    
    def evaluate_bleu(self, prediction: str, reference: str) -> float:
        """BLEU分数"""
        from nltk.translate.bleu_score import sentence_bleu
        
        pred_tokens = prediction.split()
        ref_tokens = [reference.split()]
        
        return sentence_bleu(ref_tokens, pred_tokens)
    
    def evaluate_rouge(self, prediction: str, reference: str) -> dict:
        """ROUGE分数"""
        from rouge import Rouge
        
        rouge = Rouge()
        scores = rouge.get_scores(prediction, reference)[0]
        
        return {
            "rouge-1": scores["rouge-1"]["f"],
            "rouge-2": scores["rouge-2"]["f"],
            "rouge-l": scores["rouge-l"]["f"]
        }
```

```python
class CostAwareRouter:
    def __init__(self):
        self.models = {
            "cheap": {
                "name": "gpt-3.5-turbo",
                "input_cost": 0.001,
                "output_cost": 0.002
            },
            "balanced": {
                "name": "gpt-4-turbo",
                "input_cost": 0.01,
                "output_cost": 0.03
            },
            "premium": {
                "name": "gpt-4",
                "input_cost": 0.03,
                "output_cost": 0.06
            }
        }
    
    def select_model(self, query: str, max_cost: float = None) -> str:
        """根据预算选择模型"""
        # 估算token数
        estimated_tokens = len(query) / 4 * 1.5  # 输入+输出
        
        # 计算成本
        for tier, model in self.models.items():
            cost = estimated_tokens * (model["input_cost"] + model["output_cost"]) / 1000
            
            if max_cost is None or cost <= max_cost:
                return model["name"]
        
        # 超预算，使用最便宜的模型
        return self.models["cheap"]["name"]
```

### 4.2 缓存策略

**多级缓存**：
```python
class MultiLevelCache:
    def __init__(self):
        self.l1_cache = {}  # 内存缓存（小且热）
        self.l2_cache = redis.Redis()  # Redis缓存（中等）
        self.l3_cache = {}  # 数据库缓存（大且冷）
    
    async def get(self, key: str) -> str:
        """获取缓存"""
        # L1: 内存
        if key in self.l1_cache:
            return self.l1_cache[key]
        
        # L2: Redis
        value = await self.l2_cache.get(key)
        if value:
            self.l1_cache[key] = value  # 回写L1
            return value
        
        # L3: 数据库
        value = await self.db.get(key)
        if value:
            await self.l2_cache.set(key, value, ex=3600)  # 回写L2
            self.l1_cache[key] = value  # 回写L1
            return value
        
        return None
    
    async def set(self, key: str, value: str, ttl: int = 3600):
        """设置缓存"""
        # 写入所有层级
        self.l1_cache[key] = value
        await self.l2_cache.set(key, value, ex=ttl)
        await self.db.set(key, value)
```

**语义缓存**：
```python
class SemanticCache:
    def __init__(self, embedding_model, threshold=0.95):
        self.embedding_model = embedding_model
        self.threshold = threshold
        self.cache = []  # [(embedding, response)]
    
    async def get(self, query: str) -> str:
        """基于语义相似度的缓存"""
        query_emb = self.embedding_model.encode(query)
        
        for cached_emb, response in self.cache:
            similarity = cosine_similarity([query_emb], [cached_emb])[0][0]
            
            if similarity >= self.threshold:
                return response  # 缓存命中
        
        return None
    
    async def set(self, query: str, response: str):
        """添加到缓存"""
        query_emb = self.embedding_model.encode(query)
        self.cache.append((query_emb, response))
        
        # 限制缓存大小
        if len(self.cache) > 1000:
            self.cache.pop(0)
```

```python
@app.post("/chat/stream")
async def chat_stream(request: LLMRequest):
```

```python
class PredictiveLoader:
    def __init__(self):
        self.patterns = {}
    
    def record(self, query: str, next_query: str):
        """记录查询模式"""
        if query not in self.patterns:
            self.patterns[query] = {}
        
        if next_query not in self.patterns[query]:
            self.patterns[query][next_query] = 0
        
        self.patterns[query][next_query] += 1
    
    def predict(self, query: str) -> str:
        """预测下一个查询"""
        if query in self.patterns:
            return max(self.patterns[query], key=self.patterns[query].get)
        return None
    
    async def preload(self, query: str):
        """预加载可能的下一个查询"""
        next_query = self.predict(query)
        if next_query:
            # 后台预生成
            asyncio.create_task(llm.generate(next_query))
```

### 7.1 Prompt注入防护

```python
class PromptInjectionDetector:
    def __init__(self):
        self.dangerous_patterns = [
            r"ignore previous instructions",
            r"忽略之前的指令",
            r"system:",
            r"<script>",
            r"rm -rf",
            r"DROP TABLE"
        ]
    
    def detect(self, query: str) -> bool:
        """检测Prompt注入"""
        for pattern in self.dangerous_patterns:
            if re.search(pattern, query, re.IGNORECASE):
                return True
        return False
    
    def sanitize(self, query: str) -> str:
        """清理输入"""
        # 移除特殊字符
        sanitized = re.sub(r'[<>{}]', '', query)
        return sanitized
```

## 9. LLM框架选型

### 9.1 LangChain

**核心概念**：
- **Chain**：串联多个处理步骤
- **Agent**：自主决策调用工具
- **Memory**：对话历史管理
- **Retriever**：检索器接口

```python
from langchain.chat_models import ChatOpenAI
from langchain.chains import ConversationalRetrievalChain
from langchain.memory import ConversationBufferMemory

# 创建RAG对话链
llm = ChatOpenAI(model="gpt-4")
memory = ConversationBufferMemory(memory_key="chat_history", return_messages=True)

qa_chain = ConversationalRetrievalChain.from_llm(
    llm=llm,
    retriever=vectorstore.as_retriever(),
    memory=memory
)

# 对话
response = qa_chain({"question": "什么是分布式事务？"})
```

**适用场景**：
- 快速原型开发
- RAG应用
- 简单的Agent场景

### 9.2 LangGraph

**核心概念**：
- **State**：全局状态
- **Node**：处理节点
- **Edge**：条件边（决定流转）

```python
from langgraph.graph import StateGraph, END
from typing import TypedDict

# 定义状态
class AgentState(TypedDict):
    messages: list
    next_action: str

# 创建图
workflow = StateGraph(AgentState)

# 添加节点
workflow.add_node("analyze", analyze_node)
workflow.add_node("search", search_node)
workflow.add_node("generate", generate_node)

# 添加边
workflow.add_edge("analyze", "search")
workflow.add_conditional_edges(
    "search",
    should_continue,  # 条件函数
    {"continue": "search", "generate": "generate"}
)
workflow.add_edge("generate", END)

# 编译运行
app = workflow.compile()
result = app.invoke({"messages": ["用户问题"], "next_action": ""})
```

**适用场景**：
- 复杂的多步骤Agent
- 需要条件分支的流程
- 状态管理复杂的应用

### 9.3 框架对比

| 特性 | LangChain | LangGraph | 直接调用API |
|------|-----------|-----------|-------------|
| 复杂度 | 中等 | 高 | 低 |
| 灵活性 | 一般 | 高 | 最高 |
| 学习曲线 | 中等 | 陡峭 | 平缓 |
| 适用场景 | 标准RAG、简单Agent | 复杂工作流 | 简单应用 |
| 可调试性 | 一般 | 好 | 最好 |
| 生产就绪 | 是 | 是 | 是 |

**选型建议**：
- 简单应用/学习 → 直接调用API
- 标准RAG → LangChain
- 复杂Agent/工作流 → LangGraph
- 极致性能要求 → 自研框架

### Q8: 语义缓存是什么？如何实现？

**答案**：
**语义缓存**：基于查询语义相似度的缓存，相似查询可复用结果。

**实现**：
```python
class SemanticCache:
    def __init__(self, threshold=0.95):
        self.cache = []  # [(embedding, response)]
        self.threshold = threshold
    
    def get(self, query):
        query_emb = embed(query)
        
        for cached_emb, response in self.cache:
            similarity = cosine_similarity(query_emb, cached_emb)
            if similarity >= self.threshold:
                return response  # 命中
        
        return None
```

**优势**：
- "什么是AI"和"AI是什么"可以命中同一缓存
- 大幅提高缓存命中率

**注意**：
- 阈值太低会导致错误命中
- 需要高质量Embedding模型
- 定期清理过期缓存

### Q13: 如何进行LLM应用的AB测试？

**答案**：

**测试对象**：
- 不同Prompt版本
- 不同模型
- 不同参数（temperature等）

**实现**：
```python
class ABTest:
    def __init__(self, variants, metric_fn):
        self.variants = variants  # {"A": prompt_a, "B": prompt_b}
        self.metric_fn = metric_fn
        self.results = {"A": [], "B": []}
    
    async def run(self, queries, n_per_query=5):
        for query in queries:
            for variant, prompt in self.variants.items():
                for _ in range(n_per_query):
                    response = await llm.generate(prompt.format(query=query))
                    score = await self.metric_fn(query, response)
                    self.results[variant].append(score)
        
        # 统计显著性
        return self.analyze()
```

**分析**：
- 计算各变体平均分
- t检验判断显著性
- 样本量要足够（至少30次/变体）
