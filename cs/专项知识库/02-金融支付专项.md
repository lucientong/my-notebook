## 1. 支付链路与幂等设计

### 1.1 典型支付流程

```
[客户端] → [API网关] → [支付服务] → [第三方支付] 
                              ↓
                          [回调通知]
                              ↓
                          [业务系统]（发货、充值、加积分）
```

**关键点**：
1. **请求幂等**：客户端重复提交
2. **回调幂等**：第三方重复回调
3. **补单幂等**：定时任务重复执行

### 1.2 订单状态机

```
created (已创建) → paying (支付中) → success (成功)
                          ↓
                      failed (失败)
```

**状态流转规则**：
- created → paying：发起支付
- paying → success：回调成功
- paying → failed：回调失败/超时
- **不允许**：success → paying（防止重复支付）

### 1.3 数据库设计

```sql
CREATE TABLE orders (
    order_id VARCHAR(64) PRIMARY KEY,
    user_id BIGINT NOT NULL,
    amount DECIMAL(10, 2) NOT NULL,
    status VARCHAR(20) NOT NULL,  -- 'created', 'paying', 'success', 'failed'
    pay_channel VARCHAR(20),      -- 'alipay', 'wechat', 'bank'
    third_party_order_id VARCHAR(64),
    callback_count INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_user_status (user_id, status, created_at),
    INDEX idx_third_party (third_party_order_id)
) ENGINE=InnoDB;

-- 支付日志表（审计）
CREATE TABLE payment_logs (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    order_id VARCHAR(64) NOT NULL,
    action VARCHAR(50) NOT NULL,  -- 'create', 'pay', 'callback', 'refund'
    before_status VARCHAR(20),
    after_status VARCHAR(20),
    request_data TEXT,
    response_data TEXT,
    operator VARCHAR(50),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_order_created (order_id, created_at)
) ENGINE=InnoDB;
```

```go
package payment

import (
    "context"
    "database/sql"
    "errors"
    "fmt"
    "time"
    "github.com/go-redis/redis/v8"
    "github.com/google/uuid"
)

var (
    ErrOrderNotFound    = errors.New("order not found")
    ErrOrderPaid        = errors.New("order already paid")
    ErrConcurrentPay    = errors.New("concurrent payment")
    ErrInvalidSignature = errors.New("invalid signature")
)

type PaymentService struct {
    db    *sql.DB
    redis *redis.Client
}

// CreateOrder 创建订单
func (s *PaymentService) CreateOrder(ctx context.Context, req *CreateOrderReq) (*Order, error) {
    orderID := uuid.New().String()  // 或使用雪花ID
    
    order := &Order{
        OrderID:   orderID,
        UserID:    req.UserID,
        Amount:    req.Amount,
        Status:    "created",
        PayChannel: req.PayChannel,
    }
    
    _, err := s.db.ExecContext(ctx,
        `INSERT INTO orders (order_id, user_id, amount, status, pay_channel)
         VALUES (?, ?, ?, ?, ?)`,
        order.OrderID, order.UserID, order.Amount, order.Status, order.PayChannel,
    )
    
    if err != nil {
        return nil, err
    }
    
    // 记录日志
    s.logAction(ctx, orderID, "create", "", "created", req)
    
    return order, nil
}

// Pay 发起支付（幂等）
func (s *PaymentService) Pay(ctx context.Context, orderID string) (*PayResp, error) {
    // 1. 分布式锁（防止并发支付）
    lockKey := fmt.Sprintf("lock:pay:%s", orderID)
    lock, err := s.redis.SetNX(ctx, lockKey, 1, 30*time.Second).Result()
    if err != nil || !lock {
        return nil, ErrConcurrentPay
    }
    defer s.redis.Del(ctx, lockKey)
    
    // 2. 查询订单状态
    order, err := s.getOrder(ctx, orderID)
    if err != nil {
        return nil, err
    }
    
    // 3. 状态校验（幂等核心）
    if order.Status == "success" {
        return &PayResp{OrderID: orderID, AlreadyPaid: true}, nil
    }
    if order.Status == "paying" {
        return nil, errors.New("order is paying")
    }
    if order.Status != "created" {
        return nil, errors.New("invalid order status")
    }
    
    // 4. 更新状态为paying
    _, err = s.db.ExecContext(ctx,
        "UPDATE orders SET status = 'paying', updated_at = NOW() WHERE order_id = ? AND status = 'created'",
        orderID,
    )
    if err != nil {
        return nil, err
    }
    
    s.logAction(ctx, orderID, "pay", "created", "paying", nil)
    
    // 5. 调用第三方支付
    thirdPartyOrderID, payURL, err := s.callThirdParty(ctx, order)
    if err != nil {
        // 支付失败，更新状态
        s.db.ExecContext(ctx,
            "UPDATE orders SET status = 'failed' WHERE order_id = ?",
            orderID,
        )
        s.logAction(ctx, orderID, "pay_failed", "paying", "failed", map[string]interface{}{"error": err.Error()})
        return nil, err
    }
    
    // 6. 保存第三方订单号
    _, err = s.db.ExecContext(ctx,
        "UPDATE orders SET third_party_order_id = ? WHERE order_id = ?",
        thirdPartyOrderID, orderID,
    )
    
    return &PayResp{
        OrderID:           orderID,
        ThirdPartyOrderID: thirdPartyOrderID,
        PayURL:            payURL,
    }, nil
}

// HandleCallback 处理支付回调（幂等 + 防重）
func (s *PaymentService) HandleCallback(ctx context.Context, callback *Callback) error {
    // 1. 验签（防止伪造）
    if !s.verifySignature(callback) {
        log.Warnf("invalid signature: %+v", callback)
        return ErrInvalidSignature
    }
    
    // 2. 分布式锁
    lockKey := fmt.Sprintf("lock:callback:%s", callback.OrderID)
    lock, err := s.redis.SetNX(ctx, lockKey, 1, 30*time.Second).Result()
    if err != nil || !lock {
        log.Warnf("concurrent callback: %s", callback.OrderID)
        return nil  // 并发回调，直接返回成功（第三方会重试）
    }
    defer s.redis.Del(ctx, lockKey)
    
    // 3. 查询订单
    order, err := s.getOrder(ctx, callback.OrderID)
    if err != nil {
        return err
    }
    
    // 4. 幂等校验
    if order.Status == "success" {
        log.Infof("order %s already paid", callback.OrderID)
        return nil  // 已支付成功，幂等返回
    }
    
    // 5. 更新回调次数
    _, err = s.db.ExecContext(ctx,
        "UPDATE orders SET callback_count = callback_count + 1 WHERE order_id = ?",
        callback.OrderID,
    )
    
    // 6. 根据回调结果更新订单状态
    if callback.PaySuccess {
        // 支付成功
        _, err = s.db.ExecContext(ctx,
            "UPDATE orders SET status = 'success', updated_at = NOW() WHERE order_id = ? AND status != 'success'",
            callback.OrderID,
        )
        if err != nil {
            return err
        }
        
        s.logAction(ctx, callback.OrderID, "callback_success", "paying", "success", callback)
        
        // 7. 异步处理业务逻辑（发货、充值等）
        go func() {
            if err := s.handleBusinessLogic(context.Background(), order); err != nil {
                log.Errorf("business logic failed: %v", err)
                // 记录到补偿队列
                s.pushToRetryQueue(order)
            }
        }()
    } else {
        // 支付失败
        _, err = s.db.ExecContext(ctx,
            "UPDATE orders SET status = 'failed' WHERE order_id = ?",
            callback.OrderID,
        )
        s.logAction(ctx, callback.OrderID, "callback_failed", "paying", "failed", callback)
    }
    
    return nil
}

// QueryThirdParty 主动查询第三方（补单逻辑）
func (s *PaymentService) QueryThirdParty(ctx context.Context, orderID string) error {
    order, err := s.getOrder(ctx, orderID)
    if err != nil {
        return err
    }
    
    // 只查询paying状态的订单
    if order.Status != "paying" {
        return nil
    }
    
    // 调用第三方查询接口
    result, err := s.queryThirdPartyOrder(ctx, order.ThirdPartyOrderID)
    if err != nil {
        return err
    }
    
    // 模拟回调处理
    callback := &Callback{
        OrderID:    orderID,
        PaySuccess: result.Status == "success",
    }
    
    return s.HandleCallback(ctx, callback)
}

// getOrder 查询订单
func (s *PaymentService) getOrder(ctx context.Context, orderID string) (*Order, error) {
    var order Order
    err := s.db.QueryRowContext(ctx,
        "SELECT order_id, user_id, amount, status, pay_channel, third_party_order_id FROM orders WHERE order_id = ?",
        orderID,
    ).Scan(&order.OrderID, &order.UserID, &order.Amount, &order.Status, &order.PayChannel, &order.ThirdPartyOrderID)
    
    if err == sql.ErrNoRows {
        return nil, ErrOrderNotFound
    }
    return &order, err
}

// verifySignature 验证签名
func (s *PaymentService) verifySignature(callback *Callback) bool {
    // 实际项目中需要根据第三方文档实现
    // 示例：MD5(order_id + amount + timestamp + secret_key)
    expectedSign := md5Sum(fmt.Sprintf("%s%d%d%s", 
        callback.OrderID, callback.Amount, callback.Timestamp, s.secretKey))
    return callback.Sign == expectedSign
}

// logAction 记录操作日志
func (s *PaymentService) logAction(ctx context.Context, orderID, action, beforeStatus, afterStatus string, data interface{}) {
    dataJSON, _ := json.Marshal(data)
    s.db.ExecContext(ctx,
        `INSERT INTO payment_logs (order_id, action, before_status, after_status, request_data)
         VALUES (?, ?, ?, ?, ?)`,
        orderID, action, beforeStatus, afterStatus, dataJSON,
    )
}
```

```go
func (s *PaymentService) CompensateOrders() {
    ticker := time.NewTicker(1 * time.Minute)
    defer ticker.Stop()
    
    for range ticker.C {
        // 查询超过5分钟仍在paying状态的订单
        rows, err := s.db.Query(
            `SELECT order_id FROM orders 
             WHERE status = 'paying' AND updated_at < DATE_SUB(NOW(), INTERVAL 5 MINUTE)
             LIMIT 100`,
        )
        if err != nil {
            log.Error("query compensate orders failed:", err)
            continue
        }
        
        for rows.Next() {
            var orderID string
            rows.Scan(&orderID)
            
            // 主动查询第三方
            if err := s.QueryThirdParty(context.Background(), orderID); err != nil {
                log.Errorf("compensate order %s failed: %v", orderID, err)
            } else {
                log.Infof("compensated order %s", orderID)
            }
        }
        rows.Close()
    }
}
```

## 2. 分布式事务

### 2.1 场景描述

支付成功后需要：
1. 扣减库存
2. 增加积分
3. 发送短信

如果任何一步失败，如何保证一致性？

### 2.2 方案对比

| 方案 | 一致性 | 性能 | 复杂度 | 适用场景 |
|------|--------|------|--------|----------|
| **2PC** | 强一致 | 差（阻塞） | 中 | 不推荐 |
| **TCC** | 最终一致 | 好 | 高 | 核心交易 |
| **SAGA** | 最终一致 | 好 | 中 | 非核心业务 |
| **本地消息表** | 最终一致 | 好 | 低 | **推荐** |
| **RocketMQ事务消息** | 最终一致 | 好 | 低 | 推荐 |

### 2.3 本地消息表（推荐）

**原理**：
1. 支付成功时，在**同一事务**中写入本地消息表
2. 定时扫描消息表，发送到MQ
3. 下游消费MQ，执行业务逻辑
4. 消费成功后标记消息为已处理

**数据库设计**：
```sql
CREATE TABLE local_messages (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    order_id VARCHAR(64) NOT NULL,
    message_type VARCHAR(50) NOT NULL,  -- 'add_points', 'deduct_stock', 'send_sms'
    message_body TEXT NOT NULL,
    status VARCHAR(20) NOT NULL,  -- 'pending', 'sent', 'failed'
    retry_count INT DEFAULT 0,
    max_retry INT DEFAULT 3,
    next_retry_time TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_status_retry (status, next_retry_time)
) ENGINE=InnoDB;
```

**Go实现**：
```go
// OnPaymentSuccess 支付成功后的处理
func (s *PaymentService) OnPaymentSuccess(ctx context.Context, order *Order) error {
    // 开启事务
    tx, err := s.db.BeginTx(ctx, nil)
    if err != nil {
        return err
    }
    defer tx.Rollback()
    
    // 1. 更新订单状态
    _, err = tx.ExecContext(ctx,
        "UPDATE orders SET status = 'success' WHERE order_id = ?",
        order.OrderID,
    )
    if err != nil {
        return err
    }
    
    // 2. 写入本地消息表（在同一事务中）
    messages := []struct {
        Type string
        Body map[string]interface{}
    }{
        {"add_points", map[string]interface{}{"user_id": order.UserID, "points": 100}},
        {"deduct_stock", map[string]interface{}{"product_id": 123, "count": 1}},
        {"send_sms", map[string]interface{}{"phone": order.Phone, "template": "pay_success"}},
    }
    
    for _, msg := range messages {
        body, _ := json.Marshal(msg.Body)
        _, err = tx.ExecContext(ctx,
            `INSERT INTO local_messages (order_id, message_type, message_body, status, next_retry_time)
             VALUES (?, ?, ?, 'pending', NOW())`,
            order.OrderID, msg.Type, body,
        )
        if err != nil {
            return err
        }
    }
    
    // 3. 提交事务
    return tx.Commit()
}

// SendPendingMessages 定时发送待发送的消息
func (s *PaymentService) SendPendingMessages() {
    ticker := time.NewTicker(1 * time.Second)
    defer ticker.Stop()
    
    for range ticker.C {
        // 查询待发送的消息
        rows, err := s.db.Query(
            `SELECT id, order_id, message_type, message_body 
             FROM local_messages 
             WHERE status = 'pending' AND next_retry_time <= NOW()
             LIMIT 100`,
        )
        if err != nil {
            log.Error(err)
            continue
        }
        
        for rows.Next() {
            var id int64
            var orderID, msgType, msgBody string
            rows.Scan(&id, &orderID, &msgType, &msgBody)
            
            // 发送到MQ
            err = s.mq.Publish(msgType, msgBody)
            if err != nil {
                // 发送失败，增加重试次数
                s.db.Exec(
                    `UPDATE local_messages 
                     SET retry_count = retry_count + 1,
                         next_retry_time = DATE_ADD(NOW(), INTERVAL POW(2, retry_count) SECOND),
                         status = IF(retry_count >= max_retry, 'failed', 'pending')
                     WHERE id = ?`,
                    id,
                )
                log.Errorf("send message %d failed: %v", id, err)
            } else {
                // 发送成功，标记为sent
                s.db.Exec(
                    "UPDATE local_messages SET status = 'sent', updated_at = NOW() WHERE id = ?",
                    id,
                )
                log.Infof("sent message %d", id)
            }
        }
        rows.Close()
    }
}
```

### 2.4 SAGA模式

**适用场景**：长事务，多个步骤，允许最终一致性

**核心思想**：
- 每个步骤定义**正向操作**和**补偿操作**
- 正向操作失败时，执行补偿操作回滚

**示例**：
```go
type SagaOrchestrator struct {
    steps []*SagaStep
}

type SagaStep struct {
    Name       string
    Forward    func(ctx context.Context) error  // 正向操作
    Compensate func(ctx context.Context) error  // 补偿操作
}

// Execute 执行SAGA
func (s *SagaOrchestrator) Execute(ctx context.Context) error {
    var executedSteps []*SagaStep
    
    // 顺序执行每个步骤
    for _, step := range s.steps {
        if err := step.Forward(ctx); err != nil {
            log.Errorf("step %s failed: %v", step.Name, err)
            
            // 反向补偿
            for i := len(executedSteps) - 1; i >= 0; i-- {
                if err := executedSteps[i].Compensate(ctx); err != nil {
                    log.Errorf("compensate step %s failed: %v", executedSteps[i].Name, err)
                }
            }
            
            return err
        }
        executedSteps = append(executedSteps, step)
    }
    
    return nil
}

// 使用示例
func (s *PaymentService) ProcessOrderWithSaga(ctx context.Context, order *Order) error {
    saga := &SagaOrchestrator{
        steps: []*SagaStep{
            {
                Name: "deduct_stock",
                Forward: func(ctx context.Context) error {
                    return s.stockService.Deduct(ctx, order.ProductID, 1)
                },
                Compensate: func(ctx context.Context) error {
                    return s.stockService.Add(ctx, order.ProductID, 1)
                },
            },
            {
                Name: "add_points",
                Forward: func(ctx context.Context) error {
                    return s.pointsService.Add(ctx, order.UserID, 100)
                },
                Compensate: func(ctx context.Context) error {
                    return s.pointsService.Deduct(ctx, order.UserID, 100)
                },
            },
            {
                Name: "send_sms",
                Forward: func(ctx context.Context) error {
                    return s.smsService.Send(ctx, order.Phone, "支付成功")
                },
                Compensate: func(ctx context.Context) error {
                    // 短信无法回滚，记录日志即可
                    return nil
                },
            },
        },
    }
    
    return saga.Execute(ctx)
}
```

### 2.5 2PC（两阶段提交）底层原理⭐⭐⭐

**为什么需要了解2PC？**
- 面试高频：2PC的两个阶段是什么？为什么性能差？
- 原理理解：分布式强一致性的经典方案
- 方案对比：理解为什么大厂不用2PC

---

#### 2PC工作原理

**两个阶段**：

```
阶段1：准备阶段（Prepare）
协调者 → 所有参与者："准备提交事务"
参与者执行事务，但不提交，记录redo/undo日志
参与者 → 协调者："准备就绪"或"准备失败"

阶段2：提交阶段（Commit）
情况1：所有参与者都准备就绪
  协调者 → 所有参与者："提交事务"
  参与者提交事务
  参与者 → 协调者："提交成功"

情况2：任何一个参与者准备失败
  协调者 → 所有参与者："回滚事务"
  参与者回滚事务
  参与者 → 协调者："回滚成功"
```

**时序图**：

```
协调者        参与者A        参与者B        参与者C
  |               |              |              |
  |--准备请求---->|              |              |
  |               |--准备请求--->|              |
  |               |              |--准备请求--->|
  |               |              |              |
  |<--准备就绪----|              |              |
  |               |<--准备就绪--|              |
  |               |              |<--准备就绪--|
  |               |              |              |
  |--提交请求---->|              |              |
  |               |--提交请求--->|              |
  |               |              |--提交请求--->|
  |               |              |              |
  |<--提交成功----|              |              |
  |               |<--提交成功--|              |
  |               |              |<--提交成功--|
```

#### 2PC的Go实现

```go
package twopc

import (
    "context"
    "errors"
    "fmt"
    "sync"
    "time"
)

var (
    ErrPrepareFailed = errors.New("prepare failed")
    ErrCommitFailed  = errors.New("commit failed")
    ErrRollbackFailed = errors.New("rollback failed")
)

// Coordinator 协调者
type Coordinator struct {
    participants []Participant
    timeout      time.Duration
}

// Participant 参与者接口
type Participant interface {
    Prepare(ctx context.Context, txID string) error
    Commit(ctx context.Context, txID string) error
    Rollback(ctx context.Context, txID string) error
}

// Execute 执行2PC
func (c *Coordinator) Execute(ctx context.Context, txID string) error {
    // 阶段1：准备阶段
    if err := c.preparePhase(ctx, txID); err != nil {
        // 准备失败，回滚所有参与者
        c.rollbackPhase(ctx, txID)
        return err
    }
    
    // 阶段2：提交阶段
    if err := c.commitPhase(ctx, txID); err != nil {
        // 提交失败（理论上不应该发生）
        return err
    }
    
    return nil
}

// preparePhase 准备阶段（并发执行）
func (c *Coordinator) preparePhase(ctx context.Context, txID string) error {
    type result struct {
        idx int
        err error
    }
    
    resultChan := make(chan result, len(c.participants))
    ctx, cancel := context.WithTimeout(ctx, c.timeout)
    defer cancel()
    
    // 并发向所有参与者发送准备请求
    for i, p := range c.participants {
        go func(idx int, participant Participant) {
            err := participant.Prepare(ctx, txID)
            resultChan <- result{idx: idx, err: err}
        }(i, p)
    }
    
    // 等待所有参与者响应
    for i := 0; i < len(c.participants); i++ {
        select {
        case res := <-resultChan:
            if res.err != nil {
                log.Errorf("participant %d prepare failed: %v", res.idx, res.err)
                return ErrPrepareFailed
            }
        case <-ctx.Done():
            log.Error("prepare phase timeout")
            return ErrPrepareFailed
        }
    }
    
    log.Info("all participants prepared successfully")
    return nil
}

// commitPhase 提交阶段
func (c *Coordinator) commitPhase(ctx context.Context, txID string) error {
    var wg sync.WaitGroup
    errChan := make(chan error, len(c.participants))
    
    // 并发向所有参与者发送提交请求
    for i, p := range c.participants {
        wg.Add(1)
        go func(idx int, participant Participant) {
            defer wg.Done()
            if err := participant.Commit(ctx, txID); err != nil {
                log.Errorf("participant %d commit failed: %v", idx, err)
                errChan <- err
            }
        }(i, p)
    }
    
    wg.Wait()
    close(errChan)
    
    // 检查是否有提交失败
    if len(errChan) > 0 {
        return ErrCommitFailed
    }
    
    log.Info("all participants committed successfully")
    return nil
}

// rollbackPhase 回滚阶段
func (c *Coordinator) rollbackPhase(ctx context.Context, txID string) {
    var wg sync.WaitGroup
    
    for i, p := range c.participants {
        wg.Add(1)
        go func(idx int, participant Participant) {
            defer wg.Done()
            if err := participant.Rollback(ctx, txID); err != nil {
                log.Errorf("participant %d rollback failed: %v", idx, err)
            }
        }(i, p)
    }
    
    wg.Wait()
    log.Info("rollback phase completed")
}
```

```go
// DBParticipant 数据库参与者
type DBParticipant struct {
    db         *sql.DB
    preparedTx map[string]*sql.Tx  // txID -> *sql.Tx
    mu         sync.Mutex
}

// Prepare 准备阶段
func (p *DBParticipant) Prepare(ctx context.Context, txID string) error {
    p.mu.Lock()
    defer p.mu.Unlock()
    
    // 开启事务
    tx, err := p.db.BeginTx(ctx, nil)
    if err != nil {
        return err
    }
    
    // 执行业务逻辑（如扣减库存）
    _, err = tx.ExecContext(ctx,
        "UPDATE stock SET count = count - 1 WHERE product_id = ? AND count > 0",
        123,
    )
    if err != nil {
        tx.Rollback()
        return err
    }
    
    // 保存事务（不提交）
    p.preparedTx[txID] = tx
    
    log.Infof("prepared txID %s", txID)
    return nil
}

// Commit 提交阶段
func (p *DBParticipant) Commit(ctx context.Context, txID string) error {
    p.mu.Lock()
    defer p.mu.Unlock()
    
    tx, exists := p.preparedTx[txID]
    if !exists {
        return errors.New("tx not found")
    }
    
    // 提交事务
    if err := tx.Commit(); err != nil {
        return err
    }
    
    delete(p.preparedTx, txID)
    log.Infof("committed txID %s", txID)
    return nil
}

// Rollback 回滚
func (p *DBParticipant) Rollback(ctx context.Context, txID string) error {
    p.mu.Lock()
    defer p.mu.Unlock()
    
    tx, exists := p.preparedTx[txID]
    if !exists {
        return nil  // 已回滚
    }
    
    if err := tx.Rollback(); err != nil {
        return err
    }
    
    delete(p.preparedTx, txID)
    log.Infof("rolled back txID %s", txID)
    return nil
}
```

#### 2PC的致命问题⭐⭐⭐

**问题1：同步阻塞**

```
准备阶段：
- 参与者执行事务但不提交
- 持有数据库锁（行锁/表锁）
- 等待协调者的提交命令

如果协调者宕机：
- 参与者一直等待（持有锁不释放）
- 其他事务被阻塞
- 导致整个系统不可用

示例：
t=0:   参与者A准备就绪，持有锁
t=1:   参与者B准备就绪，持有锁
t=2:   协调者宕机
t=3-∞: 参与者A/B一直等待，锁不释放

影响：
- 其他事务无法访问被锁的数据
- 系统吞吐量下降
- 用户体验差
```

**问题2：单点故障**

```
协调者宕机后：
- 参与者不知道该提交还是回滚
- 即使选举新协调者，也不知道原协调者的决策
- 需要人工介入

解决方案：
- 协调者持久化日志（记录决策）
- 新协调者读取日志，继续执行
- 但日志也可能丢失
```

**问题3：数据不一致**

```
场景：提交阶段部分成功
协调者 → 参与者A："提交"（成功）
协调者 → 参与者B："提交"（网络故障，未到达）
协调者 → 参与者C："提交"（成功）

结果：
- 参与者A/C已提交
- 参与者B未提交（仍持有锁）

风险：
- 数据不一致
- 参与者B可能超时后自动回滚
```

#### 为什么大厂不用2PC？

| 问题 | 影响 | 是否可接受 |
|------|------|-----------|
| **同步阻塞** | 性能差（RT高） | ❌ 高并发场景不可接受 |
| **单点故障** | 可用性差 | ❌ 需要高可用 |
| **数据不一致** | 可能出现不一致 | ❌ 金融场景不可接受 |

**替代方案**：
- TCC（Try-Confirm-Cancel）
- SAGA
- 本地消息表
- 事务消息（RocketMQ）

### 2.6 TCC（Try-Confirm-Cancel）底层原理⭐⭐⭐

**TCC vs 2PC的改进**：

| 对比项 | 2PC | TCC |
|--------|-----|-----|
| **阻塞** | 阻塞（持有锁） | 非阻塞（预留资源） |
| **性能** | 差 | 好 |
| **业务侵入** | 低 | 高（需要实现三个接口） |
| **一致性** | 强一致 | 最终一致 |

---

#### TCC工作原理

**三个阶段**：

```
阶段1：Try（尝试）
- 检查资源是否充足
- 预留资源（不实际扣减）
- 记录预留日志

阶段2：Confirm（确认）
- 使用预留的资源
- 完成实际扣减
- 删除预留日志

阶段3：Cancel（取消）
- 释放预留的资源
- 回滚操作
- 删除预留日志
```

**示例：转账**

```
Try阶段：
- 账户A：检查余额≥100，冻结100元（余额不变，冻结金额+100）
- 账户B：预留+100元（余额不变，预留金额+100）

Confirm阶段：
- 账户A：余额-100，冻结金额-100
- 账户B：余额+100，预留金额-100

Cancel阶段：
- 账户A：冻结金额-100（恢复）
- 账户B：预留金额-100（恢复）
```

```go
package tcc

import (
    "context"
    "database/sql"
    "encoding/json"
    "errors"
    "fmt"
)

type TCCCoordinator struct {
    db *sql.DB
}

type TCCParticipant interface {
    Try(ctx context.Context, txID string, params map[string]interface{}) error
    Confirm(ctx context.Context, txID string) error
    Cancel(ctx context.Context, txID string) error
}

// Transfer TCC转账
func (c *TCCCoordinator) Transfer(ctx context.Context, from, to int64, amount float64) error {
    txID := generateTxID()
    
    // 创建事务记录
    contextJSON, _ := json.Marshal(map[string]interface{}{
        "from":   from,
        "to":     to,
        "amount": amount,
    })
    _, err := c.db.ExecContext(ctx,
        "INSERT INTO tcc_transactions (tx_id, status, context) VALUES (?, 'trying', ?)",
        txID, contextJSON,
    )
    if err != nil {
        return err
    }
    
    // Try阶段
    if err := c.tryPhase(ctx, txID, from, to, amount); err != nil {
        // Try失败，执行Cancel
        c.cancelPhase(ctx, txID)
        return err
    }
    
    // Confirm阶段
    if err := c.confirmPhase(ctx, txID); err != nil {
        // Confirm失败，执行Cancel
        c.cancelPhase(ctx, txID)
        return err
    }
    
    return nil
}

// tryPhase Try阶段
func (c *TCCCoordinator) tryPhase(ctx context.Context, txID string, from, to int64, amount float64) error {
    tx, err := c.db.BeginTx(ctx, nil)
    if err != nil {
        return err
    }
    defer tx.Rollback()
    
    // 1. 账户A：冻结金额
    result, err := tx.ExecContext(ctx,
        `UPDATE accounts 
         SET frozen_amount = frozen_amount + ?,
             balance = balance - ?
         WHERE account_id = ? AND balance >= ?`,
        amount, amount, from, amount,
    )
    if err != nil {
        return err
    }
    
    affected, _ := result.RowsAffected()
    if affected == 0 {
        return errors.New("insufficient balance")
    }
    
    // 2. 账户B：预留金额
    _, err = tx.ExecContext(ctx,
        "UPDATE accounts SET reserved_amount = reserved_amount + ? WHERE account_id = ?",
        amount, to,
    )
    if err != nil {
        return err
    }
    
    // 3. 记录操作
    _, err = tx.ExecContext(ctx,
        "INSERT INTO tcc_actions (tx_id, action_type, account_id, amount, status) VALUES (?, 'debit', ?, ?, 'tried')",
        txID, from, amount,
    )
    if err != nil {
        return err
    }
    
    _, err = tx.ExecContext(ctx,
        "INSERT INTO tcc_actions (tx_id, action_type, account_id, amount, status) VALUES (?, 'credit', ?, ?, 'tried')",
        txID, to, amount,
    )
    if err != nil {
        return err
    }
    
    return tx.Commit()
}

// confirmPhase Confirm阶段
func (c *TCCCoordinator) confirmPhase(ctx context.Context, txID string) error {
    tx, err := c.db.BeginTx(ctx, nil)
    if err != nil {
        return err
    }
    defer tx.Rollback()
    
    // 查询Try阶段的操作
    rows, err := tx.QueryContext(ctx,
        "SELECT action_type, account_id, amount FROM tcc_actions WHERE tx_id = ? AND status = 'tried'",
        txID,
    )
    if err != nil {
        return err
    }
    defer rows.Close()
    
    for rows.Next() {
        var actionType string
        var accountID int64
        var amount float64
        rows.Scan(&actionType, &accountID, &amount)
        
        if actionType == "debit" {
            // 账户A：解冻（冻结金额已在Try时扣减余额）
            _, err = tx.ExecContext(ctx,
                "UPDATE accounts SET frozen_amount = frozen_amount - ? WHERE account_id = ?",
                amount, accountID,
            )
        } else {
            // 账户B：确认（预留金额转为实际余额）
            _, err = tx.ExecContext(ctx,
                "UPDATE accounts SET balance = balance + ?, reserved_amount = reserved_amount - ? WHERE account_id = ?",
                amount, amount, accountID,
            )
        }
        
        if err != nil {
            return err
        }
    }
    
    // 更新状态
    _, err = tx.ExecContext(ctx,
        "UPDATE tcc_actions SET status = 'confirmed' WHERE tx_id = ?",
        txID,
    )
    if err != nil {
        return err
    }
    
    _, err = tx.ExecContext(ctx,
        "UPDATE tcc_transactions SET status = 'confirmed' WHERE tx_id = ?",
        txID,
    )
    if err != nil {
        return err
    }
    
    return tx.Commit()
}

// cancelPhase Cancel阶段
func (c *TCCCoordinator) cancelPhase(ctx context.Context, txID string) error {
    tx, err := c.db.BeginTx(ctx, nil)
    if err != nil {
        return err
    }
    defer tx.Rollback()
    
    // 查询Try阶段的操作
    rows, err := tx.QueryContext(ctx,
        "SELECT action_type, account_id, amount FROM tcc_actions WHERE tx_id = ? AND status = 'tried'",
        txID,
    )
    if err != nil {
        return err
    }
    defer rows.Close()
    
    for rows.Next() {
        var actionType string
        var accountID int64
        var amount float64
        rows.Scan(&actionType, &accountID, &amount)
        
        if actionType == "debit" {
            // 账户A：恢复余额 + 解冻
            _, err = tx.ExecContext(ctx,
                "UPDATE accounts SET balance = balance + ?, frozen_amount = frozen_amount - ? WHERE account_id = ?",
                amount, amount, accountID,
            )
        } else {
            // 账户B：取消预留
            _, err = tx.ExecContext(ctx,
                "UPDATE accounts SET reserved_amount = reserved_amount - ? WHERE account_id = ?",
                amount, accountID,
            )
        }
        
        if err != nil {
            return err
        }
    }
    
    // 更新状态
    _, err = tx.ExecContext(ctx,
        "UPDATE tcc_actions SET status = 'cancelled' WHERE tx_id = ?",
        txID,
    )
    if err != nil {
        return err
    }
    
    _, err = tx.ExecContext(ctx,
        "UPDATE tcc_transactions SET status = 'cancelled' WHERE tx_id = ?",
        txID,
    )
    if err != nil {
        return err
    }
    
    return tx.Commit()
}
```

#### TCC的关键问题⭐⭐

**问题1：幂等性**

```
场景：Confirm请求重复发送
- 第1次：冻结金额-100（成功）
- 第2次：冻结金额-100（错误！变成-100）

解决方案：
- 检查tcc_actions状态（status='confirmed'则跳过）
- 或使用唯一约束（tx_id + account_id）
```

**问题2：空回滚**

```
场景：Try请求超时，但实际已执行
- 协调者：Try超时，执行Cancel
- 参与者：收到Cancel，但Try还没执行

风险：Cancel空回滚（Try的预留资源不存在）

解决方案：
- 检查tcc_actions是否有tried记录
- 如果没有，记录"空回滚"，不执行实际操作
```

**问题3：资源悬挂**

```
场景：Cancel先于Try到达
- 协调者：Try超时，发送Cancel
- 参与者：先收到Cancel（空回滚），后收到Try

风险：Try预留的资源无法释放（悬挂）

解决方案：
- Try时检查是否有"空回滚"记录
- 如果有，拒绝Try（或立即Cancel）
```

```go
// TryWithProtection Try阶段（防悬挂）
func (c *TCCCoordinator) TryWithProtection(ctx context.Context, txID string, from, to int64, amount float64) error {
    tx, err := c.db.BeginTx(ctx, nil)
    if err != nil {
        return err
    }
    defer tx.Rollback()
    
    // 1. 检查是否已有Cancel记录（防悬挂）
    var cancelCount int
    tx.QueryRowContext(ctx,
        "SELECT COUNT(*) FROM tcc_actions WHERE tx_id = ? AND status = 'cancelled'",
        txID,
    ).Scan(&cancelCount)
    
    if cancelCount > 0 {
        log.Warnf("tx %s already cancelled, reject try", txID)
        return errors.New("transaction already cancelled")
    }
    
    // 2. 检查是否已Try（幂等）
    var triedCount int
    tx.QueryRowContext(ctx,
        "SELECT COUNT(*) FROM tcc_actions WHERE tx_id = ? AND status = 'tried'",
        txID,
    ).Scan(&triedCount)
    
    if triedCount > 0 {
        log.Infof("tx %s already tried, skip", txID)
        return nil
    }
    
    // 3. 执行Try逻辑（同上）
    // ...
    
    return tx.Commit()
}

// CancelWithProtection Cancel阶段（防空回滚）
func (c *TCCCoordinator) CancelWithProtection(ctx context.Context, txID string) error {
    tx, err := c.db.BeginTx(ctx, nil)
    if err != nil {
        return err
    }
    defer tx.Rollback()
    
    // 1. 检查是否已Cancel（幂等）
    var cancelCount int
    tx.QueryRowContext(ctx,
        "SELECT COUNT(*) FROM tcc_actions WHERE tx_id = ? AND status = 'cancelled'",
        txID,
    ).Scan(&cancelCount)
    
    if cancelCount > 0 {
        log.Infof("tx %s already cancelled, skip", txID)
        return nil
    }
    
    // 2. 检查是否有Try记录
    var triedCount int
    tx.QueryRowContext(ctx,
        "SELECT COUNT(*) FROM tcc_actions WHERE tx_id = ? AND status = 'tried'",
        txID,
    ).Scan(&triedCount)
    
    if triedCount == 0 {
        // 空回滚：记录Cancel，但不执行实际操作
        log.Warnf("tx %s empty cancel", txID)
        _, err = tx.ExecContext(ctx,
            "INSERT INTO tcc_actions (tx_id, action_type, account_id, amount, status) VALUES (?, 'empty_cancel', 0, 0, 'cancelled')",
            txID,
        )
        if err != nil {
            return err
        }
        
        _, err = tx.ExecContext(ctx,
            "UPDATE tcc_transactions SET status = 'cancelled' WHERE tx_id = ?",
            txID,
        )
        return tx.Commit()
    }
    
    // 3. 执行Cancel逻辑（同上）
    // ...
    
    return tx.Commit()
}
```

#### TCC vs SAGA vs 本地消息表选择指南⭐⭐⭐

| 方案 | 一致性 | 性能 | 复杂度 | 适用场景 |
|------|--------|------|--------|----------|
| **TCC** | 最终一致 | 高 | 高（三个接口+防悬挂） | 核心金融交易（转账、支付） |
| **SAGA** | 最终一致 | 高 | 中（补偿逻辑） | 长流程（订单、工单） |
| **本地消息表** | 最终一致 | 高 | 低 | **大部分场景**（推荐） |
| **2PC** | 强一致 | 低（阻塞） | 中 | 不推荐 |

**决策树**：

```
是否需要强一致性？
├─ 是 → 考虑TCC（但通常不需要）
└─ 否 → 继续

是否是核心金融交易（如转账）？
├─ 是 → TCC
└─ 否 → 继续

是否是长流程（多个步骤）？
├─ 是 → SAGA
└─ 否 → 本地消息表
```

**阿里/腾讯的实践**：
- **核心交易**：TCC（如支付宝转账）
- **非核心交易**：本地消息表 + RocketMQ事务消息
- **长流程**：SAGA（如订单履约）

### 3.1 为什么需要对账？

**目标**：保证内外账一致
- **内账**：我们的订单系统
- **外账**：第三方支付渠道

**对账时机**：T+1（次日凌晨）

### 3.2 对账流程

```
1. 下载对账文件（第三方提供）
2. 解析对账文件
3. 与内部订单比对
4. 生成差异报告
5. 人工/自动处理差异
```

### 3.3 差异类型

| 差异类型 | 说明 | 可能原因 | 处理方式 |
|----------|------|----------|----------|
| **长款** | 第三方有，我们没有 | 回调丢失 | 补单 |
| **短款** | 我们有，第三方没有 | 下单失败 | 退款 |
| **金额不符** | 金额对不上 | 数据错误 | 人工核实 |

### 3.4 Go实现

```go
package reconcile

import (
    "context"
    "encoding/csv"
    "os"
)

type ReconcileService struct {
    db *sql.DB
}

// Reconcile 执行对账
func (s *ReconcileService) Reconcile(date string) (*ReconcileReport, error) {
    // 1. 下载对账文件
    file, err := s.downloadReconcileFile(date)
    if err != nil {
        return nil, err
    }
    defer file.Close()
    
    // 2. 解析对账文件
    thirdPartyOrders, err := s.parseReconcileFile(file)
    if err != nil {
        return nil, err
    }
    
    // 3. 查询内部订单
    internalOrders, err := s.getInternalOrders(date)
    if err != nil {
        return nil, err
    }
    
    // 4. 比对
    report := &ReconcileReport{Date: date}
    
    // 构建第三方订单Map
    thirdPartyMap := make(map[string]*ThirdPartyOrder)
    for _, order := range thirdPartyOrders {
        thirdPartyMap[order.OrderID] = order
    }
    
    // 检查内部订单
    for _, internal := range internalOrders {
        external, exists := thirdPartyMap[internal.OrderID]
        if !exists {
            // 短款：我们有，第三方没有
            report.ShortOrders = append(report.ShortOrders, internal)
        } else if internal.Amount != external.Amount {
            // 金额不符
            report.AmountMismatch = append(report.AmountMismatch, &MismatchOrder{
                OrderID:        internal.OrderID,
                InternalAmount: internal.Amount,
                ExternalAmount: external.Amount,
            })
        } else {
            // 匹配成功
            report.MatchedCount++
        }
        delete(thirdPartyMap, internal.OrderID)
    }
    
    // 剩余的是长款：第三方有，我们没有
    for _, external := range thirdPartyMap {
        report.LongOrders = append(report.LongOrders, external)
    }
    
    // 5. 保存对账报告
    s.saveReport(report)
    
    return report, nil
}

// parseReconcileFile 解析对账文件（CSV格式）
func (s *ReconcileService) parseReconcileFile(file *os.File) ([]*ThirdPartyOrder, error) {
    reader := csv.NewReader(file)
    records, err := reader.ReadAll()
    if err != nil {
        return nil, err
    }
    
    var orders []*ThirdPartyOrder
    for i, record := range records {
        if i == 0 {
            continue  // 跳过表头
        }
        // CSV格式：订单号,金额,交易时间,状态
        orders = append(orders, &ThirdPartyOrder{
            OrderID: record[0],
            Amount:  parseAmount(record[1]),
            PayTime: parseTime(record[2]),
            Status:  record[3],
        })
    }
    
    return orders, nil
}

// getInternalOrders 查询内部订单
func (s *ReconcileService) getInternalOrders(date string) ([]*InternalOrder, error) {
    rows, err := s.db.Query(
        `SELECT order_id, amount, updated_at 
         FROM orders 
         WHERE status = 'success' AND DATE(updated_at) = ?`,
        date,
    )
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    
    var orders []*InternalOrder
    for rows.Next() {
        var order InternalOrder
        rows.Scan(&order.OrderID, &order.Amount, &order.PayTime)
        orders = append(orders, &order)
    }
    
    return orders, nil
}

// HandleLongOrders 处理长款（补单）
func (s *ReconcileService) HandleLongOrders(orders []*ThirdPartyOrder) {
    for _, order := range orders {
        // 1. 再次确认是否真的没有
        exists, _ := s.checkOrderExists(order.OrderID)
        if exists {
            log.Infof("order %s already exists, skip", order.OrderID)
            continue
        }
        
        // 2. 补单
        if err := s.补单(order); err != nil {
            log.Errorf("compensate order %s failed: %v", order.OrderID, err)
        } else {
            log.Infof("compensated order %s", order.OrderID)
        }
    }
}
```

## 4. 账本设计

### 4.1 核心原则

1. **只增不改**：所有记录只插入不更新
2. **借贷平衡**：sum(借) = sum(贷)
3. **可审计**：每笔记录都有操作人、时间、原因

### 4.2 数据库设计

```sql
CREATE TABLE ledger (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    account_id BIGINT NOT NULL,
    order_id VARCHAR(64),
    amount DECIMAL(10, 2) NOT NULL,
    balance DECIMAL(10, 2) NOT NULL,  -- 余额快照
    type VARCHAR(20) NOT NULL,  -- 'debit'(借) or 'credit'(贷)
    reason VARCHAR(100),
    operator_id BIGINT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_account_created (account_id, created_at),
    INDEX idx_order (order_id)
) ENGINE=InnoDB;
```

### 4.3 Go实现

```go
// Transfer 转账（双记账）
func (s *LedgerService) Transfer(ctx context.Context, from, to int64, amount decimal.Decimal, reason string) error {
    if amount.LessThanOrEqual(decimal.Zero) {
        return errors.New("invalid amount")
    }
    
    tx, err := s.db.BeginTx(ctx, nil)
    if err != nil {
        return err
    }
    defer tx.Rollback()
    
    // 1. 扣减from账户（借方）
    _, err = tx.ExecContext(ctx,
        "INSERT INTO ledger (account_id, amount, type, reason) VALUES (?, ?, 'debit', ?)",
        from, amount.Neg(), reason,
    )
    if err != nil {
        return err
    }
    
    // 2. 增加to账户（贷方）
    _, err = tx.ExecContext(ctx,
        "INSERT INTO ledger (account_id, amount, type, reason) VALUES (?, ?, 'credit', ?)",
        to, amount, reason,
    )
    if err != nil {
        return err
    }
    
    return tx.Commit()
}

// CheckBalance 校验借贷平衡
func (s *LedgerService) CheckBalance() error {
    var debitSum, creditSum decimal.Decimal
    
    s.db.QueryRow("SELECT SUM(amount) FROM ledger WHERE type = 'debit'").Scan(&debitSum)
    s.db.QueryRow("SELECT SUM(amount) FROM ledger WHERE type = 'credit'").Scan(&creditSum)
    
    if !debitSum.Equal(creditSum) {
        return fmt.Errorf("账不平: 借=%v, 贷=%v", debitSum, creditSum)
    }
    
    return nil
}
```

## 5. 风控链路

### 5.1 风控流程

```
请求 → 特征提取 → 规则引擎 → 风险决策 → 执行/拒绝
```

### 5.2 常见风控规则

| 规则 | 说明 | 阈值 |
|------|------|------|
| **单日交易次数** | 防刷单 | >10次 |
| **单笔交易金额** | 异常大额 | >1万元 |
| **设备指纹** | 同设备多账号 | >5个账号 |
| **IP地址** | 同IP大量请求 | >100次/分钟 |
| **交易时间** | 凌晨交易 | 00:00-06:00 |

### 5.3 Go实现

```go
type RiskService struct {
    redis *redis.Client
}

// Evaluate 风险评估
func (s *RiskService) Evaluate(ctx context.Context, req *PayReq) (*RiskResult, error) {
    score := 0
    reasons := []string{}
    
    // 1. 检查单日交易次数
    dailyKey := fmt.Sprintf("risk:daily:%d:%s", req.UserID, time.Now().Format("20060102"))
    count, _ := s.redis.Incr(ctx, dailyKey).Result()
    s.redis.Expire(ctx, dailyKey, 24*time.Hour)
    
    if count > 10 {
        score += 30
        reasons = append(reasons, "单日交易次数过多")
    }
    
    // 2. 检查单笔金额
    if req.Amount > 10000 {
        score += 50
        reasons = append(reasons, "单笔金额过大")
    }
    
    // 3. 检查设备指纹
    deviceKey := fmt.Sprintf("risk:device:%s", req.DeviceID)
    userCount, _ := s.redis.SCard(ctx, deviceKey).Result()
    s.redis.SAdd(ctx, deviceKey, req.UserID)
    s.redis.Expire(ctx, deviceKey, 24*time.Hour)
    
    if userCount > 5 {
        score += 40
        reasons = append(reasons, "同设备多账号")
    }
    
    // 4. 判断风险等级
    level := "low"
    action := "pass"
    
    if score >= 80 {
        level = "high"
        action = "reject"
    } else if score >= 50 {
        level = "medium"
        action = "review"  // 人工审核
    }
    
    return &RiskResult{
        Score:   score,
        Level:   level,
        Action:  action,
        Reasons: reasons,
    }, nil
}
```

## 6. 安全防护

### 6.1 签名验证

```go
// Sign 生成签名
func Sign(params map[string]string, secretKey string) string {
    // 1. 参数排序
    keys := make([]string, 0, len(params))
    for k := range params {
        keys = append(keys, k)
    }
    sort.Strings(keys)
    
    // 2. 拼接字符串
    var sb strings.Builder
    for _, k := range keys {
        sb.WriteString(k)
        sb.WriteString("=")
        sb.WriteString(params[k])
        sb.WriteString("&")
    }
    sb.WriteString("key=")
    sb.WriteString(secretKey)
    
    // 3. MD5
    return md5Sum(sb.String())
}

// VerifySign 验证签名
func VerifySign(params map[string]string, sign, secretKey string) bool {
    expectedSign := Sign(params, secretKey)
    return expectedSign == sign
}
```

```go
func (s *PaymentService) CheckReplay(ctx context.Context, timestamp int64, nonce string) bool {
    // 1. 检查时间戳（±5分钟内有效）
    now := time.Now().Unix()
    if abs(now-timestamp) > 300 {
        return false
    }
    
    // 2. 检查nonce是否已使用
    key := fmt.Sprintf("nonce:%s", nonce)
    exists, _ := s.redis.Exists(ctx, key).Result()
    if exists > 0 {
        return false
    }
    
    // 3. 标记nonce已使用（10分钟有效期）
    s.redis.Set(ctx, key, 1, 10*time.Minute)
    
    return true
}
```

## 7. 常见面试题

### Q1: 扣款成功但回调丢失怎么办？

**答案**：
1. **主动查询**：定时任务查询第三方订单状态
2. **对账**：T+1对账发现长款，补单
3. **告警**：paying状态超时告警

### Q2: 如何防止并发扣款导致余额为负？

**答案**：
- 数据库约束：`UPDATE ... WHERE balance >= amount`
- 乐观锁：version字段
- 分布式锁：Redis SETNX

### Q3: 如何保证账不平？

**答案**：
- 双记账（借贷平衡）
- 事务保证原子性
- 定时校验 sum(借) = sum(贷)
- 只增不改，可追溯

### Q4: 对账发现差异如何处理？

**答案**：
- **长款**：查询订单，补单
- **短款**：查询第三方，确认失败后退款
- **金额不符**：人工核实
