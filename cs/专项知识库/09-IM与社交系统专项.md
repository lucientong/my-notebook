### 1.5 消息ID生成方案对比

| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| **Snowflake** | 有序、高性能、不依赖DB | 时钟回拨问题、机器ID分配 | 通用场景 |
| **DB自增** | 简单、严格递增 | 单点瓶颈、性能有限 | 小规模系统 |
| **Redis INCR** | 简单、高性能 | Redis故障风险 | 中等规模 |
| **Leaf-Segment** | 高可用、高性能 | 依赖DB、实现复杂 | 大规模分布式 |
| **Leaf-Snowflake** | 解决时钟回拨 | 依赖ZK | 大规模分布式 |
| **会话级SeqID** | 会话内严格有序 | 需要额外全局ID | IM消息排序 |

```go
// Snowflake ID生成器
type Snowflake struct {
    mu        sync.Mutex
    epoch     int64 // 起始时间戳
    timestamp int64
    workerID  int64 // 机器ID (10bit)
    sequence  int64 // 序列号 (12bit)
}

func NewSnowflake(workerID int64) *Snowflake {
    return &Snowflake{
        epoch:    1609459200000, // 2021-01-01
        workerID: workerID,
    }
}

func (s *Snowflake) NextID() int64 {
    s.mu.Lock()
    defer s.mu.Unlock()

    now := time.Now().UnixMilli()
    if now == s.timestamp {
        s.sequence = (s.sequence + 1) & 0xFFF // 12bit
        if s.sequence == 0 {
            // 序列号用尽，等待下一毫秒
            for now <= s.timestamp {
                now = time.Now().UnixMilli()
            }
        }
    } else {
        s.sequence = 0
    }

    if now < s.timestamp {
        panic("clock moved backwards")
    }
    s.timestamp = now

    id := ((now - s.epoch) << 22) | (s.workerID << 12) | s.sequence
    return id
}

// 会话级SeqID生成器（基于Redis）
type SeqGenerator struct {
    rdb *redis.Client
}

func (g *SeqGenerator) NextSeq(ctx context.Context, convID string) (int64, error) {
    key := fmt.Sprintf("seq:%s", convID)
    seq, err := g.rdb.Incr(ctx, key).Result()
    if err != nil {
        return 0, fmt.Errorf("gen seq failed: %w", err)
    }
    return seq, nil
}
```

## 02. 长连接管理 ⭐⭐⭐

### 2.1 WebSocket vs TCP 选型

| 特性 | WebSocket | TCP |
|------|-----------|-----|
| **协议层** | 应用层（基于HTTP升级） | 传输层 |
| **浏览器支持** | 原生支持 | 不支持 |
| **移动端** | 支持 | 支持 |
| **代理穿透** | 好（HTTP兼容） | 差（可能被防火墙拦截） |
| **性能** | 略低（帧头开销） | 更高 |
| **协议自定义** | 受限 | 完全自定义 |
| **典型场景** | Web端、跨平台 | 移动端、高性能场景 |
| **推荐** | Web + 移动端统一 | 追求极致性能 |

**实际选型建议**: 对外用WebSocket（兼容性好），内部网关间通信用TCP/gRPC。

### 2.2 连接保活与心跳机制

```
客户端                          服务端
  │                               │
  │────── Ping (30s周期) ────────→│
  │                               │ 更新最后心跳时间
  │←────── Pong ──────────────────│
  │                               │
  │                               │ 检测: now - lastPing > 90s ?
  │                               │ → 断开连接，清理Session
  │                               │
  │                          (连接断开)
  │                               │
  │───── 重连 (指数退避) ────────→│
  │                               │ 重建Session
  │←──── 同步离线消息 ───────────│
```

**心跳策略优化**:
- **固定心跳**: 每30s发送一次，简单但浪费流量
- **智能心跳**: 根据网络状况动态调整（WiFi 5min / 4G 2min / 弱网 30s）
- **系统级心跳**: 利用TCP keepalive，省电但不够灵活
- **应用级心跳**: 业务层自定义，灵活可控

### 2.3 Go实现WebSocket连接管理器

```go
package gateway

import (
    "context"
    "encoding/json"
    "fmt"
    "net/http"
    "sync"
    "sync/atomic"
    "time"

    "github.com/gorilla/websocket"
)

// ======================== 协议定义 ========================

const (
    CmdHeartbeat = 1  // 心跳
    CmdAuth      = 2  // 认证
    CmdMsg       = 3  // 消息
    CmdAck       = 4  // ACK
    CmdSync      = 5  // 同步
)

type Packet struct {
    Cmd  int             `json:"cmd"`
    Seq  int64           `json:"seq"`
    Data json.RawMessage `json:"data"`
}

// ======================== 连接抽象 ========================

type Connection struct {
    ID         string
    UID        int64
    DeviceType int
    DeviceID   string
    conn       *websocket.Conn
    sendCh     chan []byte
    closeCh    chan struct{}
    closeOnce  sync.Once
    lastPing   atomic.Int64
    manager    *ConnManager
}

func NewConnection(id string, ws *websocket.Conn, mgr *ConnManager) *Connection {
    c := &Connection{
        ID:      id,
        conn:    ws,
        sendCh:  make(chan []byte, 256),
        closeCh: make(chan struct{}),
        manager: mgr,
    }
    c.lastPing.Store(time.Now().Unix())
    return c
}

func (c *Connection) ReadLoop() {
    defer c.Close()
    c.conn.SetReadLimit(64 * 1024) // 64KB
    c.conn.SetReadDeadline(time.Now().Add(90 * time.Second))
    c.conn.SetPongHandler(func(string) error {
        c.lastPing.Store(time.Now().Unix())
        c.conn.SetReadDeadline(time.Now().Add(90 * time.Second))
        return nil
    })

    for {
        _, message, err := c.conn.ReadMessage()
        if err != nil {
            if websocket.IsUnexpectedCloseError(err,
                websocket.CloseGoingAway, websocket.CloseNormalClosure) {
                fmt.Printf("read error: %v\n", err)
            }
            return
        }

        var pkt Packet
        if err := json.Unmarshal(message, &pkt); err != nil {
            continue
        }
        c.manager.HandlePacket(c, &pkt)
    }
}

func (c *Connection) WriteLoop() {
    ticker := time.NewTicker(30 * time.Second)
    defer func() {
        ticker.Stop()
        c.Close()
    }()

    for {
        select {
        case msg, ok := <-c.sendCh:
            if !ok {
                c.conn.WriteMessage(websocket.CloseMessage, []byte{})
                return
            }
            c.conn.SetWriteDeadline(time.Now().Add(10 * time.Second))
            if err := c.conn.WriteMessage(websocket.TextMessage, msg); err != nil {
                return
            }
        case <-ticker.C:
            // 发送Ping心跳
            c.conn.SetWriteDeadline(time.Now().Add(10 * time.Second))
            if err := c.conn.WriteMessage(websocket.PingMessage, nil); err != nil {
                return
            }
        case <-c.closeCh:
            return
        }
    }
}

func (c *Connection) Send(data []byte) error {
    select {
    case c.sendCh <- data:
        return nil
    default:
        return fmt.Errorf("send buffer full, conn=%s", c.ID)
    }
}

func (c *Connection) Close() {
    c.closeOnce.Do(func() {
        close(c.closeCh)
        c.conn.Close()
        c.manager.RemoveConn(c)
    })
}

// ======================== 连接管理器 ========================

type ConnManager struct {
    mu       sync.RWMutex
    connByID map[string]*Connection            // connID -> conn
    connByUID map[int64]map[string]*Connection  // uid -> {connID: conn} 支持多端
    total    atomic.Int64
    upgrader websocket.Upgrader
}

func NewConnManager() *ConnManager {
    return &ConnManager{
        connByID:  make(map[string]*Connection),
        connByUID: make(map[int64]map[string]*Connection),
        upgrader: websocket.Upgrader{
            ReadBufferSize:  4096,
            WriteBufferSize: 4096,
            CheckOrigin:     func(r *http.Request) bool { return true },
        },
    }
}

func (m *ConnManager) HandleWebSocket(w http.ResponseWriter, r *http.Request) {
    ws, err := m.upgrader.Upgrade(w, r, nil)
    if err != nil {
        fmt.Printf("upgrade error: %v\n", err)
        return
    }

    connID := generateConnID()
    conn := NewConnection(connID, ws, m)

    m.mu.Lock()
    m.connByID[connID] = conn
    m.mu.Unlock()
    m.total.Add(1)

    go conn.ReadLoop()
    go conn.WriteLoop()
}

func (m *ConnManager) AuthConn(conn *Connection, uid int64, deviceType int, deviceID string) {
    conn.UID = uid
    conn.DeviceType = deviceType
    conn.DeviceID = deviceID

    m.mu.Lock()
    if _, ok := m.connByUID[uid]; !ok {
        m.connByUID[uid] = make(map[string]*Connection)
    }

    // 踢掉同类型设备的旧连接（互斥登录）
    for cid, old := range m.connByUID[uid] {
        if old.DeviceType == deviceType {
            delete(m.connByUID[uid], cid)
            go old.Close() // 异步关闭，避免死锁
        }
    }

    m.connByUID[uid][conn.ID] = conn
    m.mu.Unlock()
}

func (m *ConnManager) RemoveConn(conn *Connection) {
    m.mu.Lock()
    defer m.mu.Unlock()

    delete(m.connByID, conn.ID)
    if conn.UID > 0 {
        if conns, ok := m.connByUID[conn.UID]; ok {
            delete(conns, conn.ID)
            if len(conns) == 0 {
                delete(m.connByUID, conn.UID)
            }
        }
    }
    m.total.Add(-1)
}

// SendToUser 向指定用户所有在线设备推送
func (m *ConnManager) SendToUser(uid int64, data []byte) error {
    m.mu.RLock()
    conns, ok := m.connByUID[uid]
    if !ok {
        m.mu.RUnlock()
        return fmt.Errorf("user %d offline", uid)
    }
    // 复制一份避免长时间持锁
    targets := make([]*Connection, 0, len(conns))
    for _, c := range conns {
        targets = append(targets, c)
    }
    m.mu.RUnlock()

    for _, c := range targets {
        _ = c.Send(data)
    }
    return nil
}

// GetOnlineUIDs 获取在线用户列表
func (m *ConnManager) GetOnlineUIDs() []int64 {
    m.mu.RLock()
    defer m.mu.RUnlock()
    uids := make([]int64, 0, len(m.connByUID))
    for uid := range m.connByUID {
        uids = append(uids, uid)
    }
    return uids
}

// Stats 统计信息
func (m *ConnManager) Stats() (total int64, users int) {
    m.mu.RLock()
    users = len(m.connByUID)
    m.mu.RUnlock()
    return m.total.Load(), users
}

func (m *ConnManager) HandlePacket(conn *Connection, pkt *Packet) {
    switch pkt.Cmd {
    case CmdHeartbeat:
        conn.lastPing.Store(time.Now().Unix())
        resp, _ := json.Marshal(&Packet{Cmd: CmdHeartbeat, Seq: pkt.Seq})
        conn.Send(resp)
    case CmdAuth:
        // 解析认证数据，校验Token
        m.handleAuth(conn, pkt)
    case CmdMsg:
        m.handleMsg(conn, pkt)
    case CmdAck:
        m.handleAck(conn, pkt)
    }
}

func (m *ConnManager) handleAuth(conn *Connection, pkt *Packet) {
    var auth struct {
        Token      string `json:"token"`
        DeviceType int    `json:"device_type"`
        DeviceID   string `json:"device_id"`
    }
    json.Unmarshal(pkt.Data, &auth)
    uid, err := verifyToken(auth.Token)
    if err != nil {
        return
    }
    m.AuthConn(conn, uid, auth.DeviceType, auth.DeviceID)
}

func (m *ConnManager) handleMsg(conn *Connection, pkt *Packet)  { /* 转发到Logic层 */ }
func (m *ConnManager) handleAck(conn *Connection, pkt *Packet)  { /* 处理ACK */ }
func verifyToken(token string) (int64, error)                    { return 0, nil }
func generateConnID() string                                     { return fmt.Sprintf("%d", time.Now().UnixNano()) }
```

### 2.4 断线重连策略

```go
// 客户端断线重连（指数退避）
type ReconnectPolicy struct {
    BaseDelay  time.Duration // 初始延迟 1s
    MaxDelay   time.Duration // 最大延迟 30s
    MaxRetries int           // 最大重试次数 0=无限
    Jitter     float64       // 抖动因子 0.1
}

func (p *ReconnectPolicy) NextDelay(attempt int) time.Duration {
    if p.MaxRetries > 0 && attempt >= p.MaxRetries {
        return -1 // 不再重连
    }
    delay := p.BaseDelay * time.Duration(1<<uint(attempt))
    if delay > p.MaxDelay {
        delay = p.MaxDelay
    }
    // 添加随机抖动
    jitter := time.Duration(float64(delay) * p.Jitter * (rand.Float64()*2 - 1))
    return delay + jitter
}

// 重连后同步流程
// 1. 建立连接 → 2. 发送Auth(token + lastSeq) → 3. 服务端推送增量消息
```

## 03. 消息投递模型 ⭐⭐⭐

### 3.1 推模式 vs 拉模式 vs 推拉结合

```
推模式 (Push):
  发送方 → 服务端 → 直接推给所有接收方
  
  优点: 实时性最好
  缺点: 群消息写放大严重（1条群消息 → N次推送）

拉模式 (Pull):
  发送方 → 服务端存储 → 接收方定期拉取
  
  优点: 写入简单，适合大群
  缺点: 实时性差（取决于拉取频率），空拉浪费

推拉结合 (Push + Pull):
  发送方 → 服务端存储 → 推送通知(轻量) → 接收方拉取消息(按需)
  
  推: 只推送一个信号（有新消息 + 最新seq）
  拉: 客户端收到信号后，拉取 lastReadSeq ~ latestSeq 之间的消息
  
  ★ 业界主流方案
```

### 3.2 写扩散 vs 读扩散

```
写扩散 (Fan-out on Write / 推模式):
┌────────┐     ┌─────────────────────────┐
│ 发送方  │────→│    为每个接收方写一份     │
└────────┘     │  inbox_A: [msg1]        │
               │  inbox_B: [msg1]        │
               │  inbox_C: [msg1]        │
               │  ...                    │
               │  inbox_N: [msg1]        │
               └─────────────────────────┘

读扩散 (Fan-out on Read / 拉模式):
┌────────┐     ┌─────────────────────────┐
│ 发送方  │────→│    只写一份到群timeline   │
└────────┘     │  group_timeline: [msg1] │
               └─────────────────────────┘
                        │
               每个成员读取时从group_timeline拉
```

| 维度 | 写扩散 | 读扩散 |
|------|--------|--------|
| **写放大** | 严重（N成员写N次） | 无（只写1次） |
| **读放大** | 无（直接读自己inbox） | 有（需合并多个timeline） |
| **存储开销** | 大（N倍冗余） | 小 |
| **实时性** | 好（写入即可见） | 一般（需主动拉取） |
| **适合场景** | 单聊、小群 | 大群、Feed流 |
| **典型产品** | 微信（单聊写扩散） | Twitter Feed（读扩散） |

### 3.3 混合策略（最佳实践）

```
┌────────────────────────────────────────────┐
│              消息类型判断                     │
├─────────────┬──────────────────────────────┤
│  单聊消息    │  写扩散: 写入双方inbox        │
```

### 3.4 Timeline模型（钉钉方案）

```
Timeline模型核心思想:
- 每个用户有一个全局递增的Sync Timeline
- 所有消息（单聊+群聊）都写入用户自己的Timeline
- 客户端只需记住自己的lastSyncSeq，增量同步即可

用户A的Timeline:
┌──────┬──────┬──────────────────────────────┐
│ Seq  │ Type │ Content                      │
├──────┼──────┼──────────────────────────────┤
│ 101  │ 单聊 │ B→A: "你好"                  │
│ 102  │ 群聊 │ 群1: C说"开会"               │
│ 103  │ 单聊 │ D→A: "文件收到"              │
│ 104  │ 系统 │ 关注了你                      │
│ 105  │ 群聊 │ 群2: E说"下班"               │
└──────┴──────┴──────────────────────────────┘

同步方式:
  Client → Server: sync(lastSeq=102)
  Server → Client: [seq103, seq104, seq105]
```

```go
// Timeline同步服务
type TimelineService struct {
    store TimelineStore
    seq   SeqGenerator
}

type TimelineItem struct {
    Seq      int64  `json:"seq"`
    MsgID    int64  `json:"msg_id"`
    ConvID   string `json:"conv_id"`
    ConvType int    `json:"conv_type"`
}

// Sync 增量同步
func (s *TimelineService) Sync(ctx context.Context, uid int64, lastSeq int64, limit int) ([]TimelineItem, error) {
    items, err := s.store.GetAfterSeq(ctx, uid, lastSeq, limit)
    if err != nil {
        return nil, err
    }
    return items, nil
}

// Append 追加到用户Timeline
func (s *TimelineService) Append(ctx context.Context, uid int64, msgID int64, convID string, convType int) (int64, error) {
    seq, err := s.seq.NextSeq(ctx, fmt.Sprintf("timeline:%d", uid))
    if err != nil {
        return 0, err
    }
    item := TimelineItem{
        Seq:      seq,
        MsgID:    msgID,
        ConvID:   convID,
        ConvType: convType,
    }
    return seq, s.store.Save(ctx, uid, item)
}

// DispatchMessage 消息分发（写扩散到各用户Timeline）
func (s *TimelineService) DispatchMessage(ctx context.Context, msg *Message, receivers []int64) error {
    var wg sync.WaitGroup
    errCh := make(chan error, len(receivers))

    for _, uid := range receivers {
        wg.Add(1)
        go func(uid int64) {
            defer wg.Done()
            _, err := s.Append(ctx, uid, msg.MsgID, msg.ConvID, msg.ConvType)
            if err != nil {
                errCh <- fmt.Errorf("append timeline uid=%d: %w", uid, err)
            }
        }(uid)
    }
    wg.Wait()
    close(errCh)

    // 收集错误
    var errs []error
    for err := range errCh {
        errs = append(errs, err)
    }
    if len(errs) > 0 {
        return fmt.Errorf("dispatch errors: %v", errs)
    }
    return nil
}
```

## 04. 消息可靠性 ⭐⭐⭐

### 4.1 消息投递的三种语义

| 语义 | 说明 | 实现难度 | IM要求 |
|------|------|----------|--------|
| **At Most Once** | 最多一次，可能丢 | 低 | 不满足 |
| **At Least Once** | 至少一次，可能重 | 中 | 基本要求 |
| **Exactly Once** | 恰好一次 | 高 | 理想目标 |

**IM实际策略**: At Least Once + 客户端去重 ≈ Exactly Once

### 4.2 消息ACK机制

```
三次握手保证消息必达:

发送方A          服务端S          接收方B
  │                │                │
  │── 1.发送Msg ──→│                │
  │                │── 2.推送Msg ──→│
  │                │←── 3.Ack ──────│  B确认收到
  │←── 4.Ack ──────│                │  通知A：B已收
  │                │                │

超时重传:
  - S→B 推送后启动定时器（如5s）
  - 超时未收到B的Ack → 重推
  - 重试N次仍失败 → 转离线消息
  
  - A→S 发送后启动定时器
  - 超时未收到S的Ack → 重发
  - 重发携带相同ClientMsgID用于去重
```

### 4.3 Go实现可靠投递

```go
// ReliableDelivery 可靠投递服务
type ReliableDelivery struct {
    connMgr   *ConnManager
    store     MessageStore
    pushSvc   PushService
    pending   sync.Map // msgID -> *PendingMsg
    retryInterval time.Duration
    maxRetry  int
}

type PendingMsg struct {
    Msg        *Message
    TargetUID  int64
    RetryCount int
    CreateAt   time.Time
    Timer      *time.Timer
}

// Deliver 投递消息给在线用户
func (d *ReliableDelivery) Deliver(ctx context.Context, msg *Message, targetUID int64) error {
    // 1. 持久化消息
    if err := d.store.Save(ctx, msg); err != nil {
        return fmt.Errorf("save msg: %w", err)
    }

    // 2. 尝试在线推送
    data, _ := json.Marshal(&Packet{Cmd: CmdMsg, Seq: msg.MsgID, Data: mustMarshal(msg)})
    err := d.connMgr.SendToUser(targetUID, data)
    if err != nil {
        // 用户离线，走离线推送
        return d.pushSvc.PushOffline(ctx, targetUID, msg)
    }

    // 3. 等待ACK，设置超时重传
    pending := &PendingMsg{
        Msg:       msg,
        TargetUID: targetUID,
        CreateAt:  time.Now(),
    }
    d.pending.Store(msg.MsgID, pending)
    pending.Timer = time.AfterFunc(d.retryInterval, func() {
        d.retryDeliver(msg.MsgID)
    })

    return nil
}

// HandleAck 处理客户端ACK
func (d *ReliableDelivery) HandleAck(msgID int64) {
    if v, ok := d.pending.LoadAndDelete(msgID); ok {
        p := v.(*PendingMsg)
        p.Timer.Stop()
    }
}

// retryDeliver 重试投递
func (d *ReliableDelivery) retryDeliver(msgID int64) {
    v, ok := d.pending.Load(msgID)
    if !ok {
        return // 已ACK
    }
    p := v.(*PendingMsg)
    p.RetryCount++

    if p.RetryCount >= d.maxRetry {
        d.pending.Delete(msgID)
        // 超过最大重试，转离线推送
        d.pushSvc.PushOffline(context.Background(), p.TargetUID, p.Msg)
        return
    }

    // 重新推送
    data, _ := json.Marshal(&Packet{Cmd: CmdMsg, Seq: p.Msg.MsgID, Data: mustMarshal(p.Msg)})
    d.connMgr.SendToUser(p.TargetUID, data)

    // 设置下一次重试
    p.Timer = time.AfterFunc(d.retryInterval*time.Duration(1<<uint(p.RetryCount)), func() {
        d.retryDeliver(msgID)
    })
}

func mustMarshal(v interface{}) json.RawMessage {
    data, _ := json.Marshal(v)
    return data
}
```

```go
func (d *MsgDeduplicator) IsDuplicate(ctx context.Context, fromUID int64, clientMsgID string) (bool, error) {
    key := fmt.Sprintf("dedup:%d:%s", fromUID, clientMsgID)
    ok, err := d.rdb.SetNX(ctx, key, 1, d.ttl).Result()
    if err != nil {
        return false, err
    }
    return !ok, nil // ok=true表示设置成功(不重复); ok=false表示已存在(重复)
}

// 客户端去重: 维护已接收消息ID集合
type ClientDedup struct {
    mu       sync.RWMutex
    received map[int64]struct{} // msgID set
    maxSize  int
}

func (d *ClientDedup) IsNew(msgID int64) bool {
    d.mu.RLock()
    _, exists := d.received[msgID]
    d.mu.RUnlock()
    if exists {
        return false
    }
    d.mu.Lock()
    d.received[msgID] = struct{}{}
    // 简单淘汰
    if len(d.received) > d.maxSize {
        d.evict()
    }
    d.mu.Unlock()
    return true
}
```

### 4.5 消息有序性保证

```
问题: 网络延迟导致消息乱序

方案1: 会话级别SeqID
  - 每个会话维护递增Seq
  - 客户端按Seq排序展示
  - 发现Seq gap时主动拉取补齐

方案2: 服务端排序
  - 服务端为每条消息分配全局有序ID
  - 客户端拉取时已排好序

方案3: 客户端本地排序
  - 按服务端时间戳排序
  - 同一秒内按MsgID排序（MsgID单调递增）

实际方案:
  - 单聊: SeqID由服务端分配，客户端按Seq排序
  - 群聊: 群维度的SeqID，客户端按GroupSeq排序
  - 多端同步: 用户Timeline的Seq保证多端一致
```

```go
// 消息排序插入（客户端逻辑）
type OrderedMsgList struct {
    mu   sync.RWMutex
    msgs []*Message // 按SeqID有序
}

func (l *OrderedMsgList) Insert(msg *Message) {
    l.mu.Lock()
    defer l.mu.Unlock()

    // 二分查找插入位置
    pos := sort.Search(len(l.msgs), func(i int) bool {
        return l.msgs[i].SeqID >= msg.SeqID
    })

    // 去重
    if pos < len(l.msgs) && l.msgs[pos].SeqID == msg.SeqID {
        return
    }

    // 插入
    l.msgs = append(l.msgs, nil)
    copy(l.msgs[pos+1:], l.msgs[pos:])
    l.msgs[pos] = msg
}

// CheckGap 检测Seq缺口
func (l *OrderedMsgList) CheckGap() []int64 {
    l.mu.RLock()
    defer l.mu.RUnlock()

    var gaps []int64
    for i := 1; i < len(l.msgs); i++ {
        expected := l.msgs[i-1].SeqID + 1
        actual := l.msgs[i].SeqID
        for s := expected; s < actual; s++ {
            gaps = append(gaps, s)
        }
    }
    return gaps
}
```

```go
func (s *UnreadService) BatchGetUnread(ctx context.Context, uid int64, convIDs []string) (map[string]int64, error) {
    pipe := s.rdb.Pipeline()
    cmds := make(map[string]*redis.StringCmd)
    for _, convID := range convIDs {
        key := fmt.Sprintf("unread:%d:%s", uid, convID)
        cmds[convID] = pipe.Get(ctx, key)
    }
    _, _ = pipe.Exec(ctx)

    result := make(map[string]int64)
    for convID, cmd := range cmds {
        val, err := cmd.Int64()
        if err == redis.Nil {
            val = 0
        }
        result[convID] = val
    }
    return result, nil
}

func (s *UnreadService) MarkRead(ctx context.Context, uid int64, convID string, readSeq int64) error {
    // 1. 更新已读位点
    cursorKey := fmt.Sprintf("read_cursor:%s:%d", convID, uid)
    s.rdb.Set(ctx, cursorKey, readSeq, 0)

    // 2. 清除未读计数
    unreadKey := fmt.Sprintf("unread:%d:%s", uid, convID)
    oldCount, _ := s.rdb.Get(ctx, unreadKey).Int64()
    s.rdb.Del(ctx, unreadKey)

    // 3. 更新总未读数
    totalKey := fmt.Sprintf("unread_total:%d", uid)
    s.rdb.DecrBy(ctx, totalKey, oldCount)

    return nil
}
```

```go
// ReadReceiptBatcher 批量已读回执上报
type ReadReceiptBatcher struct {
    mu      sync.Mutex
    pending map[string]int64 // convID -> maxReadSeq
    flush   *time.Ticker
    svc     *UnreadService
    uid     int64
}

func NewReadReceiptBatcher(uid int64, svc *UnreadService) *ReadReceiptBatcher {
    b := &ReadReceiptBatcher{
        pending: make(map[string]int64),
        flush:   time.NewTicker(3 * time.Second),
        svc:     svc,
        uid:     uid,
    }
    go b.flushLoop()
    return b
}

func (b *ReadReceiptBatcher) Report(convID string, readSeq int64) {
    b.mu.Lock()
    if cur, ok := b.pending[convID]; !ok || readSeq > cur {
        b.pending[convID] = readSeq
    }
    b.mu.Unlock()
}

func (b *ReadReceiptBatcher) flushLoop() {
    for range b.flush.C {
        b.mu.Lock()
        batch := b.pending
        b.pending = make(map[string]int64)
        b.mu.Unlock()

        for convID, readSeq := range batch {
            b.svc.MarkRead(context.Background(), b.uid, convID, readSeq)
        }
    }
}
```

### 6.1 群消息扩散方案

```go
// GroupMsgDispatcher 群消息分发器
type GroupMsgDispatcher struct {
    groupSvc    *GroupService
    timelineSvc *TimelineService
    connMgr     *ConnManager
    pushSvc     PushService
    workerPool  chan struct{}
}

func NewGroupMsgDispatcher(poolSize int) *GroupMsgDispatcher {
    return &GroupMsgDispatcher{
        workerPool: make(chan struct{}, poolSize),
    }
}

func (d *GroupMsgDispatcher) Dispatch(ctx context.Context, msg *Message) error {
    group, err := d.groupSvc.GetGroup(ctx, msg.GroupID)
    if err != nil {
        return err
    }

    // 根据群大小选择分发策略
    if group.MemberCnt < 200 {
        return d.dispatchSmallGroup(ctx, msg, group)
    }
    return d.dispatchLargeGroup(ctx, msg, group)
}

// 小群: 写扩散
func (d *GroupMsgDispatcher) dispatchSmallGroup(ctx context.Context, msg *Message, group *Group) error {
    members, err := d.groupSvc.GetMembers(ctx, group.GroupID)
    if err != nil {
        return err
    }

    var wg sync.WaitGroup
    for _, uid := range members {
        if uid == msg.FromUID {
            continue // 跳过发送者自己
        }
        wg.Add(1)
        d.workerPool <- struct{}{} // 限流
        go func(uid int64) {
            defer func() {
                <-d.workerPool
                wg.Done()
            }()
            // 写入用户Timeline
            d.timelineSvc.Append(ctx, uid, msg.MsgID, msg.ConvID, msg.ConvType)
            // 尝试在线推送
            d.pushToUser(ctx, uid, msg)
        }(uid)
    }
    wg.Wait()
    return nil
}

// 大群: 读扩散 + 在线推送通知
func (d *GroupMsgDispatcher) dispatchLargeGroup(ctx context.Context, msg *Message, group *Group) error {
    // 1. 只写入群Timeline（1次写入）
    groupConvID := fmt.Sprintf("group_%d", group.GroupID)
    if err := d.saveToGroupTimeline(ctx, groupConvID, msg); err != nil {
        return err
    }

    // 2. 获取在线成员，推送轻量通知
    members, _ := d.groupSvc.GetMembers(ctx, group.GroupID)
    notification := &Packet{
        Cmd: CmdSync,
        Data: mustMarshal(map[string]interface{}{
            "conv_id":   groupConvID,
            "latest_seq": msg.SeqID,
        }),
    }
    notifyData, _ := json.Marshal(notification)

                if uid != msg.FromUID {
                    d.connMgr.SendToUser(uid, notifyData)
                }
            }
        }(batch)
    }

    return nil
```

```go
// ShardRouter 分片路由
type ShardRouter struct {
    shardCount int
    dbCount    int
}

func (r *ShardRouter) GetShard(key int64) (db int, table int) {
    shard := key % int64(r.shardCount)
    db = int(shard) / (r.shardCount / r.dbCount)
    table = int(shard)
    return
}

func (r *ShardRouter) GetShardByStr(key string) (db int, table int) {
    h := fnv.New32a()
    h.Write([]byte(key))
    shard := int(h.Sum32()) % r.shardCount
    db = shard / (r.shardCount / r.dbCount)
    table = shard
    return
}

// MessageStore 消息存储
type MessageStore struct {

    _, err := s.dbs[db].ExecContext(ctx, query,
        msg.MsgID, msg.MsgType, msg.Content, msg.Extra, msg.Status, msg.CreateTime)
    return err
}

    msg := &Message{}
    err := row.Scan(&msg.MsgID, &msg.MsgType, &msg.Content, &msg.Extra, &msg.Status, &msg.CreateTime)
    return msg, err
}

func (s *MessageStore) GetHistoryMessages(ctx context.Context, convID string, startSeq, endSeq int64, limit int) ([]*Message, error) {
    db, table := s.router.GetShardByStr(convID)

            "FROM msg_index_p2p_%d idx "+
            "JOIN message_%d m ON idx.msg_id = m.msg_id "+
            "WHERE idx.conv_id = ? AND idx.seq_id > ? AND idx.seq_id <= ? "+
            "ORDER BY idx.seq_id ASC LIMIT ?", table, table)
    rows, err := s.dbs[db].QueryContext(ctx, query, convID, startSeq, endSeq, limit)
    if err != nil {
        return nil, err
```

## 08. Feed流系统 ⭐⭐⭐

### 8.1 Feed流类型

| 类型 | 说明 | 排序方式 | 典型产品 |
|------|------|----------|----------|
| **Timeline** | 关注者动态 | 时间倒序 | Twitter/微博 |
| **关注流** | 关注的人的内容 | 时间倒序 | Instagram |
| **推荐流** | 算法推荐 | 兴趣分值 | 抖音/头条 |
| **混合流** | 关注+推荐混合 | 混合排序 | 微博发现 |

### 8.2 推拉模式对比（Feed场景）

```
写扩散 (Push/Fan-out-on-write):
  用户A发帖 → 写入A所有粉丝的Feed收件箱
  
  timeline:user_B = [postA_1, postC_2, postD_3, ...]
  timeline:user_C = [postA_1, postE_2, ...]
  
  读: 直接读自己的Feed收件箱，已排序
  
  适合: 粉丝少的普通用户

读扩散 (Pull/Fan-out-on-read):
  用户B打开Feed → 查询B关注的所有人 → 拉取每人最新发帖 → 合并排序
  
  读: SELECT * FROM posts WHERE uid IN (关注列表) ORDER BY time DESC LIMIT 20
  
  适合: 关注少的场景、大V发帖

混合模式:
  普通用户(粉丝<1万): 写扩散
  大V(粉丝>1万): 读扩散
  读取时: 先读inbox(写扩散部分) → 再拉取大V动态(读扩散部分) → 合并
```

### 8.3 大V问题与解决方案

```
问题: 大V(千万粉丝)发一条动态 → 写扩散需要写千万次

方案1: 大V用读扩散
  - 大V发帖只写一份
  - 粉丝读取时实时拉取大V动态
  - 问题: 如果用户关注了多个大V，读取时要合并排序

方案2: 大V延迟扩散
  - 大V发帖先存储，不立即扩散
  - 粉丝打开Feed时再触发扩散（按需）
  - 活跃粉丝实时扩散，不活跃粉丝跳过

方案3: 大V预计算 + 缓存
  - 大V动态维护一个公共timeline
  - 粉丝拉取时合并：个人inbox + 关注大V列表的公共timeline
  
  ★ 推荐方案3
```

```go
package feed

import (
    "context"
    "fmt"
    "sort"
    "time"
)

const (
    FanoutThreshold = 10000 // 粉丝数阈值，超过则为大V
    FeedPageSize    = 20
)

type FeedItem struct {
    PostID    int64   `json:"post_id"`
    AuthorUID int64   `json:"author_uid"`
    Content   string  `json:"content"`
    Score     float64 `json:"score"`     // 排序分值
    CreateAt  int64   `json:"create_at"`
}

type FeedService struct {
    rdb         *redis.Client
    postStore   PostStore
    relationSvc RelationService
}

// PublishPost 发布动态
func (s *FeedService) PublishPost(ctx context.Context, authorUID int64, post *FeedItem) error {
    // 1. 存储原始帖子
    if err := s.postStore.Save(ctx, post); err != nil {
        return err
    }

    // 2. 写入作者自己的timeline
    authorKey := fmt.Sprintf("user_timeline:%d", authorUID)
    s.rdb.ZAdd(ctx, authorKey, &redis.Z{
        Score:  float64(post.CreateAt),
        Member: post.PostID,
    })

    // 3. 判断是否大V
    followerCount, _ := s.relationSvc.GetFollowerCount(ctx, authorUID)
    if followerCount >= FanoutThreshold {
        // 大V: 不做写扩散，粉丝读取时拉
        return nil
    }

    // 4. 普通用户: 写扩散到粉丝inbox
    go s.fanoutToFollowers(ctx, authorUID, post)
    return nil
}

// fanoutToFollowers 异步写扩散
func (s *FeedService) fanoutToFollowers(ctx context.Context, authorUID int64, post *FeedItem) {
    followers, _ := s.relationSvc.GetFollowers(ctx, authorUID)

    // 分批写入
    batchSize := 1000
    for i := 0; i < len(followers); i += batchSize {
        end := i + batchSize
        if end > len(followers) {
            end = len(followers)
        }
        batch := followers[i:end]

        pipe := s.rdb.Pipeline()
        for _, uid := range batch {
            key := fmt.Sprintf("feed_inbox:%d", uid)
            pipe.ZAdd(ctx, key, &redis.Z{
                Score:  float64(post.CreateAt),
                Member: post.PostID,
            })
            // 限制inbox大小
            pipe.ZRemRangeByRank(ctx, key, 0, -1001) // 保留最新1000条
        }
        pipe.Exec(ctx)
    }
}

// GetFeed 获取Feed流
func (s *FeedService) GetFeed(ctx context.Context, uid int64, cursor int64, limit int) ([]*FeedItem, int64, error) {
    if cursor == 0 {
        cursor = time.Now().UnixMilli()
    }

    // 1. 从inbox获取写扩散的内容
    inboxKey := fmt.Sprintf("feed_inbox:%d", uid)
    inboxPostIDs, _ := s.rdb.ZRevRangeByScore(ctx, inboxKey, &redis.ZRangeBy{
        Max:    fmt.Sprintf("%d", cursor-1),
        Min:    "-inf",
        Count:  int64(limit),
    }).Result()

    // 2. 获取关注的大V列表，拉取他们的最新动态
    bigVUIDs, _ := s.getFollowingBigV(ctx, uid)
    var bigVPostIDs []string
    for _, vUID := range bigVUIDs {
        vKey := fmt.Sprintf("user_timeline:%d", vUID)
        posts, _ := s.rdb.ZRevRangeByScore(ctx, vKey, &redis.ZRangeBy{
            Max:   fmt.Sprintf("%d", cursor-1),
            Min:   "-inf",
            Count: 5, // 每个大V取最近5条
        }).Result()
        bigVPostIDs = append(bigVPostIDs, posts...)
    }

    // 3. 合并去重
    allPostIDs := mergeAndDedup(inboxPostIDs, bigVPostIDs)

    // 4. 批量获取帖子详情
    items, _ := s.postStore.BatchGet(ctx, allPostIDs)

    // 5. 按时间排序
    sort.Slice(items, func(i, j int) bool {
        return items[i].CreateAt > items[j].CreateAt
    })

    // 6. 截断
    if len(items) > limit {
        items = items[:limit]
    }

    // 7. 计算下一页cursor
    var nextCursor int64
    if len(items) > 0 {
        nextCursor = items[len(items)-1].CreateAt
    }

    return items, nextCursor, nil
}

func (s *FeedService) getFollowingBigV(ctx context.Context, uid int64) ([]int64, error) {
    // 从关注列表中筛选粉丝数 > FanoutThreshold 的用户
    followings, _ := s.relationSvc.GetFollowings(ctx, uid)
    var bigVs []int64
    for _, fUID := range followings {
        cnt, _ := s.relationSvc.GetFollowerCount(ctx, fUID)
        if cnt >= FanoutThreshold {
            bigVs = append(bigVs, fUID)
        }
    }
    return bigVs, nil
}

func mergeAndDedup(a, b []string) []int64 {
    seen := make(map[int64]struct{})
    var result []int64
    for _, list := range [][]string{a, b} {
        for _, s := range list {
            id, _ := strconv.ParseInt(s, 10, 64)
            if _, ok := seen[id]; !ok {
                seen[id] = struct{}{}
                result = append(result, id)
            }
        }
    }
    return result
}
```

### 8.5 Feed流排序策略

| 策略 | 公式示例 | 适用场景 |
|------|----------|----------|
| **时间倒序** | score = timestamp | 关注流 |
| **EdgeRank** | score = affinity × weight × decay | Facebook |
| **热度排序** | score = log(likes) + age_hours/45 | Reddit/HN |
| **兴趣推荐** | score = model_predict(user, item) | 抖音/头条 |

```go
// HotScore Reddit风格热度算法
func HotScore(ups, downs int, createAt time.Time) float64 {
    score := float64(ups - downs)
    order := math.Log10(math.Max(math.Abs(score), 1))
    var sign float64
    if score > 0 {
        sign = 1
    } else if score < 0 {
        sign = -1
    }
    seconds := float64(createAt.Unix()-1134028003) / 45000.0
    return sign*order + seconds
}

// DecayScore 时间衰减算法
func DecayScore(baseScore float64, createAt time.Time, halfLife time.Duration) float64 {
    age := time.Since(createAt)
    decay := math.Pow(0.5, float64(age)/float64(halfLife))
    return baseScore * decay
}
```

```go
type RelationService struct {
    db  *sql.DB
    rdb *redis.Client

// Follow 关注
func (s *RelationService) Follow(ctx context.Context, fromUID, toUID int64) error {
    tx, err := s.db.BeginTx(ctx, nil)
    if err != nil {
        return err
    }
    defer tx.Rollback()

    // 1. 写入关注记录
    _, err = tx.ExecContext(ctx,
        "INSERT IGNORE INTO follow (from_uid, to_uid, create_time) VALUES (?, ?, NOW())",
        fromUID, toUID)
    if err != nil {
        return err
    }

    // 2. 更新计数
    tx.ExecContext(ctx,
        "INSERT INTO follow_count (uid, following_cnt) VALUES (?, 1) ON DUPLICATE KEY UPDATE following_cnt = following_cnt + 1",
        fromUID)
    tx.ExecContext(ctx,
        "INSERT INTO follow_count (uid, follower_cnt) VALUES (?, 1) ON DUPLICATE KEY UPDATE follower_cnt = follower_cnt + 1",
        toUID)

    // 3. 检查是否互相关注 → 好友
    var cnt int
    tx.QueryRowContext(ctx,
        "SELECT COUNT(*) FROM follow WHERE from_uid = ? AND to_uid = ?",
        toUID, fromUID).Scan(&cnt)
    if cnt > 0 {
        // 互相关注，更新好友计数
        tx.ExecContext(ctx,
            "UPDATE follow_count SET friend_cnt = friend_cnt + 1 WHERE uid IN (?, ?)",
            fromUID, toUID)
    }

    if err := tx.Commit(); err != nil {
        return err
    }

    // 4. 更新Redis缓存
    s.rdb.SAdd(ctx, fmt.Sprintf("following:%d", fromUID), toUID)
    s.rdb.SAdd(ctx, fmt.Sprintf("followers:%d", toUID), fromUID)

    return nil
}

// GetMutualFriends 共同关注/共同好友
func (s *RelationService) GetMutualFriends(ctx context.Context, uidA, uidB int64) ([]int64, error) {
    keyA := fmt.Sprintf("following:%d", uidA)
    keyB := fmt.Sprintf("following:%d", uidB)

    // Redis交集运算
    result, err := s.rdb.SInter(ctx, keyA, keyB).Result()
    if err != nil {
        return nil, err
    }

    uids := make([]int64, len(result))
    for i, v := range result {
```

### 9.4 六度分离与图遍历

```
六度分离理论: 世界上任意两个人之间最多通过6个中间人就能建立联系

应用场景:
  - LinkedIn: "你和XX之间有2度关系"
  - 推荐关注: "你可能认识的人"（基于2度关系）
  
实现: BFS (广度优先搜索)
  - 1度: 直接好友
  - 2度: 好友的好友
  - N度: BFS第N层

图数据库选型:
  - Neo4j: 最成熟的图数据库
  - DGraph: 分布式图数据库
  - TigerGraph: 高性能图分析
  - Redis + 应用层: 简单场景
```

```go
// 推荐可能认识的人（2度关系）
func (s *RelationService) RecommendFriends(ctx context.Context, uid int64, limit int) ([]int64, error) {
    // 1. 获取我的关注列表
    myFollowings, _ := s.GetFollowings(ctx, uid)
    myFollowSet := make(map[int64]struct{})
    for _, f := range myFollowings {
        myFollowSet[f] = struct{}{}
    }

    // 2. 获取好友的好友
    candidateCount := make(map[int64]int) // uid -> 共同好友数
    for _, friendUID := range myFollowings {
        fof, _ := s.GetFollowings(ctx, friendUID)
        for _, candidate := range fof {
            if candidate == uid {
                continue
            }
            if _, isMyFriend := myFollowSet[candidate]; isMyFriend {
                continue
            }
            candidateCount[candidate]++
        }
    }

    // 3. 按共同好友数排序
    type pair struct {
        UID   int64
        Count int
    }
    var pairs []pair
    for uid, cnt := range candidateCount {
        pairs = append(pairs, pair{uid, cnt})
    }
    sort.Slice(pairs, func(i, j int) bool {
        return pairs[i].Count > pairs[j].Count
    })

    // 4. 取TopN
    result := make([]int64, 0, limit)
    for i := 0; i < len(pairs) && i < limit; i++ {
        result = append(result, pairs[i].UID)
    }
    return result, nil
}
```

```go
type IMServer struct {
    gateway   *ConnManager
    router    *RouterService
    logic     *LogicService
    store     *MessageStore
    timeline  *TimelineService
    unread    *UnreadService
    push      PushService
    dedup     *MsgDeduplicator
    producer  *kafka.Producer
}

// SendMessage 发送消息（入口）
func (s *IMServer) SendMessage(ctx context.Context, fromConn *Connection, req *SendMsgReq) (*SendMsgResp, error) {
    // 1. 参数校验
    if err := req.Validate(); err != nil {
        return nil, fmt.Errorf("invalid request: %w", err)
    }

    // 2. 权限校验（是否被拉黑、是否有发言权限等）
    if err := s.logic.CheckPermission(ctx, fromConn.UID, req); err != nil {
        return nil, err
    }

    // 3. 消息去重
    isDup, err := s.dedup.IsDuplicate(ctx, fromConn.UID, req.ClientMsgID)
    if err != nil {
        return nil, err
    }
    if isDup {
        // 返回之前的MsgID（幂等）
        return s.logic.GetDupResp(ctx, fromConn.UID, req.ClientMsgID)
    }

    // 4. 生成消息ID和Seq
    msg := &Message{
        MsgID:      s.logic.GenMsgID(),
        ConvID:     req.ConvID,
        ConvType:   req.ConvType,
        FromUID:    fromConn.UID,
        ToUID:      req.ToUID,
        GroupID:    req.GroupID,
        MsgType:    req.MsgType,
        Content:    req.Content,
        CreateTime: time.Now(),
    }

    // 5. 内容审核（敏感词过滤）
    msg.Content = s.logic.FilterContent(msg.Content)

    // 6. 持久化消息
    if err := s.store.SaveMessage(ctx, msg); err != nil {
        return nil, fmt.Errorf("save message: %w", err)
    }

    // 7. 异步投递（通过Kafka解耦）
    event := &MsgEvent{Msg: msg, Receivers: s.logic.GetReceivers(ctx, msg)}
    s.producer.Produce("msg_deliver", msg.ConvID, mustMarshal(event))

    // 8. 给发送方返回ACK
    return &SendMsgResp{
        MsgID:     msg.MsgID,
        SeqID:     msg.SeqID,
        Timestamp: msg.CreateTime.UnixMilli(),
    }, nil
}

type SendMsgReq struct {
    ClientMsgID string `json:"client_msg_id"` // 客户端幂等ID
    ConvID      string `json:"conv_id"`
    ConvType    int    `json:"conv_type"`
    ToUID       int64  `json:"to_uid"`
    GroupID     int64  `json:"group_id"`
    MsgType     int    `json:"msg_type"`
    Content     string `json:"content"`

    MsgID     int64 `json:"msg_id"`
    SeqID     int64 `json:"seq_id"`
    Timestamp int64 `json:"timestamp"`

    Msg       *Message `json:"msg"`

// ======================== 消息投递消费者 ========================

type MsgDeliverConsumer struct {
    server *IMServer
}

func (c *MsgDeliverConsumer) Consume(data []byte) error {
    var event MsgEvent
    json.Unmarshal(data, &event)
    msg := event.Msg

    for _, uid := range event.Receivers {
        // 1. 写入用户Timeline
        seq, _ := c.server.timeline.Append(context.Background(), uid, msg.MsgID, msg.ConvID, msg.ConvType)

        // 2. 更新未读计数
        c.server.unread.IncrUnread(context.Background(), uid, msg.ConvID)
        c.server.unread.IncrTotalUnread(context.Background(), uid)

        // 3. 更新会话
        c.server.logic.UpdateConversation(context.Background(), uid, msg, seq)

        // 4. 在线推送
        pushData, _ := json.Marshal(&Packet{
            Cmd: CmdMsg,
            Data: mustMarshal(map[string]interface{}{
                "msg":       msg,
                "seq":       seq,
                "unread_cnt": c.server.unread.GetConvUnread(context.Background(), uid, msg.ConvID),
            }),
        })

        err := c.server.gateway.SendToUser(uid, pushData)
        if err != nil {
            // 5. 离线推送
            c.server.push.PushOffline(context.Background(), uid, msg)
        }
    }
    return nil
}
```

### 12.2 初级高频速刷 (5题)

**Q1: IM系统为什么要用长连接而不是HTTP轮询？**

> 长连接(WebSocket/TCP)相比HTTP轮询：① 实时性好，服务端可主动推送；② 减少连接建立开销（HTTP每次请求建立TCP+TLS）；③ 节省带宽（无HTTP头重复开销）；④ 降低服务端资源消耗。短轮询：延迟=轮询间隔，长轮询：每30s一次空请求。WebSocket可在毫秒级推送到客户端。

**Q2: 消息的写扩散和读扩散有什么区别？**

> 写扩散(Fan-out on Write): 发消息时写入所有接收方的收件箱。优点是读取快，缺点是写放大严重。适用于单聊和小群。
> 读扩散(Fan-out on Read): 只写一份到发送方timeline，接收方读取时拉取合并。优点是写入少，缺点是读放大。适用于大群和Feed流。
> 实践中常用混合模式：单聊+小群写扩散，大群读扩散。

**Q3: 如何保证消息不丢失？**

> 三个环节保障：① 发送方→服务端：客户端收到服务端ACK后才确认发送成功，超时重传；② 服务端持久化：消息写入DB/MQ成功后才返回ACK；③ 服务端→接收方：推送后等待客户端ACK，超时重传，多次失败转离线消息。加上客户端去重，实现At Least Once + 去重≈Exactly Once。

**Q4: 未读消息计数怎么实现？**

> Redis维护两级计数：① 会话未读数 `unread:{uid}:{convID}`，收到新消息INCR，用户已读时DEL；② 总未读数 `unread_total:{uid}`，所有会话未读之和。用户进入会话时上报已读位点（readSeq），服务端清除该会话未读并更新总数。群消息可以批量延迟更新。

**Q5: WebSocket和TCP长连接怎么选？**

> WebSocket：基于HTTP升级，浏览器原生支持，穿透代理防火墙好。TCP：协议可完全自定义，无帧头开销，性能略优。实践选择：对外统一WebSocket（兼容Web端），内部服务间gRPC/TCP。移动端可用WebSocket或自定义TCP协议。

### 12.3 中级进阶 (5题)

**Q1: 如何设计支持百万在线的长连接网关？**

> 核心要素：① 使用epoll/kqueue管理连接（Go的netpoll已封装）；② 每连接控制在两个goroutine（读/写），内存控制8KB以内；③ 单机50W连接 × 8KB ≈ 4GB内存；④ 无锁/分段锁的连接管理（sync.Map或分桶锁）；⑤ 连接生命周期管理：鉴权→保活→优雅关闭；⑥ 多机部署+LB，通过Router服务维护uid→gateway映射。

**Q2: 万人群消息如何高效投递？**

> 读扩散+分级推送：① 消息只写一份到群timeline（O(1)写入）；② 在线成员按活跃度分级推送通知（活跃用户实时推，不活跃批量推）；③ 客户端收到通知后按需拉取消息；④ 群成员列表缓存在Redis Set，支持快速查询；⑤ 分批推送避免瞬间流量尖峰；⑥ @消息单独推送被@用户。

**Q3: Feed流的大V问题怎么解决？**

> 混合推拉模式：① 普通用户(粉丝<1W)用写扩散，发帖时写入粉丝inbox；② 大V(粉丝>1W)不做写扩散，只写入自己的timeline；③ 粉丝读取Feed时：先读inbox(写扩散部分) + 拉取关注大V的timeline(读扩散部分) → 合并排序。关键优化：大V timeline缓存在Redis ZSet，每个大V只取最近N条。

**Q4: 消息ID如何设计？全局有序还是会话有序？**

> 分两层ID：① 全局MsgID：Snowflake生成，用于消息唯一标识和存储路由，不要求严格有序；② 会话SeqID：每个会话维护递增序列号（Redis INCR），用于消息排序和同步。客户端按SeqID排序展示，发现SeqID gap时主动拉取补齐。好处：解耦全局唯一性和有序性，各自优化。

**Q5: 多端消息同步如何设计？**

> 基于用户Timeline增量同步：① 每个用户维护全局递增Seq的Timeline；② 所有消息（单聊+群聊）写入用户Timeline；③ 各端记录lastSyncSeq；④ 同步时：client发送sync(lastSeq) → server返回(lastSeq, latestSeq]的消息列表；⑤ 触发时机：长连接建立、收到推送、APP切前台。优势：多端一致、支持任意端增量同步。

**Q1: IM系统中TCP粘包/拆包怎么处理？**

A: 自定义协议帧格式解决。常见方案：

```go
// 协议帧格式: [魔数4B][版本1B][命令2B][序列号4B][包体长度4B][包体...]
type Frame struct {
    Magic   uint32 // 魔数，用于快速识别协议
    Version uint8
    Cmd     uint16
    SeqNum  uint32
    BodyLen uint32
    Body    []byte
}

const (
    MagicNumber = 0x12345678
    HeaderSize  = 4 + 1 + 2 + 4 + 4 // 15 bytes
    MaxBodySize = 1 << 20            // 1MB
)

// Codec 编解码器
type Codec struct {
    reader *bufio.Reader
    writer *bufio.Writer
}

func (c *Codec) ReadFrame() (*Frame, error) {
    // 1. 读取固定长度的头部
    header := make([]byte, HeaderSize)
    if _, err := io.ReadFull(c.reader, header); err != nil {
        return nil, err
    }

    f := &Frame{
        Magic:   binary.BigEndian.Uint32(header[0:4]),
        Version: header[4],
        Cmd:     binary.BigEndian.Uint16(header[5:7]),
        SeqNum:  binary.BigEndian.Uint32(header[7:11]),
        BodyLen: binary.BigEndian.Uint32(header[11:15]),
    }

    // 2. 校验魔数
    if f.Magic != MagicNumber {
        return nil, fmt.Errorf("invalid magic: %x", f.Magic)
    }

    // 3. 校验包体长度
    if f.BodyLen > MaxBodySize {
        return nil, fmt.Errorf("body too large: %d", f.BodyLen)
    }

    // 4. 读取包体
    if f.BodyLen > 0 {
        f.Body = make([]byte, f.BodyLen)
        if _, err := io.ReadFull(c.reader, f.Body); err != nil {
            return nil, err
        }
    }
    return f, nil
}

func (c *Codec) WriteFrame(f *Frame) error {
    header := make([]byte, HeaderSize)
    binary.BigEndian.PutUint32(header[0:4], MagicNumber)
    header[4] = f.Version
    binary.BigEndian.PutUint16(header[5:7], f.Cmd)
    binary.BigEndian.PutUint32(header[7:11], f.SeqNum)
    binary.BigEndian.PutUint32(header[11:15], uint32(len(f.Body)))

    if _, err := c.writer.Write(header); err != nil {
        return err
    }
    if len(f.Body) > 0 {
        if _, err := c.writer.Write(f.Body); err != nil {
            return err
        }
    }
    return c.writer.Flush()
}
```

**Q2: 如何实现消息的端到端加密(E2E)？**

A: Signal Protocol方案（WhatsApp/Signal使用）：

```
密钥交换流程 (X3DH):
1. 每个用户生成长期身份密钥对 (IK) 和一批一次性预密钥 (OPK)
2. 上传公钥到服务器
3. 发送方获取接收方公钥，通过X3DH协商出共享密钥
4. 使用共享密钥加密消息（AES-256-GCM）
5. 服务器只存储密文，无法解密

核心特性:
- 前向安全: 每条消息使用不同密钥(Double Ratchet)
- 即使长期密钥泄露，历史消息也无法解密
- 服务端无法解密，保证隐私

代价:
- 不支持服务端搜索
- 不支持服务端内容审核
- 多端同步更复杂（密钥需要同步）
```

```go
func (s *IMServer) RecallMessage(ctx context.Context, uid int64, msgID int64) error {
    // 1. 校验: 只能撤回自己的消息 + 2分钟内
    msg, err := s.store.GetMessage(ctx, msgID)
    if err != nil {
        return err
    }
    if msg.FromUID != uid {
        return fmt.Errorf("can only recall own message")
    }
    if time.Since(msg.CreateTime) > 2*time.Minute {

    s.store.UpdateStatus(ctx, msgID, StatusRecalled)

    recallNotify := &Message{
        MsgID:    s.logic.GenMsgID(),
        ConvID:   msg.ConvID,
        ConvType: msg.ConvType,
        FromUID:  uid,
        MsgType:  MsgTypeRecall,
        Content:  fmt.Sprintf(`{"recall_msg_id":%d}`, msgID),
    }

    // 4. 推送撤回通知给会话中的其他人
    receivers := s.logic.GetReceivers(ctx, msg)
    for _, r := range receivers {
        data, _ := json.Marshal(&Packet{Cmd: CmdMsg, Data: mustMarshal(recallNotify)})
        s.gateway.SendToUser(r, data)
    }

    // 5. 客户端收到撤回通知后，替换本地消息为"XX撤回了一条消息"
    return nil
}
```

**Q7: 如何处理消息中的敏感词？**

A:

```go
// DFA(确定有限自动机)敏感词过滤
type TrieNode struct {
    children map[rune]*TrieNode
    isEnd    bool
}

type SensitiveFilter struct {
    root *TrieNode
}

func NewSensitiveFilter(words []string) *SensitiveFilter {
    f := &SensitiveFilter{root: &TrieNode{children: make(map[rune]*TrieNode)}}
    for _, word := range words {
        f.addWord(word)
    }
    return f
}

func (f *SensitiveFilter) addWord(word string) {
    node := f.root
    for _, ch := range word {
        if _, ok := node.children[ch]; !ok {
            node.children[ch] = &TrieNode{children: make(map[rune]*TrieNode)}
        }
        node = node.children[ch]
    }
    node.isEnd = true
}

// Filter 将敏感词替换为 ***
func (f *SensitiveFilter) Filter(text string) string {
    runes := []rune(text)
    result := make([]rune, len(runes))
    copy(result, runes)

    for i := 0; i < len(runes); i++ {
        node := f.root
        j := i
        for j < len(runes) {
            child, ok := node.children[runes[j]]
            if !ok {
                break
            }
            node = child
            if node.isEnd {
                // 命中敏感词，替换为*
                for k := i; k <= j; k++ {
                    result[k] = '*'
                }
            }
            j++
        }
    }
    return string(result)
}
```

```go
        result[uid] = cmd.Val() > 0
    }
    return result, nil
```

**Q11: Gateway如何做优雅重启/发布？**

A:

```
优雅重启流程:
1. 新实例启动，开始接受新连接
2. 旧实例: 从LB摘除（不再接受新连接）
3. 旧实例: 给所有在线连接发送GOAWAY信号
4. 客户端收到GOAWAY → 主动断开 → 重连到新实例
5. 旧实例: 等待一段时间(如30s)，确保连接迁移完
6. 旧实例: 强制关闭剩余连接，退出进程

关键: 客户端要支持GOAWAY协议，收到后无感重连
```

```go
func (m *ConnManager) GracefulShutdown(timeout time.Duration) {
    // 1. 停止接受新连接
    // 2. 向所有连接发送GOAWAY
    m.mu.RLock()
    for _, conn := range m.connByID {
        goaway, _ := json.Marshal(&Packet{Cmd: CmdGoaway})
        conn.Send(goaway)
    }
    m.mu.RUnlock()

    // 3. 等待连接自然断开
    deadline := time.After(timeout)
    ticker := time.NewTicker(time.Second)
    defer ticker.Stop()

    for {
        select {
        case <-deadline:
            // 超时强制关闭
            m.mu.Lock()
            for _, conn := range m.connByID {
                conn.Close()
            }
            m.mu.Unlock()
            return
        case <-ticker.C:
            if m.total.Load() == 0 {
                return // 所有连接已断开
            }
        }
    }
}
```

```go
// Cursor: SELECT * FROM posts WHERE create_at < cursor ORDER BY create_at DESC LIMIT 20
//   优点: 不受新数据影响，稳定翻页

type FeedPagination struct {
    Cursor string `json:"cursor"` // 加密的时间戳或ID
    Limit  int    `json:"limit"`
    HasMore bool  `json:"has_more"`
}

func (s *FeedService) GetFeedPage(ctx context.Context, uid int64, cursor string, limit int) (*FeedPage, error) {
    var cursorTS int64
    if cursor == "" {
        cursorTS = time.Now().UnixMilli()
    } else {
        cursorTS = decodeCursor(cursor)
    }

    // 多取一条用于判断hasMore
    items, _ := s.getFeedItems(ctx, uid, cursorTS, limit+1)

    hasMore := len(items) > limit
    if hasMore {
        items = items[:limit]
    }

    var nextCursor string
    if hasMore && len(items) > 0 {
        nextCursor = encodeCursor(items[len(items)-1].CreateAt)
    }

    return &FeedPage{
        Items:   items,
        Cursor:  nextCursor,
        HasMore: hasMore,
    }, nil
}

type FeedPage struct {
    Items   []*FeedItem `json:"items"`
    Cursor  string      `json:"cursor"`
    HasMore bool        `json:"has_more"`
}
```

```go
// 用户级别限流: 每秒最多5条消息
func (l *RateLimiter) CheckUserRate(ctx context.Context, uid int64) (bool, error) {
    key := fmt.Sprintf("rate:user:%d", uid)
    count, _ := l.rdb.Incr(ctx, key).Result()
    if count == 1 {
        l.rdb.Expire(ctx, key, time.Second)
    }
    return count <= 5, nil
}

// 群级别限流: 每秒最多50条消息
func (l *RateLimiter) CheckGroupRate(ctx context.Context, groupID int64) (bool, error) {
    key := fmt.Sprintf("rate:group:%d", groupID)
    count, _ := l.rdb.Incr(ctx, key).Result()
    if count == 1 {
        l.rdb.Expire(ctx, key, time.Second)
    }
    return count <= 50, nil
}

// IP级别限流: 滑动窗口
func (l *RateLimiter) CheckIPRate(ctx context.Context, ip string) (bool, error) {
    key := fmt.Sprintf("rate:ip:%s", ip)
    now := time.Now().UnixMilli()
    windowStart := now - 60000 // 1分钟窗口

    pipe := l.rdb.Pipeline()
    pipe.ZRemRangeByScore(ctx, key, "0", fmt.Sprintf("%d", windowStart))
    pipe.ZAdd(ctx, key, &redis.Z{Score: float64(now), Member: now})
    pipe.ZCard(ctx, key)
    pipe.Expire(ctx, key, time.Minute)
    results, _ := pipe.Exec(ctx)

    count := results[2].(*redis.IntCmd).Val()
    return count <= 100, nil // 每分钟最多100条
}
```

```go
type ReactionService struct {
    rdb *redis.Client
}

func (s *ReactionService) AddReaction(ctx context.Context, convID string, msgID, uid int64, emoji string) error {
    key := fmt.Sprintf("reaction:%s:%d", convID, msgID)
    
    // 获取当前该emoji的用户列表
    val, _ := s.rdb.HGet(ctx, key, emoji).Result()
    var uids []int64
    if val != "" {
        json.Unmarshal([]byte(val), &uids)
    }
    
    // 去重添加
    for _, u := range uids {
        if u == uid {
            return nil // 已经reaction过
        }
    }
    uids = append(uids, uid)
    
    data, _ := json.Marshal(uids)
    return s.rdb.HSet(ctx, key, emoji, string(data)).Err()
}

func (s *ReactionService) GetReactions(ctx context.Context, convID string, msgID int64) (map[string][]int64, error) {
    key := fmt.Sprintf("reaction:%s:%d", convID, msgID)
    result, err := s.rdb.HGetAll(ctx, key).Result()
    if err != nil {
        return nil, err
    }
    reactions := make(map[string][]int64)
    for emoji, val := range result {
        var uids []int64
        json.Unmarshal([]byte(val), &uids)
        reactions[emoji] = uids
    }
    return reactions, nil
}
```

```go
func (s *IMServer) HandleBotMessage(ctx context.Context, msg *Message) {

    bot, err := s.botSvc.GetBot(ctx, msg.ToUID)

    if bot.Webhook != "" {

        reply, _ = s.callWebhook(ctx, bot.Webhook, msg)
    } else if bot.Handler != nil {

        reply, _ = bot.Handler.OnMessage(ctx, msg)
    }

    // 3. 机器人回复消息
    if reply != nil {
        reply.FromUID = bot.BotUID
        reply.ToUID = msg.FromUID
        reply.ConvID = msg.ConvID
        s.SendMessage(ctx, nil, &SendMsgReq{
            ConvID:  reply.ConvID,
            Content: reply.Content,
        })
    }
}
```

```go
type BenchmarkClient struct {

func RunBenchmark(gatewayAddr string, clientCount, msgPerSec int) {
    clients := make([]*BenchmarkClient, clientCount)
    
    // 1. 建立连接
    var wg sync.WaitGroup
    for i := 0; i < clientCount; i++ {
        wg.Add(1)
        go func(idx int) {

            ws, _, _ := websocket.DefaultDialer.Dial(gatewayAddr, nil)

            }
            // 鉴权
            auth, _ := json.Marshal(&Packet{Cmd: CmdAuth, Data: mustMarshal(map[string]interface{}{
                "token": fmt.Sprintf("test_token_%d", idx),
            })})
            ws.WriteMessage(websocket.TextMessage, auth)
        }(i)
    }
    wg.Wait()
    fmt.Printf("Connected %d clients\n", clientCount)

    // 2. 发送消息
    ticker := time.NewTicker(time.Second / time.Duration(msgPerSec))
    ctx, cancel := context.WithTimeout(context.Background(), 60*time.Second)
    defer cancel()

    go func() {
        for {
            select {
            case <-ctx.Done():
                return
            case <-ticker.C:
                // 随机选一个客户端发消息
                idx := rand.Intn(clientCount)
                toIdx := rand.Intn(clientCount)
                if idx == toIdx {
                    continue
                }
                msg, _ := json.Marshal(&Packet{Cmd: CmdMsg, Data: mustMarshal(map[string]interface{}{
                    "to_uid":  clients[toIdx].uid,
                    "content": fmt.Sprintf("bench msg %d", time.Now().UnixNano()),
                })})
                clients[idx].ws.WriteMessage(websocket.TextMessage, msg)
                clients[idx].sendCnt.Add(1)
            }
        }
    }()

    // 3. 统计
    <-ctx.Done()
    var totalSend, totalRecv int64
    for _, c := range clients {
        totalSend += c.sendCnt.Load()
        totalRecv += c.recvCnt.Load()
    }
    fmt.Printf("Total sent: %d, received: %d, delivery rate: %.2f%%\n",
        totalSend, totalRecv, float64(totalRecv)/float64(totalSend)*100)
}
```

| 维度 | 方案A | 方案B | 推荐 |
|------|-------|-------|------|
| 长连接 | WebSocket | TCP | Web用WS，移动端可TCP |

| Feed排序 | 时间序 | 算法推荐 | 关注流时间序 + 推荐流算法 |
| 推送 | 自建通道 | 第三方 | 在线自建 + 离线第三方 |
