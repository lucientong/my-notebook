### 1.2 各阶段核心职责

| 阶段 | 候选量级 | 延迟要求 | 核心目标 | 典型方法 |
|------|---------|---------|---------|---------|
| 召回 | 亿→万 | <50ms | 高召回率,不漏掉好结果 | 倒排索引/ANN/图检索 |
| 粗排 | 万→千 | <20ms | 快速筛选,计算效率优先 | 双塔模型/简单LR |
| 精排 | 千→百 | <50ms | 准确排序,效果优先 | GBDT/DNN/Cross特征 |
| 重排 | 百→十 | <10ms | 业务目标,多样性/时效性 | 规则+Listwise模型 |

### 1.3 搜索 vs 推荐对比

| 维度 | 搜索系统 | 推荐系统 |
|------|---------|---------|
| 用户意图 | 明确(有Query) | 模糊(无Query) |
| 触发方式 | 用户主动查询 | 系统主动推送 |
| 核心目标 | 相关性(Relevance) | 兴趣匹配(Interest) |
| 实时性要求 | 极高(毫秒级响应) | 相对宽松 |
| 冷启动 | Query理解可缓解 | 新用户冷启动严重 |
| 评价指标 | NDCG/MAP/MRR | CTR/CVR/停留时长 |
| 结果可解释性 | 高(关键词匹配可解释) | 较低 |
| 多样性需求 | 满足精准需求为主 | 需探索+利用平衡 |
| 召回方式 | 基于Query的多路召回 | 基于User的多路召回 |
| 时效性 | Query变则结果立即变 | 兴趣需要持续建模 |

### 1.4 搜索系统核心链路 Go 代码框架

```go
package search

import (
    "context"
    "sort"
    "time"
)

// Document 搜索文档
type Document struct {
    ID       int64
    Title    string
    Content  string
    Score    float64
    Features map[string]float64
}

// SearchRequest 搜索请求
type SearchRequest struct {
    Query    string
    UserID   int64
    Page     int
    PageSize int
    Filters  map[string]string
}

// SearchResponse 搜索响应
type SearchResponse struct {
    Total    int64
    Items    []*Document
    Suggest  []string
    CostMs   int64
}

// SearchEngine 搜索引擎核心
type SearchEngine struct {
    queryProcessor  QueryProcessor
    recallService   RecallService
    preRanker       PreRanker
    ranker          Ranker
    reRanker        ReRanker
}

// Search 搜索主流程
func (e *SearchEngine) Search(ctx context.Context, req *SearchRequest) (*SearchResponse, error) {
    start := time.Now()

    // 1. Query理解
    parsedQuery, err := e.queryProcessor.Process(ctx, req.Query)
    if err != nil {
        return nil, err
    }

    // 2. 多路召回
    candidates, err := e.recallService.Recall(ctx, parsedQuery, 10000)
    if err != nil {
        return nil, err
    }

    // 3. 粗排
    preRanked, err := e.preRanker.PreRank(ctx, parsedQuery, candidates, 500)
    if err != nil {
        return nil, err
    }

    // 4. 精排
    ranked, err := e.ranker.Rank(ctx, parsedQuery, preRanked, req.UserID)
    if err != nil {
        return nil, err
    }

    // 5. 重排 (多样性/业务规则)
    reRanked := e.reRanker.ReRank(ctx, ranked, req)

    // 6. 分页
    offset := req.Page * req.PageSize
    end := offset + req.PageSize
    if end > len(reRanked) {
        end = len(reRanked)
    }

    return &SearchResponse{
        Total:  int64(len(reRanked)),
        Items:  reRanked[offset:end],
        CostMs: time.Since(start).Milliseconds(),
    }, nil
}

// QueryProcessor Query理解接口
type QueryProcessor interface {
    Process(ctx context.Context, query string) (*ParsedQuery, error)
}

// ParsedQuery Query理解结果
type ParsedQuery struct {
    Original    string
    Corrected   string            // 纠错后Query
    Tokens      []string          // 分词结果
    Synonyms    map[string][]string // 同义词
    Intent      string            // 意图识别
    Rewritten   []string          // 改写后的Query列表
    Entities    []Entity          // 实体识别
}

// Entity 实体
type Entity struct {
    Text  string
    Type  string // brand/category/color 等
    Start int
    End   int
}

// RecallService 召回服务接口
type RecallService interface {
    Recall(ctx context.Context, query *ParsedQuery, topN int) ([]*Document, error)
}

// PreRanker 粗排接口
type PreRanker interface {
    PreRank(ctx context.Context, query *ParsedQuery, docs []*Document, topN int) ([]*Document, error)
}

// Ranker 精排接口
type Ranker interface {
    Rank(ctx context.Context, query *ParsedQuery, docs []*Document, userID int64) ([]*Document, error)
}

// ReRanker 重排接口
type ReRanker interface {
    ReRank(ctx context.Context, docs []*Document, req *SearchRequest) []*Document
}
```

| jieba(综合) | 字典+HMM+Viterbi | 工业常用 | 领域适配需定制 |

```python
def tokenize_query(query: str, use_search_mode: bool = True) -> list[str]:
    """搜索分词 - 使用搜索引擎模式"""
    if use_search_mode:
        # 搜索模式: 在精确模式基础上对长词再次切分

def load_stopwords(filepath: str) -> set:
```

#### 2.3.2 编辑距离纠错实现 (Go)

```go
package spell

import "unicode/utf8"

// EditDistance 计算编辑距离(Levenshtein)
func EditDistance(s1, s2 string) int {
    r1 := []rune(s1)
    r2 := []rune(s2)
    m, n := len(r1), len(r2)

    dp := make([][]int, m+1)
    for i := range dp {
        dp[i] = make([]int, n+1)
        dp[i][0] = i
    }
    for j := 0; j <= n; j++ {
        dp[0][j] = j
    }

    for i := 1; i <= m; i++ {
        for j := 1; j <= n; j++ {
            if r1[i-1] == r2[j-1] {
                dp[i][j] = dp[i-1][j-1]
            } else {
                dp[i][j] = min3(
                    dp[i-1][j]+1,   // 删除
                    dp[i][j-1]+1,   // 插入
                    dp[i-1][j-1]+1, // 替换
                )
            }
        }
    }
    return dp[m][n]
}

func min3(a, b, c int) int {
    if a < b {
        if a < c {
            return a
        }
        return c
    }
    if b < c {
        return b
    }
    return c
}

// SpellCorrector 拼写纠错器
type SpellCorrector struct {
    dictionary map[string]int // 词 -> 词频
}

func NewSpellCorrector(dict map[string]int) *SpellCorrector {
    return &SpellCorrector{dictionary: dict}
}

// Correct 返回最可能的纠正词
func (sc *SpellCorrector) Correct(word string) string {
    // 如果词在词典中,不需要纠错
    if _, ok := sc.dictionary[word]; ok {
        return word
    }

    bestWord := word
    bestScore := 0
    maxDist := 2 // 最大编辑距离

    for candidate, freq := range sc.dictionary {
        dist := EditDistance(word, candidate)
        if dist <= maxDist && freq > bestScore {
            bestWord = candidate
            bestScore = freq
        }
    }
    return bestWord
}

// CorrectQuery 纠错整个Query
func (sc *SpellCorrector) CorrectQuery(tokens []string) []string {
    result := make([]string, len(tokens))
    for i, token := range tokens {
        result[i] = sc.Correct(token)
    }
    return result
}
```

### 2.4 同义词扩展

```python
from typing import Dict, List, Set

class SynonymExpander:
    """同义词扩展器"""

    def __init__(self):
        # 同义词组: 同一组内的词互为同义词
        self.synonym_groups: List[Set[str]] = []
        self.word_to_group: Dict[str, int] = {}

    def load_synonyms(self, filepath: str):
        """加载同义词表
        格式: 每行一组同义词, 空格分隔
        示例: 手机 手提电话 移动电话 cell_phone
        """
        with open(filepath, "r", encoding="utf-8") as f:
            for line in f:
                words = set(line.strip().split())
                if len(words) < 2:
                    continue
                group_id = len(self.synonym_groups)
                self.synonym_groups.append(words)
                for w in words:
                    self.word_to_group[w] = group_id

    def expand(self, word: str) -> List[str]:
        """获取同义词列表"""
        if word not in self.word_to_group:
            return [word]
        group_id = self.word_to_group[word]
        return list(self.synonym_groups[group_id])

    def expand_query(self, tokens: List[str]) -> List[List[str]]:
        """扩展Query中每个词的同义词"""
        return [self.expand(token) for token in tokens]


# 示例
expander = SynonymExpander()
expander.synonym_groups = [
    {"手机", "移动电话", "cellphone"},
    {"便宜", "廉价", "实惠"},
]
expander.word_to_group = {
    "手机": 0, "移动电话": 0, "cellphone": 0,
    "便宜": 1, "廉价": 1, "实惠": 1,
}
print(expander.expand("手机"))
# ['手机', '移动电话', 'cellphone']
```

#### 2.5.2 意图识别模型 (Python)

```python
import torch
import torch.nn as nn
from transformers import BertTokenizer, BertModel

class IntentClassifier(nn.Module):
    """基于BERT的意图识别模型"""

    def __init__(self, num_intents: int, bert_model: str = "bert-base-chinese"):
        super().__init__()
        self.bert = BertModel.from_pretrained(bert_model)
        self.dropout = nn.Dropout(0.1)
        self.classifier = nn.Linear(self.bert.config.hidden_size, num_intents)

    def forward(self, input_ids, attention_mask):
        outputs = self.bert(input_ids=input_ids, attention_mask=attention_mask)
        pooled = outputs.pooler_output  # [CLS] token
        pooled = self.dropout(pooled)
        logits = self.classifier(pooled)
        return logits


class IntentService:
    """意图识别服务"""

    INTENTS = ["navigational", "informational", "transactional", "local"]

    def __init__(self, model_path: str):
        self.tokenizer = BertTokenizer.from_pretrained("bert-base-chinese")
        self.model = IntentClassifier(num_intents=len(self.INTENTS))
        self.model.load_state_dict(torch.load(model_path, map_location="cpu"))
        self.model.eval()

    def predict(self, query: str) -> dict:
        inputs = self.tokenizer(
            query, return_tensors="pt",
            max_length=64, truncation=True, padding="max_length"
        )
        with torch.no_grad():
            logits = self.model(inputs["input_ids"], inputs["attention_mask"])
            probs = torch.softmax(logits, dim=-1).squeeze()

        result = {}
        for i, intent in enumerate(self.INTENTS):
            result[intent] = round(probs[i].item(), 4)
        return result
```

### 2.6 Query改写

#### 2.6.1 Query改写策略

| 策略 | 方法 | 示例 |
|------|------|------|
| 同义替换 | 同义词表替换 | 手机→移动电话 |
| 纠错改写 | 拼写纠正 | ipone→iphone |
| 扩展改写 | 添加相关词 | iphone→iphone手机 |
| 缩写改写 | 去除无效词 | 好看的红色连衣裙→红色连衣裙 |
| 实体改写 | 实体标准化 | 苹果13→iPhone 13 |
| 关联改写 | Session/点击日志挖掘 | 对相机→单反 |

```python
from dataclasses import dataclass
from typing import List, Optional

@dataclass
class RewriteResult:
    original: str
    rewrites: List[str]
    strategy: str
    confidence: float

class QueryRewriter:
    """Query改写引擎"""

    def __init__(self, synonym_dict, entity_normalizer, click_graph):
        self.synonym_dict = synonym_dict
        self.entity_normalizer = entity_normalizer
        self.click_graph = click_graph  # 基于点击日志构建的关联图

    def rewrite(self, query: str, tokens: List[str]) -> List[RewriteResult]:
        results = []

        # 策略1: 同义词改写
        syn_rewrites = self._synonym_rewrite(query, tokens)
        if syn_rewrites:
            results.append(syn_rewrites)

        # 策略2: 实体标准化
        entity_rewrite = self._entity_rewrite(query)
        if entity_rewrite:
            results.append(entity_rewrite)

        # 策略3: 点击图改写(挖掘session日志)
        click_rewrites = self._click_graph_rewrite(query)
        if click_rewrites:
            results.append(click_rewrites)

        # 策略4: 删除无效词
        trim_rewrite = self._trim_rewrite(query, tokens)
        if trim_rewrite:
            results.append(trim_rewrite)

        return results

    def _synonym_rewrite(self, query: str, tokens: List[str]) -> Optional[RewriteResult]:
        rewrites = []
        for i, token in enumerate(tokens):
            syns = self.synonym_dict.get(token, [])
            for syn in syns:
                new_tokens = tokens[:i] + [syn] + tokens[i+1:]
                rewrites.append("".join(new_tokens))
        if rewrites:
            return RewriteResult(query, rewrites, "synonym", 0.8)
        return None

    def _entity_rewrite(self, query: str) -> Optional[RewriteResult]:
        normalized = self.entity_normalizer.normalize(query)
        if normalized != query:
            return RewriteResult(query, [normalized], "entity", 0.9)
        return None

    def _click_graph_rewrite(self, query: str) -> Optional[RewriteResult]:
        """基于用户点击行为挖掘query间关联"""
        related = self.click_graph.get_related_queries(query, top_k=3)
        if related:
            return RewriteResult(query, related, "click_graph", 0.7)
        return None

    def _trim_rewrite(self, query: str, tokens: List[str]) -> Optional[RewriteResult]:
        """去除无效修饰词"""
        useless_words = {"好看的", "最好的", "推荐", "有没有"}
        trimmed = [t for t in tokens if t not in useless_words]
        if len(trimmed) < len(tokens):
            return RewriteResult(query, ["".join(trimmed)], "trim", 0.6)
        return None
```

### 3.2 倒排索引召回

#### 3.2.1 倒排索引原理

```
正排索引 (Forward Index):
  Doc1 → ["搜索", "引擎", "技术"]
  Doc2 → ["搜索", "算法", "排序"]
  Doc3 → ["引擎", "优化"]

倒排索引 (Inverted Index):
  "搜索" → [Doc1, Doc2]
  "引擎" → [Doc1, Doc3]
  "技术" → [Doc1]
  "算法" → [Doc2]
  "排序" → [Doc2]
  "优化" → [Doc3]
```

#### 3.2.2 倒排索引实现 (Go)

```go
package index

import (
    "math"
    "sort"
    "strings"
    "sync"
)

// Posting 倒排列表项
type Posting struct {
    DocID     int64
    TermFreq  int     // 词频 TF
    Positions []int   // 词在文档中的位置
}

// PostingList 倒排列表
type PostingList []*Posting

// InvertedIndex 倒排索引
type InvertedIndex struct {
    mu       sync.RWMutex
    index    map[string]PostingList  // term -> posting list
    docCount int64                   // 文档总数
    docLen   map[int64]int           // 文档ID -> 文档长度
    avgDL    float64                 // 平均文档长度
}

func NewInvertedIndex() *InvertedIndex {
    return &InvertedIndex{
        index:  make(map[string]PostingList),
        docLen: make(map[int64]int),
    }
}

// AddDocument 添加文档到索引
func (idx *InvertedIndex) AddDocument(docID int64, tokens []string) {
    idx.mu.Lock()
    defer idx.mu.Unlock()

    // 统计词频和位置
    termFreq := make(map[string][]int)
    for pos, token := range tokens {
        termFreq[token] = append(termFreq[token], pos)
    }

    // 更新倒排索引
    for term, positions := range termFreq {
        posting := &Posting{
            DocID:     docID,
            TermFreq:  len(positions),
            Positions: positions,
        }
        idx.index[term] = append(idx.index[term], posting)
    }

    // 更新文档统计
    idx.docLen[docID] = len(tokens)
    idx.docCount++

    // 重新计算平均文档长度
    totalLen := 0
    for _, l := range idx.docLen {
        totalLen += l
    }
    idx.avgDL = float64(totalLen) / float64(idx.docCount)
}

// Search 基于BM25的搜索
func (idx *InvertedIndex) Search(queryTokens []string, topK int) []ScoredDoc {
    idx.mu.RLock()
    defer idx.mu.RUnlock()

    scores := make(map[int64]float64)

    for _, token := range queryTokens {
        postings, exists := idx.index[token]
        if !exists {
            continue
        }

        // IDF
        df := float64(len(postings))
        idf := math.Log((float64(idx.docCount)-df+0.5) / (df + 0.5))

        for _, p := range postings {
            // BM25 scoring
            tf := float64(p.TermFreq)
            dl := float64(idx.docLen[p.DocID])
            k1 := 1.2
            b := 0.75

            tfNorm := (tf * (k1 + 1)) / (tf + k1*(1-b+b*dl/idx.avgDL))
            scores[p.DocID] += idf * tfNorm
        }
    }

    // 排序取TopK
    result := make([]ScoredDoc, 0, len(scores))
    for docID, score := range scores {
        result = append(result, ScoredDoc{DocID: docID, Score: score})
    }
    sort.Slice(result, func(i, j int) bool {
        return result[i].Score > result[j].Score
    })
    if len(result) > topK {
        result = result[:topK]
    }
    return result
}

type ScoredDoc struct {
    DocID int64
    Score float64
}

// BooleanSearch 布尔检索 (AND / OR)
func (idx *InvertedIndex) BooleanSearch(queryTokens []string, op string) []int64 {
    idx.mu.RLock()
    defer idx.mu.RUnlock()

    if len(queryTokens) == 0 {
        return nil
    }

    if op == "AND" {
        return idx.intersect(queryTokens)
    }
    return idx.union(queryTokens)
}

func (idx *InvertedIndex) intersect(tokens []string) []int64 {
    // 获取每个词的docID列表(已排序)
    lists := make([][]int64, 0)
    for _, token := range tokens {
        postings := idx.index[token]
        if len(postings) == 0 {
            return nil // AND 操作: 有一个词不存在则无结果
        }
        docIDs := make([]int64, len(postings))
        for i, p := range postings {
            docIDs[i] = p.DocID
        }
        sort.Slice(docIDs, func(i, j int) bool { return docIDs[i] < docIDs[j] })
        lists = append(lists, docIDs)
    }

    // 多路归并求交集
    result := lists[0]
    for i := 1; i < len(lists); i++ {
        result = intersectTwo(result, lists[i])
    }
    return result
}

func intersectTwo(a, b []int64) []int64 {
    var result []int64
    i, j := 0, 0
    for i < len(a) && j < len(b) {
        if a[i] == b[j] {
            result = append(result, a[i])
            i++
            j++
        } else if a[i] < b[j] {
            i++
        } else {
            j++
        }
    }
    return result
}

func (idx *InvertedIndex) union(tokens []string) []int64 {
    seen := make(map[int64]bool)
    for _, token := range tokens {
        for _, p := range idx.index[token] {
            seen[p.DocID] = true
        }
    }
    result := make([]int64, 0, len(seen))
    for docID := range seen {
        result = append(result, docID)
    }
    sort.Slice(result, func(i, j int) bool { return result[i] < result[j] })
    return result
}
```

### 3.3 向量召回 (ANN)

#### 3.3.1 向量召回原理

```
离线阶段:
  文档 → Encoder → 文档向量 → 存入向量索引(FAISS/Milvus)

在线阶段:
  Query → Encoder → Query向量 → ANN检索 → TopK文档
```

#### 3.3.2 ANN算法对比

| 算法 | 原理 | 时间复杂度 | 内存 | 精度 | 适用场景 |
|------|------|-----------|------|------|---------|
| 暴力检索 | 全量计算距离 | O(N*D) | O(N*D) | 100% | 小规模 |
| KD-Tree | 空间划分树 | O(logN) | O(N*D) | 高 | 低维(<20) |
| LSH | 局部敏感哈希 | O(1)~O(N) | O(N*D*L) | 中等 | 高维/大规模 |
| IVF | 聚类+倒排 | O(N/K*nprobe) | O(N*D) | 可调 | 大规模 |
| HNSW | 层次化小世界图 | O(logN) | O(N*D*M) | 高 | 通用首选 |
| PQ | 乘积量化 | O(N/K) | O(N*M) | 中等 | 超大规模+省内存 |
| IVF-PQ | IVF+PQ组合 | O(N/K*nprobe) | O(N*M) | 可调 | 工业界常用 |

#### 3.3.3 HNSW原理

```
Layer 2:  O ────────────────────── O     (最稀疏,长程连接)
          │                        │
Layer 1:  O ──── O ──── O ──── O ──O     (中等密度)
          │      │      │      │   │
Layer 0:  O─O─O─O─O─O─O─O─O─O─O─O─O     (最密集,所有节点)

搜索过程:
1. 从最高层开始 → 贪心找最近邻
2. 下降到下一层 → 继续贪心搜索
3. 在第0层返回TopK结果
```

#### 3.3.4 FAISS向量检索 (Python)

```python
import faiss
import numpy as np
from typing import List, Tuple

class VectorIndex:
    """基于FAISS的向量召回"""

    def __init__(self, dim: int, index_type: str = "hnsw"):
        self.dim = dim
        self.index_type = index_type
        self.index = self._create_index(dim, index_type)
        self.id_map: List[int] = []  # faiss内部ID -> 业务docID

    def _create_index(self, dim: int, index_type: str) -> faiss.Index:
        if index_type == "flat":
            # 暴力搜索(精确)
            return faiss.IndexFlatIP(dim)  # 内积距离
        elif index_type == "ivf":
            # IVF 倒排索引
            nlist = 100  # 聚类中心数
            quantizer = faiss.IndexFlatIP(dim)
            index = faiss.IndexIVFFlat(quantizer, dim, nlist)
            return index
        elif index_type == "hnsw":
            # HNSW
            M = 32  # 每个节点最大连接数
            index = faiss.IndexHNSWFlat(dim, M)
            index.hnsw.efConstruction = 200  # 构建时搜索宽度
            index.hnsw.efSearch = 64         # 搜索时宽度
            return index
        elif index_type == "ivf_pq":
            # IVF + PQ 量化(大规模省内存)
            nlist = 1024
            m = 32        # 子向量数
            nbits = 8     # 每个子向量编码位数
            quantizer = faiss.IndexFlatIP(dim)
            index = faiss.IndexIVFPQ(quantizer, dim, nlist, m, nbits)
            return index
        else:
            raise ValueError(f"Unknown index type: {index_type}")

    def build(self, vectors: np.ndarray, doc_ids: List[int]):
        """构建索引"""
        assert vectors.shape[1] == self.dim
        # 归一化(用于内积 = 余弦相似度)
        faiss.normalize_L2(vectors)

        # 需要训练的索引类型
        if hasattr(self.index, 'train') and not self.index.is_trained:
            self.index.train(vectors)

        self.index.add(vectors)
        self.id_map = doc_ids

    def search(self, query_vec: np.ndarray, top_k: int = 100) -> List[Tuple[int, float]]:
        """检索TopK"""
        query_vec = query_vec.reshape(1, -1).astype(np.float32)
        faiss.normalize_L2(query_vec)

        if hasattr(self.index, 'nprobe'):
            self.index.nprobe = 10  # IVF 探测聚类数

        scores, indices = self.index.search(query_vec, top_k)

        results = []
        for score, idx in zip(scores[0], indices[0]):
            if idx == -1:
                continue
            doc_id = self.id_map[idx]
            results.append((doc_id, float(score)))
        return results
```

```go
package recall

// GraphRecall 基于图的召回
// 适用场景: 商品知识图谱、用户行为图
type GraphRecall struct {
    // 邻接表: nodeID -> [(neighborID, edgeType, weight)]
    adjList map[int64][]Edge
}

type Edge struct {
    TargetID int64
    EdgeType string  // "also_buy", "similar", "same_brand"
    Weight   float64
}

func NewGraphRecall() *GraphRecall {
    return &GraphRecall{
        adjList: make(map[int64][]Edge),
    }
}

func (g *GraphRecall) AddEdge(from, to int64, edgeType string, weight float64) {
    g.adjList[from] = append(g.adjList[from], Edge{
        TargetID: to,
        EdgeType: edgeType,
        Weight:   weight,
    })
}

// Recall 从种子节点出发,BFS扩展召回
func (g *GraphRecall) Recall(seedIDs []int64, maxDepth int, topK int) []ScoredDoc {
    scores := make(map[int64]float64)
    visited := make(map[int64]bool)

    type QueueItem struct {
        ID    int64
        Depth int
        Score float64
    }

    queue := make([]QueueItem, 0)
    for _, id := range seedIDs {
        queue = append(queue, QueueItem{ID: id, Depth: 0, Score: 1.0})
        visited[id] = true
    }

    for len(queue) > 0 {
        item := queue[0]
        queue = queue[1:]

        if item.Depth >= maxDepth {
            continue
        }

        for _, edge := range g.adjList[item.ID] {
            neighborScore := item.Score * edge.Weight * decayFactor(item.Depth)
            scores[edge.TargetID] += neighborScore

            if !visited[edge.TargetID] {
                visited[edge.TargetID] = true
                queue = append(queue, QueueItem{
                    ID:    edge.TargetID,
                    Depth: item.Depth + 1,
                    Score: neighborScore,
                })
            }
        }
    }

    // 排序返回TopK
    result := make([]ScoredDoc, 0, len(scores))
    for id, score := range scores {
        result = append(result, ScoredDoc{DocID: id, Score: score})
    }
    sort.Slice(result, func(i, j int) bool {
        return result[i].Score > result[j].Score
    })
    if len(result) > topK {
        result = result[:topK]
    }
    return result
}

func decayFactor(depth int) float64 {
    // 距离衰减: 越远的邻居权重越低
    return 1.0 / float64(depth+1)
}
```

```go
package recall

import (
    "context"
    "math"
    "sort"
    "sync"

// MultiRecallService 多路召回服务
type MultiRecallService struct {
    invertedRecall *InvertedRecall
    vectorRecall   *VectorRecall
    graphRecall    *GraphRecall
}

// RecallResult 单路召回结果
type RecallResult struct {
    Source string
    Docs   []ScoredDoc
}

// Recall 多路召回
func (m *MultiRecallService) Recall(ctx context.Context, query *ParsedQuery, topN int) ([]*Document, error) {
    var (
        wg      sync.WaitGroup
        mu      sync.Mutex
        results []RecallResult
    )

    // 并行多路召回
    recallFuncs := []struct {
        name string
        fn   func() []ScoredDoc
    }{
        {"inverted", func() []ScoredDoc {
            return m.invertedRecall.Search(query.Tokens, topN)
        }},
        {"vector", func() []ScoredDoc {
            return m.vectorRecall.Search(query.Original, topN)
        }},
        {"graph", func() []ScoredDoc {
            // 用Query命中的实体作为种子节点
            seedIDs := extractEntityIDs(query.Entities)
            return m.graphRecall.Recall(seedIDs, 2, topN)
        }},
    }

    for _, rf := range recallFuncs {
        wg.Add(1)
        go func(name string, fn func() []ScoredDoc) {
            defer wg.Done()
            docs := fn()
            mu.Lock()
            results = append(results, RecallResult{Source: name, Docs: docs})
            mu.Unlock()
        }(rf.name, rf.fn)
    }
    wg.Wait()

    // 融合: 分数归一化 + 加权合并
    return mergeResults(results, topN), nil
}

// mergeResults 融合多路召回结果
func mergeResults(results []RecallResult, topN int) []*Document {
    // 各路权重
    sourceWeights := map[string]float64{
        "inverted": 0.4,
        "vector":   0.4,
        "graph":    0.2,
    }

    mergedScores := make(map[int64]float64)

    for _, r := range results {
        // MinMax归一化
        normalized := minMaxNormalize(r.Docs)
        weight := sourceWeights[r.Source]

        for _, doc := range normalized {
            mergedScores[doc.DocID] += doc.Score * weight
        }
    }

    // 排序
    result := make([]ScoredDoc, 0, len(mergedScores))
    for docID, score := range mergedScores {
        result = append(result, ScoredDoc{DocID: docID, Score: score})
    }
    sort.Slice(result, func(i, j int) bool {
        return result[i].Score > result[j].Score
    })
    if len(result) > topN {
        result = result[:topN]
    }

    // 转换为Document
    docs := make([]*Document, len(result))
    for i, sd := range result {
        docs[i] = &Document{ID: sd.DocID, Score: sd.Score}
    }
    return docs
}

// minMaxNormalize MinMax归一化到[0,1]
func minMaxNormalize(docs []ScoredDoc) []ScoredDoc {
    if len(docs) == 0 {
        return docs
    }
    minScore, maxScore := docs[0].Score, docs[0].Score
    for _, d := range docs {
        if d.Score < minScore {
            minScore = d.Score
        }
        if d.Score > maxScore {
            maxScore = d.Score
        }
    }
    rangeScore := maxScore - minScore
    if rangeScore == 0 {
        rangeScore = 1
    }
    result := make([]ScoredDoc, len(docs))
    for i, d := range docs {
        result[i] = ScoredDoc{
            DocID: d.DocID,
            Score: (d.Score - minScore) / rangeScore,
        }
    }
    return result
}
```

## 4. 排序模型 ⭐⭐⭐

### 4.1 BM25 详解

#### 4.1.1 BM25 公式

```
BM25(Q, D) = Σ IDF(qi) * TF(qi, D)

其中:
  IDF(qi) = log((N - df(qi) + 0.5) / (df(qi) + 0.5) + 1)

  TF(qi, D) = f(qi, D) * (k1 + 1) / (f(qi, D) + k1 * (1 - b + b * |D| / avgdl))

参数:
  N     = 文档总数
  df(qi) = 包含词qi的文档数
  f(qi,D) = 词qi在文档D中的词频
  |D|   = 文档D的长度
  avgdl = 所有文档的平均长度
  k1    = 1.2 (控制TF饱和度)
  b     = 0.75 (控制文档长度归一化程度)
```

#### 4.1.2 BM25 参数影响

| 参数 | 默认值 | 作用 | 调参方向 |
|------|-------|------|---------|
| k1 | 1.2 | TF饱和速度 | 增大→TF影响增大 |
| b | 0.75 | 文档长度惩罚 | b=0不惩罚,b=1完全惩罚 |

### 4.2 Learning to Rank (LTR)

#### 4.2.1 三大范式对比

| 范式 | 损失函数输入 | 代表算法 | 优点 | 缺点 |
|------|------------|---------|------|------|
| Pointwise | 单个文档的相关性分数 | 回归/分类模型 | 简单直接 | 忽略文档间关系 |
| Pairwise | 文档对的相对顺序 | RankSVM, RankNet, LambdaRank | 考虑相对顺序 | 文档对数量大 |
| Listwise | 整个排序列表 | ListNet, LambdaMART, SoftRank | 直接优化排序指标 | 训练复杂 |

#### 4.2.2 Pointwise: 逻辑回归排序

```python
        features.append(exact_match(query, doc["title"]))         # 标题精确匹配

        features.append(doc.get("page_rank", 0))                  # PageRank
        features.append(doc.get("click_count", 0))                # 历史点击数
        features.append(doc.get("freshness", 0))                  # 时效性分数

        # 用户行为特征
        features.append(doc.get("ctr", 0))                        # 历史CTR
        features.append(doc.get("avg_dwell_time", 0))             # 平均停留时间

        return np.array(features)

        """训练: y为0/1标签(不相关/相关)"""
        self.model.fit(X, y)

        """预测相关性概率"""
        return self.model.predict_proba(X)[:, 1]

    def rank(self, query: str, docs: List[dict]) -> List[dict]:
        """对文档列表排序"""
        features = np.array([self.extract_features(query, d) for d in docs])
        scores = self.predict(features)
        ranked_indices = np.argsort(-scores)
        return [docs[i] for i in ranked_indices]
```

#### 4.2.3 Pairwise: RankNet (Python)

```python
import torch
import torch.nn as nn
import torch.optim as optim

class RankNet(nn.Module):
    """RankNet: Pairwise排序模型"""

    def __init__(self, input_dim: int, hidden_dim: int = 128):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(hidden_dim, hidden_dim // 2),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(hidden_dim // 2, 1),
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """输出单文档分数"""
        return self.net(x)

    def compute_loss(self, score_i: torch.Tensor, score_j: torch.Tensor,
                     label: torch.Tensor) -> torch.Tensor:
        """
        RankNet Loss
        label: 1 if doc_i > doc_j, 0 if doc_i < doc_j, 0.5 if equal
        """
        sigma = 1.0
        s_ij = sigma * (score_i - score_j)
        loss = -label * s_ij + torch.log(1 + torch.exp(s_ij))
        return loss.mean()


class RankNetTrainer:
    """RankNet训练器"""

    def __init__(self, input_dim: int, lr: float = 1e-3):
        self.model = RankNet(input_dim)
        self.optimizer = optim.Adam(self.model.parameters(), lr=lr)

    def train_epoch(self, pairs: list):
        """
        pairs: [(features_i, features_j, label), ...]
        label: 1 if doc_i更相关
        """
        self.model.train()
        total_loss = 0

        for feat_i, feat_j, label in pairs:
            feat_i = torch.FloatTensor(feat_i).unsqueeze(0)
            feat_j = torch.FloatTensor(feat_j).unsqueeze(0)
            label_t = torch.FloatTensor([label])

            score_i = self.model(feat_i)
            score_j = self.model(feat_j)

            loss = self.model.compute_loss(score_i, score_j, label_t)

            self.optimizer.zero_grad()
            loss.backward()
            self.optimizer.step()

            total_loss += loss.item()

        return total_loss / len(pairs)

    def predict(self, features: np.ndarray) -> np.ndarray:
        self.model.eval()
        with torch.no_grad():
            scores = self.model(torch.FloatTensor(features))
        return scores.numpy().flatten()
```

```python
            "verbose": -1,
        }

    def train(self, X_train: np.ndarray, y_train: np.ndarray,
              group_train: List[int],  # 每个query对应的文档数
              X_val: np.ndarray = None, y_val: np.ndarray = None,
              group_val: List[int] = None):
        """
        训练LambdaMART
        y_train: 相关性等级 (0-4)
        group_train: [20, 15, 30, ...] 每个query有多少个候选文档
        """
        train_data = lgb.Dataset(
            X_train, label=y_train, group=group_train
        )

        valid_sets = []
        if X_val is not None:
            val_data = lgb.Dataset(X_val, label=y_val, group=group_val)
            valid_sets = [val_data]

        self.model = lgb.train(
            self.params,
            train_data,
            num_boost_round=500,
            valid_sets=valid_sets,
            callbacks=[lgb.early_stopping(50), lgb.log_evaluation(100)],
        )

    def predict(self, X: np.ndarray) -> np.ndarray:
        return self.model.predict(X)

    def rank(self, query_features: np.ndarray, doc_ids: List[int]) -> List[int]:
        """对一组候选文档排序,返回排序后的doc_ids"""
        scores = self.predict(query_features)
        sorted_indices = np.argsort(-scores)
        return [doc_ids[i] for i in sorted_indices]

    def feature_importance(self, importance_type: str = "gain") -> dict:
        """特征重要性"""
        importance = self.model.feature_importance(importance_type=importance_type)
        names = self.model.feature_name()
        return dict(sorted(zip(names, importance), key=lambda x: -x[1]))
```

```go
        brandPreference(profile, doc.Brand),               // 品牌偏好
        pricePreference(profile, doc.Price),               // 价格偏好
    )

    // === 上下文特征 ===
    now := time.Now()
    features = append(features,
        float64(now.Hour()),                               // 小时
        boolToFloat(isWeekend(now)),                       // 是否周末
    )

    return features

func boolToFloat(b bool) float64 {
    if b {
        return 1.0
    }
    return 0.0
}

func termCoverage(queryTokens []string, text string) float64 {
    if len(queryTokens) == 0 {
        return 0
    }
    hit := 0
    lower := strings.ToLower(text)
    for _, token := range queryTokens {
        if strings.Contains(lower, strings.ToLower(token)) {
            hit++
        }
    }
    return float64(hit) / float64(len(queryTokens))
}

    hours := time.Since(updatedAt).Hours()
    return 1.0 / (1.0 + hours/24.0) // 越新越接近1
}

    return t.Weekday() == time.Saturday || t.Weekday() == time.Sunday
```

#### 4.4.2 Cross Network (DCN) 精排模型

```python
import torch
import torch.nn as nn

class CrossNetwork(nn.Module):
    """Cross Network: 显式特征交叉"""

    def __init__(self, input_dim: int, num_layers: int):
        super().__init__()
        self.num_layers = num_layers
        self.W = nn.ParameterList([
            nn.Parameter(torch.randn(input_dim, 1) * 0.01)
            for _ in range(num_layers)
        ])
        self.b = nn.ParameterList([
            nn.Parameter(torch.zeros(input_dim))
            for _ in range(num_layers)
        ])

    def forward(self, x0: torch.Tensor) -> torch.Tensor:
        x = x0
        for i in range(self.num_layers):
            # x_{l+1} = x0 * (x_l^T * W_l) + b_l + x_l
            xw = torch.matmul(x, self.W[i])          # (batch, 1)
            x = x0 * xw + self.b[i] + x              # (batch, dim)
        return x


class DCNv2(nn.Module):
    """DCN v2: Deep & Cross Network for Ranking"""

    def __init__(self, input_dim: int, cross_layers: int = 3,
                 deep_layers: list = [256, 128, 64]):
        super().__init__()
        # Cross部分
        self.cross_net = CrossNetwork(input_dim, cross_layers)

        # Deep部分
        layers = []
        prev_dim = input_dim
        for dim in deep_layers:
            layers.extend([
                nn.Linear(prev_dim, dim),
                nn.ReLU(),
                nn.BatchNorm1d(dim),
                nn.Dropout(0.1),
            ])
            prev_dim = dim
        self.deep_net = nn.Sequential(*layers)

        # 融合层
        self.output_layer = nn.Linear(input_dim + deep_layers[-1], 1)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        cross_out = self.cross_net(x)           # (batch, input_dim)
        deep_out = self.deep_net(x)             # (batch, 64)
        combined = torch.cat([cross_out, deep_out], dim=-1)
        logit = self.output_layer(combined)     # (batch, 1)
        return torch.sigmoid(logit).squeeze(-1)
```

```go
package index

import (
    "sync"
    "time"
)

// RealtimeIndex 实时索引(双Buffer切换)
type RealtimeIndex struct {
    mu         sync.RWMutex
    baseIndex  *InvertedIndex  // 基础索引(全量)
    bufferA    *InvertedIndex  // 增量Buffer A
    bufferB    *InvertedIndex  // 增量Buffer B
    activeIdx  int             // 当前活跃Buffer (0=A, 1=B)
}

func NewRealtimeIndex(base *InvertedIndex) *RealtimeIndex {
    return &RealtimeIndex{
        baseIndex: base,
        bufferA:   NewInvertedIndex(),
        bufferB:   NewInvertedIndex(),
        activeIdx: 0,
    }
}

// AddDocument 实时添加文档
func (ri *RealtimeIndex) AddDocument(docID int64, tokens []string) {
    ri.mu.RLock()
    defer ri.mu.RUnlock()

    active := ri.getActiveBuffer()
    active.AddDocument(docID, tokens)
}

// Search 搜索: 同时查询基础索引和增量Buffer
func (ri *RealtimeIndex) Search(queryTokens []string, topK int) []ScoredDoc {
    ri.mu.RLock()
    defer ri.mu.RUnlock()

    // 从基础索引和活跃Buffer分别搜索
    baseResults := ri.baseIndex.Search(queryTokens, topK)
    bufferResults := ri.getActiveBuffer().Search(queryTokens, topK)

    // 合并去重
    return mergeAndDedup(baseResults, bufferResults, topK)
}

// SwapBuffer 切换Buffer(定期触发)
func (ri *RealtimeIndex) SwapBuffer() {
    ri.mu.Lock()
    defer ri.mu.Unlock()

    // 将活跃Buffer合并到基础索引
    // 切换到另一个空Buffer
    ri.activeIdx = 1 - ri.activeIdx

    // 异步清空旧Buffer
    go func() {
        old := ri.getInactiveBuffer()
        // merge old into baseIndex ...
        *old = *NewInvertedIndex()
    }()
}

func (ri *RealtimeIndex) getActiveBuffer() *InvertedIndex {
    if ri.activeIdx == 0 {
        return ri.bufferA
    }
    return ri.bufferB
}

func (ri *RealtimeIndex) getInactiveBuffer() *InvertedIndex {
    if ri.activeIdx == 0 {
        return ri.bufferB
    }
    return ri.bufferA
}

func mergeAndDedup(a, b []ScoredDoc, topK int) []ScoredDoc {
    seen := make(map[int64]float64)
    for _, d := range a {
        seen[d.DocID] = d.Score
    }
    for _, d := range b {
        if existing, ok := seen[d.DocID]; ok {
            if d.Score > existing {
                seen[d.DocID] = d.Score
            }
        } else {
            seen[d.DocID] = d.Score
        }
    }

    result := make([]ScoredDoc, 0, len(seen))
    for id, score := range seen {
        result = append(result, ScoredDoc{DocID: id, Score: score})
    }
    sort.Slice(result, func(i, j int) bool {
        return result[i].Score > result[j].Score
    })
    if len(result) > topK {
        result = result[:topK]
    }
    return result
}
```

```go
package compress

// VByteEncode 变长字节编码
// 每个字节用7位存数据,最高位标记是否结束
func VByteEncode(nums []int64) []byte {
    var result []byte
    for _, n := range nums {
        var buf []byte
        for {
            b := byte(n & 0x7F)
            n >>= 7
            buf = append(buf, b)
            if n == 0 {
                break
            }
        }
        // 设置最后一个字节的最高位为1(结束标记)
        buf[len(buf)-1] |= 0x80
        result = append(result, buf...)
    }
    return result
}

// VByteDecode 变长字节解码
func VByteDecode(data []byte) []int64 {
    var result []int64
    var n int64
    var shift uint
    for _, b := range data {
        n |= int64(b&0x7F) << shift
        shift += 7
        if b&0x80 != 0 {
            result = append(result, n)
            n = 0
            shift = 0
        }
    }
    return result
}

// DeltaEncode 差分编码(用于有序DocID列表)
func DeltaEncode(docIDs []int64) []int64 {
    if len(docIDs) == 0 {
        return nil
    }
    deltas := make([]int64, len(docIDs))
    deltas[0] = docIDs[0]
    for i := 1; i < len(docIDs); i++ {
        deltas[i] = docIDs[i] - docIDs[i-1]
    }
    return deltas
}

// DeltaDecode 差分解码
func DeltaDecode(deltas []int64) []int64 {
    if len(deltas) == 0 {
        return nil
    }
    docIDs := make([]int64, len(deltas))
    docIDs[0] = deltas[0]
    for i := 1; i < len(deltas); i++ {
        docIDs[i] = docIDs[i-1] + deltas[i]
    }
    return docIDs
}

// CompressPostingList 压缩倒排列表
// DocID列表先差分再VByte编码
func CompressPostingList(docIDs []int64) []byte {
    deltas := DeltaEncode(docIDs)
    return VByteEncode(deltas)
}

// DecompressPostingList 解压倒排列表
func DecompressPostingList(data []byte) []int64 {
    deltas := VByteDecode(data)
    return DeltaDecode(deltas)
}
```

## 6. 搜索质量评估 ⭐⭐

### 6.1 离线评估指标

#### 6.1.1 指标对比

| 指标 | 含义 | 是否考虑位置 | 是否考虑等级 | 适用场景 |
|------|------|------------|------------|---------|
| Precision@K | Top-K中相关文档比例 | 是 | 否(二值) | 简单评估 |
| Recall@K | 相关文档在Top-K中的覆盖率 | 是 | 否(二值) | 召回评估 |
| MAP | 平均精确率均值 | 是 | 否(二值) | 整体排序质量 |
| MRR | 第一个正确结果的倒数排名 | 是 | 否(二值) | QA/导航搜索 |
| DCG/NDCG | 折损累计增益 | 是 | 是(多等级) | 排序质量(工业标准) |

#### 6.1.2 NDCG详解与计算

```
DCG@K = Σ(i=1 to K) (2^rel_i - 1) / log2(i+1)

NDCG@K = DCG@K / IDCG@K

其中 IDCG@K 是理想排序下的DCG@K(按相关度降序)

示例:
  排序结果的相关度: [3, 2, 3, 0, 1, 2]

  DCG@5 = (2^3-1)/log2(2) + (2^2-1)/log2(3) + (2^3-1)/log2(4) 
         + (2^0-1)/log2(5) + (2^1-1)/log2(6)
        = 7/1 + 3/1.585 + 7/2 + 0/2.322 + 1/2.585
        = 7 + 1.893 + 3.5 + 0 + 0.387 = 12.78

  理想排序: [3, 3, 2, 2, 1, 0]
  IDCG@5 = 7/1 + 7/1.585 + 3/2 + 3/2.322 + 1/2.585 = 14.60

  NDCG@5 = 12.78 / 14.60 = 0.875
```

#### 6.1.3 评估指标实现 (Python)

```python
import numpy as np
from typing import List

def precision_at_k(relevant: List[int], k: int) -> float:
    """Precision@K: Top-K中相关文档比例"""
    return sum(relevant[:k]) / k

def recall_at_k(relevant: List[int], k: int, total_relevant: int) -> float:
    """Recall@K: 相关文档被召回的比例"""
    if total_relevant == 0:
        return 0.0
    return sum(relevant[:k]) / total_relevant

def average_precision(relevant: List[int]) -> float:
    """AP: 单个Query的平均精确率"""
    hits = 0
    sum_precisions = 0.0
    for i, rel in enumerate(relevant):
        if rel == 1:
            hits += 1
            sum_precisions += hits / (i + 1)
    if hits == 0:
        return 0.0
    return sum_precisions / hits

def mean_average_precision(relevants: List[List[int]]) -> float:
    """MAP: 多个Query的AP均值"""
    return np.mean([average_precision(r) for r in relevants])

def mrr(relevants: List[List[int]]) -> float:
    """MRR: Mean Reciprocal Rank"""
    reciprocal_ranks = []
    for relevant in relevants:
        for i, rel in enumerate(relevant):
            if rel == 1:
                reciprocal_ranks.append(1.0 / (i + 1))
                break
        else:
            reciprocal_ranks.append(0.0)
    return np.mean(reciprocal_ranks)

def dcg_at_k(relevance: List[int], k: int) -> float:
    """DCG@K"""
    relevance = relevance[:k]
    return sum(
        (2**rel - 1) / np.log2(i + 2)
        for i, rel in enumerate(relevance)
    )

def ndcg_at_k(relevance: List[int], k: int) -> float:
    """NDCG@K"""
    dcg = dcg_at_k(relevance, k)
    ideal = sorted(relevance, reverse=True)
    idcg = dcg_at_k(ideal, k)
    if idcg == 0:
        return 0.0
    return dcg / idcg

# 批量评估
def evaluate_search(predictions: List[List[int]], k_values: List[int] = [1, 3, 5, 10]) -> dict:
    """批量评估搜索结果"""
    results = {}
    for k in k_values:
        ndcg_scores = [ndcg_at_k(pred, k) for pred in predictions]
        results[f"NDCG@{k}"] = np.mean(ndcg_scores)

        precision_scores = [precision_at_k([1 if r > 0 else 0 for r in pred], k)
                           for pred in predictions]
        results[f"P@{k}"] = np.mean(precision_scores)

    results["MAP"] = mean_average_precision(
        [[1 if r > 0 else 0 for r in pred] for pred in predictions]
    )
    results["MRR"] = mrr(
        [[1 if r > 0 else 0 for r in pred] for pred in predictions]
    )
    return results


# 示例
predictions = [
    [3, 2, 3, 0, 1, 2],  # Query1的排序结果相关度
    [3, 0, 2, 1, 0, 0],  # Query2
]
print(evaluate_search(predictions))
# {'NDCG@1': 1.0, 'NDCG@3': 0.92, 'NDCG@5': 0.87, 'NDCG@10': 0.90,
#  'P@1': 1.0, 'P@3': 0.83, 'P@5': 0.7, 'MAP': 0.88, 'MRR': 1.0}
```

### 6.3 人工评估

#### Side-by-Side (SBS) 评估

```python
from dataclasses import dataclass
from enum import Enum
from typing import List, Optional
import random

class Preference(Enum):
    A_MUCH_BETTER = 2
    A_SLIGHTLY_BETTER = 1
    TIE = 0
    B_SLIGHTLY_BETTER = -1
    B_MUCH_BETTER = -2

@dataclass
class SBSTask:
    query: str
    results_a: List[str]  # 系统A的结果
    results_b: List[str]  # 系统B的结果
    is_swapped: bool      # 是否交换了AB展示顺序(消除位置偏差)

@dataclass
class SBSJudgment:
    task: SBSTask
    preference: Preference
    reason: Optional[str]
    annotator_id: str

class SBSEvaluator:
    """Side-by-Side人工评估"""

    def generate_tasks(self, queries: List[str], system_a, system_b,
                       swap_ratio: float = 0.5) -> List[SBSTask]:
        """生成评估任务"""
        tasks = []
        for query in queries:
            results_a = system_a.search(query)
            results_b = system_b.search(query)
            is_swapped = random.random() < swap_ratio
            if is_swapped:
                results_a, results_b = results_b, results_a
            tasks.append(SBSTask(query, results_a, results_b, is_swapped))
        return tasks

    def compute_stats(self, judgments: List[SBSJudgment]) -> dict:
        """统计评估结果"""
        a_wins = sum(1 for j in judgments if self._get_real_pref(j) > 0)
        b_wins = sum(1 for j in judgments if self._get_real_pref(j) < 0)
        ties = sum(1 for j in judgments if self._get_real_pref(j) == 0)
        total = len(judgments)

        return {
            "a_win_rate": a_wins / total,
            "b_win_rate": b_wins / total,
            "tie_rate": ties / total,
            "a_wins": a_wins,
            "b_wins": b_wins,
            "ties": ties,
            "total": total,
        }

    def _get_real_pref(self, j: SBSJudgment) -> int:
        """还原真实偏好(考虑swap)"""
        pref = j.preference.value
        return -pref if j.task.is_swapped else pref
```

### 7.2 Trie树实现前缀匹配 (Go)

```go
package suggest

import (
    "sort"
    "strings"
    "sync"
)

// TrieNode Trie树节点
type TrieNode struct {
    children map[rune]*TrieNode
    isEnd    bool
    weight   int64  // 热度权重
    query    string // 完整Query
}

// Trie 前缀树
type Trie struct {
    mu   sync.RWMutex
    root *TrieNode
}

func NewTrie() *Trie {
    return &Trie{
        root: &TrieNode{children: make(map[rune]*TrieNode)},
    }
}

// Insert 插入Query及其权重
func (t *Trie) Insert(query string, weight int64) {
    t.mu.Lock()
    defer t.mu.Unlock()

    node := t.root
    lower := strings.ToLower(query)

    for _, ch := range lower {
        if _, ok := node.children[ch]; !ok {
            node.children[ch] = &TrieNode{children: make(map[rune]*TrieNode)}
        }
        node = node.children[ch]
    }
    node.isEnd = true
    node.weight = weight
    node.query = query
}

// SearchPrefix 前缀搜索,返回TopK结果
func (t *Trie) SearchPrefix(prefix string, topK int) []SuggestItem {
    t.mu.RLock()
    defer t.mu.RUnlock()

    // 找到前缀对应的节点
    node := t.root
    lower := strings.ToLower(prefix)
    for _, ch := range lower {
        if _, ok := node.children[ch]; !ok {
            return nil
        }
        node = node.children[ch]
    }

    // DFS收集所有以该前缀开头的Query
    var results []SuggestItem
    t.dfs(node, &results)

    // 按权重排序取TopK
    sort.Slice(results, func(i, j int) bool {
        return results[i].Weight > results[j].Weight
    })
    if len(results) > topK {
        results = results[:topK]
    }
    return results
}

func (t *Trie) dfs(node *TrieNode, results *[]SuggestItem) {
    if node.isEnd {
        *results = append(*results, SuggestItem{
            Query:  node.query,
            Weight: node.weight,
        })
    }
    for _, child := range node.children {
        t.dfs(child, results)
    }
}

type SuggestItem struct {
    Query  string
    Weight int64
}

// AutoCompleteService 搜索建议服务
type AutoCompleteService struct {
    trie      *Trie
    hotQueries []SuggestItem // 热搜榜
}

func NewAutoCompleteService() *AutoCompleteService {
    return &AutoCompleteService{
        trie: NewTrie(),
    }
}

// LoadQueries 加载Query库(从日志/统计)
func (s *AutoCompleteService) LoadQueries(queries map[string]int64) {
    for query, weight := range queries {
        s.trie.Insert(query, weight)
    }
}

// Suggest 搜索建议
func (s *AutoCompleteService) Suggest(prefix string, topK int) []SuggestItem {
    if prefix == "" {
        // 空输入返回热搜
        if len(s.hotQueries) > topK {
            return s.hotQueries[:topK]
        }
        return s.hotQueries
    }
    return s.trie.SearchPrefix(prefix, topK)
}
```

### 7.3 基于编辑距离的 Did You Mean

```python
from typing import List, Optional, Tuple
import heapq

class DidYouMean:
    """拼写纠错建议"""

    def __init__(self, query_freq: dict):
        """
        query_freq: {"iphone 15": 100000, "华为手机": 50000, ...}
        """
        self.query_freq = query_freq
        self.queries = list(query_freq.keys())

    def suggest(self, query: str, max_edit_dist: int = 2, top_k: int = 3) -> List[Tuple[str, int]]:
        """
        返回: [(建议Query, 编辑距离), ...]
        """
        if query in self.query_freq:
            return []  # 完全匹配,无需纠错

        candidates = []
        for candidate in self.queries:
            dist = self._edit_distance(query, candidate)
            if 0 < dist <= max_edit_dist:
                # 综合编辑距离和热度打分
                score = self.query_freq[candidate] / (dist * dist)
                candidates.append((candidate, dist, score))

        # 按score降序
        candidates.sort(key=lambda x: -x[2])
        return [(c[0], c[1]) for c in candidates[:top_k]]

    def _edit_distance(self, s1: str, s2: str) -> int:
        m, n = len(s1), len(s2)
        if abs(m - n) > 3:
            return 999  # 快速剪枝
        dp = list(range(n + 1))
        for i in range(1, m + 1):
            prev = dp[0]
            dp[0] = i
            for j in range(1, n + 1):
                temp = dp[j]
                if s1[i-1] == s2[j-1]:
                    dp[j] = prev
                else:
                    dp[j] = 1 + min(prev, dp[j], dp[j-1])
                prev = temp
        return dp[n]
```

### 8.3 代码搜索

```python
import ast
from typing import List, Dict

class CodeSearchIndex:
    """代码搜索索引"""

    def __init__(self):
        self.func_index: Dict[str, List[dict]] = {}  # 函数名 -> [代码片段]
        self.doc_index: Dict[str, List[dict]] = {}    # 文档注释 -> [代码片段]
        self.api_index: Dict[str, List[dict]] = {}    # API调用 -> [代码片段]

    def index_python_file(self, filepath: str, content: str):
        """解析Python文件建立索引"""
        try:
            tree = ast.parse(content)
        except SyntaxError:
            return

        for node in ast.walk(tree):
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                snippet = {
                    "file": filepath,
                    "name": node.name,
                    "line": node.lineno,
                    "docstring": ast.get_docstring(node) or "",
                    "args": [arg.arg for arg in node.args.args],
                }
                # 索引函数名
                self._add_to_index(self.func_index, node.name.lower(), snippet)
                # 索引文档注释
                if snippet["docstring"]:
                    for word in snippet["docstring"].lower().split():
                        self._add_to_index(self.doc_index, word, snippet)

            elif isinstance(node, ast.Call):
                # 索引API调用
                func_name = self._get_call_name(node)
                if func_name:
                    snippet = {"file": filepath, "line": node.lineno, "api": func_name}
                    self._add_to_index(self.api_index, func_name.lower(), snippet)

    def search(self, query: str, top_k: int = 10) -> List[dict]:
        """搜索代码"""
        query_lower = query.lower()
        results = []

        # 函数名匹配(权重最高)
        for key, snippets in self.func_index.items():
            if query_lower in key:
                for s in snippets:
                    results.append({**s, "score": 3.0, "match_type": "function"})

        # 文档注释匹配
        for word in query_lower.split():
            for key, snippets in self.doc_index.items():
                if word == key:
                    for s in snippets:
                        results.append({**s, "score": 2.0, "match_type": "docstring"})

        # API匹配
        for key, snippets in self.api_index.items():
            if query_lower in key:
                for s in snippets:
                    results.append({**s, "score": 1.0, "match_type": "api"})

        # 去重排序
        seen = set()
        unique_results = []
        for r in sorted(results, key=lambda x: -x["score"]):
            key = (r["file"], r.get("line", 0))
            if key not in seen:
                seen.add(key)
                unique_results.append(r)
        return unique_results[:top_k]

    def _add_to_index(self, index, key, value):
        if key not in index:
            index[key] = []
        index[key].append(value)

    def _get_call_name(self, node):
        if isinstance(node.func, ast.Name):
            return node.func.id
        elif isinstance(node.func, ast.Attribute):
            return node.func.attr
        return None
```

#### 9.2.2 RAG完整实现 (Python)

```python
import numpy as np
from typing import List, Optional, Tuple
from dataclasses import dataclass

@dataclass
class RAGDocument:
    id: str
    content: str
    metadata: dict
    embedding: Optional[np.ndarray] = None

@dataclass
class RAGResponse:
    answer: str
    sources: List[RAGDocument]
    confidence: float


class RAGPipeline:
    """RAG检索增强生成Pipeline"""

    def __init__(self, embedder, vector_store, llm_client, reranker=None):
        self.embedder = embedder         # 向量化模型
        self.vector_store = vector_store  # 向量数据库
        self.llm_client = llm_client      # LLM API客户端
        self.reranker = reranker          # 可选的重排模型

    def ingest(self, documents: List[dict]):
        """文档入库: 切片 → 向量化 → 存储"""
        for doc in documents:
            chunks = self._chunk_document(doc["content"])
            for i, chunk in enumerate(chunks):
                embedding = self.embedder.encode(chunk)
                rag_doc = RAGDocument(
                    id=f"{doc['id']}_chunk_{i}",
                    content=chunk,
                    metadata={"source": doc.get("source", ""), "chunk_idx": i},
                    embedding=embedding,
                )
                self.vector_store.add(rag_doc)

    def query(self, question: str, top_k: int = 5,
              rerank_top_k: int = 3) -> RAGResponse:
        """RAG查询主流程"""
        # Step 1: 向量检索
        query_embedding = self.embedder.encode(question)
        candidates = self.vector_store.search(query_embedding, top_k=top_k)

        # Step 2: 重排(可选)
        if self.reranker and len(candidates) > rerank_top_k:
            candidates = self._rerank(question, candidates, rerank_top_k)

        # Step 3: 构建Prompt
        prompt = self._build_prompt(question, candidates)

        # Step 4: LLM生成
        answer = self.llm_client.generate(prompt)

        # Step 5: 置信度评估
        confidence = self._estimate_confidence(question, candidates, answer)

        return RAGResponse(
            answer=answer,
            sources=candidates,
            confidence=confidence,
        )

    def _chunk_document(self, text: str, chunk_size: int = 512,
                        overlap: int = 64) -> List[str]:
        """文档切片(滑动窗口)"""
        chunks = []
        sentences = text.split("。")  # 简单按句号分割
        current_chunk = []
        current_len = 0

        for sent in sentences:
            sent = sent.strip()
            if not sent:
                continue
            sent_len = len(sent)

            if current_len + sent_len > chunk_size and current_chunk:
                chunks.append("。".join(current_chunk) + "。")
                # 保留overlap部分
                overlap_sents = []
                overlap_len = 0
                for s in reversed(current_chunk):
                    if overlap_len + len(s) > overlap:
                        break
                    overlap_sents.insert(0, s)
                    overlap_len += len(s)
                current_chunk = overlap_sents
                current_len = overlap_len

            current_chunk.append(sent)
            current_len += sent_len

        if current_chunk:
            chunks.append("。".join(current_chunk) + "。")
        return chunks

    def _rerank(self, question: str, docs: List[RAGDocument],
                top_k: int) -> List[RAGDocument]:
        """Cross-Encoder重排"""
        pairs = [(question, doc.content) for doc in docs]
        scores = self.reranker.predict(pairs)
        sorted_indices = np.argsort(-np.array(scores))
        return [docs[i] for i in sorted_indices[:top_k]]

    def _build_prompt(self, question: str, docs: List[RAGDocument]) -> str:
        """构建RAG Prompt"""
        context_parts = []
        for i, doc in enumerate(docs):
            context_parts.append(f"[文档{i+1}] {doc.content}")
        context = "\n\n".join(context_parts)

        prompt = f"""你是一个知识助手。请根据以下参考文档回答用户的问题。
如果文档中没有相关信息,请明确说明"根据现有资料无法回答"。
请在回答中标注信息来源(如[文档1])。

参考文档:
{context}

用户问题: {question}

请回答:"""
        return prompt

    def _estimate_confidence(self, question: str, docs: List[RAGDocument],
                             answer: str) -> float:
        """简单的置信度评估"""
        if not docs:
            return 0.0
        # 基于检索分数的简单估计
        avg_similarity = np.mean([
            np.dot(self.embedder.encode(question), doc.embedding)
            for doc in docs if doc.embedding is not None
        ])
        return min(max(float(avg_similarity), 0.0), 1.0)
```

### 9.3 Hybrid Search (混合检索)

#### 9.3.1 混合检索策略

| 策略 | 方法 | 优点 | 缺点 |
|------|------|------|------|
| 线性加权 | score = α*BM25 + (1-α)*向量 | 简单 | α需调参 |
| RRF | 基于排名的倒数融合 | 无需归一化 | 不利用分数信息 |
| 学习融合 | 训练融合模型 | 效果最好 | 需要标注数据 |

#### 9.3.2 Hybrid Search实现 (Go)

```go
package hybrid

import (
    "math"
    "sort"
)

// HybridSearcher 混合检索
type HybridSearcher struct {
    bm25Searcher   BM25Searcher
    vectorSearcher VectorSearcher
    alpha          float64 // BM25权重 (1-alpha为向量权重)
    fusionMethod   string  // "linear", "rrf"
}

type SearchResult struct {
    DocID int64
    Score float64
}

// Search 混合搜索
func (hs *HybridSearcher) Search(query string, queryVec []float64, topK int) []SearchResult {
    // 并行执行两路检索
    bm25Results := hs.bm25Searcher.Search(query, topK*2)
    vecResults := hs.vectorSearcher.Search(queryVec, topK*2)

    switch hs.fusionMethod {
    case "rrf":
        return hs.rrfFusion(bm25Results, vecResults, topK)
    default:
        return hs.linearFusion(bm25Results, vecResults, topK)
    }
}

// linearFusion 线性加权融合
func (hs *HybridSearcher) linearFusion(bm25Results, vecResults []SearchResult, topK int) []SearchResult {
    // MinMax归一化
    bm25Norm := normalize(bm25Results)
    vecNorm := normalize(vecResults)

    scores := make(map[int64]float64)
    for _, r := range bm25Norm {
        scores[r.DocID] += hs.alpha * r.Score
    }
    for _, r := range vecNorm {
        scores[r.DocID] += (1 - hs.alpha) * r.Score
    }

    return topKResults(scores, topK)
}

// rrfFusion Reciprocal Rank Fusion
// RRF(d) = Σ 1 / (k + rank_i(d))
func (hs *HybridSearcher) rrfFusion(bm25Results, vecResults []SearchResult, topK int) []SearchResult {
    const k = 60.0 // RRF常数

    scores := make(map[int64]float64)

    // BM25排名贡献
    for rank, r := range bm25Results {
        scores[r.DocID] += 1.0 / (k + float64(rank+1))
    }
    // 向量排名贡献
    for rank, r := range vecResults {
        scores[r.DocID] += 1.0 / (k + float64(rank+1))
    }

    return topKResults(scores, topK)
}

func normalize(results []SearchResult) []SearchResult {
    if len(results) == 0 {
        return results
    }
    minS, maxS := results[0].Score, results[0].Score
    for _, r := range results {
        if r.Score < minS {
            minS = r.Score
        }
        if r.Score > maxS {
            maxS = r.Score
        }
    }
    rangeS := maxS - minS
    if rangeS == 0 {
        rangeS = 1
    }
    normed := make([]SearchResult, len(results))
    for i, r := range results {
        normed[i] = SearchResult{
            DocID: r.DocID,
            Score: (r.Score - minS) / rangeS,
        }
    }
    return normed
}

func topKResults(scores map[int64]float64, topK int) []SearchResult {
    result := make([]SearchResult, 0, len(scores))
    for docID, score := range scores {
        result = append(result, SearchResult{DocID: docID, Score: score})
    }
    sort.Slice(result, func(i, j int) bool {
        return result[i].Score > result[j].Score
    })
    if len(result) > topK {
        result = result[:topK]
    }
    return result
}
```

```go
// EcomSearchEngine 电商搜索引擎
type EcomSearchEngine struct {
    queryUnderstanding *EcomQueryUnderstanding
    esClient           *ElasticsearchClient
    milvusClient       *MilvusClient
    rankService        *RankService
    cacheService       *CacheService
    suggestService     *AutoCompleteService
    abTestService      *ABTestService

// Search 电商搜索主流程
func (e *EcomSearchEngine) Search(ctx context.Context, req *EcomSearchRequest) (*EcomSearchResponse, error) {
    start := time.Now()

    // 0. 缓存检查(热门Query)
    if cached, ok := e.cacheService.Get(req.CacheKey()); ok {
        return cached.(*EcomSearchResponse), nil
    }

    // 1. Query理解
    tokens := tokenize(req.Query)
    corrected := spellCorrect(req.Query)
    if corrected != req.Query {
        tokens = tokenize(corrected)
    }
    parsedQuery := e.queryUnderstanding.Parse(corrected, tokens)

    // 2. 多路召回
    var candidates []*Product

    // 2.1 ES倒排召回(主力)
    esQuery := e.buildESQuery(parsedQuery, req)
    esResults, err := e.esClient.Search(ctx, esQuery)
    if err != nil {
        return nil, fmt.Errorf("es search failed: %w", err)
    }
    candidates = append(candidates, esResults...)

    // 2.2 向量召回(语义补充)
    queryVec := e.encodeQuery(parsedQuery)
    vecResults, err := e.milvusClient.Search(ctx, queryVec, 200)
    if err == nil {
        candidates = append(candidates, vecResults...)
    }

    // 2.3 去重
    candidates = dedup(candidates)

    // 3. 排序
    if req.SortBy == "relevance" || req.SortBy == "" {
        // 综合排序: 使用排序模型
        candidates, err = e.rankService.Rank(ctx, parsedQuery, candidates, req)
        if err != nil {
            return nil, err
        }
    } else {
        // 确定性排序: 价格/销量
        candidates = e.deterministicSort(candidates, req.SortBy)
    }

    // 4. 筛选面聚合
    facets := e.buildFacets(candidates)

    // 5. 分页
    offset := req.Page * req.PageSize
    end := offset + req.PageSize
    if end > len(candidates) {
        end = len(candidates)
    }
    pageItems := candidates[offset:end]

    resp := &EcomSearchResponse{
        Total:  int64(len(candidates)),
        Items:  pageItems,
        Facets: facets,
    }

    // 6. 缓存热门Query
    if isHotQuery(req.Query) {
        e.cacheService.Set(req.CacheKey(), resp, 5*time.Minute)
    }

    // 7. 日志(用于后续优化)
    e.logSearchEvent(ctx, req, resp, time.Since(start))

    return resp, nil
}

// buildESQuery 构建ES查询
func (e *EcomSearchEngine) buildESQuery(parsed *EcomParsedQuery, req *EcomSearchRequest) map[string]interface{} {
    must := []map[string]interface{}{}

    // 关键词匹配
    if len(parsed.Keywords) > 0 {
        must = append(must, map[string]interface{}{
            "multi_match": map[string]interface{}{
                "query":  joinTokens(parsed.Keywords),
                "fields": []string{"title^3", "brand^2", "category", "attributes"},
                "type":   "best_fields",
            },
        })
    }

    // 品牌过滤
    filter := []map[string]interface{}{}
    if parsed.BrandID > 0 {
        filter = append(filter, map[string]interface{}{
            "term": map[string]interface{}{"brand_id": parsed.BrandID},
        })
    }

    // 价格范围
    if req.PriceRange[1] > 0 {
        filter = append(filter, map[string]interface{}{
            "range": map[string]interface{}{
                "price": map[string]interface{}{
                    "gte": req.PriceRange[0],
                    "lte": req.PriceRange[1],
                },
            },
        })
    }

    // 属性筛选
    for attr, values := range req.Filters {
        filter = append(filter, map[string]interface{}{
            "terms": map[string]interface{}{
                "attributes." + attr: values,
            },
        })
    }

    return map[string]interface{}{
        "query": map[string]interface{}{
            "bool": map[string]interface{}{
                "must":   must,
                "filter": filter,
            },
        },
        "size": 1000,
    }
}

func (e *EcomSearchEngine) deterministicSort(items []*Product, sortBy string) []*Product {
    sort.Slice(items, func(i, j int) bool {
        switch sortBy {
        case "price_asc":
            return items[i].Price < items[j].Price
        case "price_desc":
            return items[i].Price > items[j].Price
        case "sales":
            return items[i].SalesCount > items[j].SalesCount
        default:
            return items[i].Score > items[j].Score
        }
    })
    return items
}

func (e *EcomSearchEngine) buildFacets(items []*Product) []FacetResult {
    // 聚合筛选面
    brandCount := make(map[string]int64)
    categoryCount := make(map[string]int64)
    priceRanges := make(map[string]int64)

    for _, item := range items {
        brandCount[item.Brand]++
        categoryCount[item.Category]++
        priceRange := getPriceRange(item.Price)
        priceRanges[priceRange]++
    }

    return []FacetResult{
        buildFacetResult("brand", brandCount),
        buildFacetResult("category", categoryCount),
        buildFacetResult("price_range", priceRanges),
    }
}
```

### 11.1 治理补充 (3条)

1. **搜索日志体系**: 建立完善的搜索日志采集(Query日志、点击日志、曝光日志),用于效果分析、模型训练、ABTest评估,是搜索系统持续优化的基础

3. **搜索降级策略**: 多路召回要有降级方案(向量服务异常→仅用倒排,精排超时→用粗排分数),保证搜索核心链路高可用

### 11.2 初级高频速刷 (5题)

**Q1: 什么是倒排索引?和正排索引有什么区别?**

正排索引: DocID → [Term1, Term2, ...]，通过文档ID找到包含的词列表。倒排索引: Term → [DocID1, DocID2, ...]，通过词找到包含该词的文档列表。搜索引擎核心使用倒排索引,因为搜索时是给定Query词找文档,倒排索引能直接定位。

**Q2: BM25是什么?核心参数是什么?**

BM25是基于概率模型的文本相关性打分算法,是TF-IDF的改进版。核心公式结合了IDF(逆文档频率)和归一化的TF(词频)。关键参数: k1(默认1.2,控制TF饱和度)和b(默认0.75,控制文档长度归一化)。

**Q3: 搜索系统的四个阶段是什么?**

召回→粗排→精排→重排。召回从亿级文档中快速筛选万级候选;粗排用轻量模型筛到千级;精排用复杂模型精确打分到百级;重排加入业务策略(多样性/去重)输出最终结果。

**Q4: NDCG是什么?为什么搜索常用NDCG?**

NDCG(Normalized Discounted Cumulative Gain)是归一化折损累计增益,衡量排序质量。优势: 考虑位置(越靠前权重越高)、支持多级相关度(不只是0/1)、归一化后可跨Query比较。搜索中标准评估指标。

**Q5: 什么是多路召回?为什么需要多路召回?**

多路召回是指并行使用多种召回策略(倒排/向量/图/标签)获取候选文档,再融合去重。原因: 单一召回有盲区(倒排无法处理语义匹配,向量对精确匹配不足),多路互补提高整体召回率。

### 11.3 中级进阶 (5题)

**Q1: 对比Pointwise/Pairwise/Listwise三种LTR方法**

- Pointwise: 独立预测每个文档相关性(回归/分类),简单但忽略文档间关系
- Pairwise: 学习文档对的相对顺序(RankNet/LambdaRank),考虑了序关系但文档对数量O(n²)
- Listwise: 直接优化整个列表(LambdaMART/ListNet),最接近真实排序目标但训练复杂
- 工业界主流: LambdaMART(GBDT) + 深度模型(如DCN)

**Q2: HNSW算法原理?为什么搜索场景常用?**

HNSW(Hierarchical Navigable Small World): 多层图结构,底层包含所有节点,上层逐步稀疏。搜索从顶层贪心寻找最近邻,逐层下降到底层精确搜索。优势: 查询O(logN)复杂度、召回率高(>95%)、支持动态插入、无需训练步骤。常用于向量召回场景。

**Q3: RAG的核心流程?常见优化手段?**

核心流程: 文档切片→向量化→索引 | 查询→向量化→检索→重排→LLM生成。优化手段:
- 文档切片: 语义切片(按段落/章节),适当overlap
- 检索增强: HyDE(生成假设文档)、多Query检索、Query改写
- 重排: Cross-Encoder重排提高相关性
- Prompt优化: 结构化Context、引用来源

**Q4: 如何设计搜索建议(AutoComplete)系统?**

数据源: 搜索日志中高频Query + 人工补充 + 实时热点。核心结构: Trie树支持前缀匹配,按热度/个性化权重排序。工程实现: Redis存储热词+Trie前缀、拼音匹配(处理中文输入法场景)、个性化加权(结合用户历史)。延迟要求: <50ms(keypress触发)。

**Q5: 搜索系统如何处理"无结果"问题?**

分层策略:
1. Query改写: 纠错、同义词替换、去除限定词
2. 放宽召回: AND→OR、降低匹配阈值
3. 语义扩展: 向量召回补充、类目推荐
4. 兜底策略: 推荐相关类目、展示"您是否在找xxx"
5. 长期治理: 分析高频无结果Query,补充商品/索引

### 11.4 深入问答 (15+ Q&A)

**Q1: 搜索系统和推荐系统的本质区别是什么?**

A: 核心区别在于用户意图的明确性。搜索有明确Query,目标是找到与Query最相关的结果(Relevance);推荐无Query,目标是猜测用户兴趣(Interest)。这导致了:
- 搜索以Query-Doc匹配为核心,推荐以User-Item匹配为核心
- 搜索强调精确度(找到就行),推荐强调多样性(探索+利用)
- 搜索结果可解释(关键词高亮),推荐可解释性较弱
- 搜索实时性极高(Query变结果立刻变),推荐可离线更新

**Q2: 详细解释BM25相比TF-IDF的改进?**

A: 两个关键改进:

```python
# TF-IDF
score = tf(t,d) * idf(t)
# 问题1: TF线性增长,词频100 vs 50差距过大
# 问题2: 不考虑文档长度

# BM25
tf_bm25 = tf * (k1+1) / (tf + k1*(1-b+b*dl/avgdl))
# 改进1: TF饱和 — 词频增加到一定程度后增益递减(k1控制)
# 改进2: 文档长度归一化 — 长文档自然词频高,需要惩罚(b控制)
# 当k1=0时退化为IDF,b=0时不考虑文档长度
```

**Q3: 实时索引如何实现?有哪些挑战?**

A: 常见方案:

2. **ES Near Real-Time**: ES的refresh机制(默认1s),新文档写入内存Buffer,refresh后生成新Segment可搜索

- 性能: 频繁refresh/merge影响查询性能
- 并发: 读写并发控制

```go
func (ri *RealtimeIndex) SwapBuffer() {
    ri.mu.Lock()
    // 1. 切换活跃Buffer指针(原子操作)
    ri.activeIdx = 1 - ri.activeIdx
    ri.mu.Unlock()
    // 2. 异步将旧Buffer合并到基础索引
    go ri.mergeToBase()
```

**Q4: 向量检索中,IVF-PQ的原理是什么?什么场景用?**

A: IVF-PQ结合了倒排文件(IVF)和乘积量化(PQ):
- **IVF**: 先用K-Means将向量空间分为N个聚类,检索时只在最近的nprobe个聚类内搜索,减少计算量
- **PQ**: 将D维向量分成M个子空间,每个子空间独立做K-Means量化(256个中心),原始向量压缩为M个字节

```
原始: 768维 float32 = 3072 bytes/向量
PQ(M=96): 96 bytes/向量 → 压缩32倍

5000万商品: 原始~143GB → PQ后~4.5GB(可放内存)
```

适用场景: 超大规模(>1亿)且内存有限的场景。精度比HNSW略低,但内存开销小很多。

```python
def chinese_spell_correct(query):
    # 1. 检查是否在词典中
    if is_valid_query(query):

    # 2. 拼音级纠错: 将query转拼音,匹配拼音相似的候选
```

**Q7: 多路召回的融合策略有哪些?RRF为什么好用?**

A: 融合策略:
1. **分数加权**: 各路分数归一化后加权求和,需要调α
2. **RRF(Reciprocal Rank Fusion)**: `score(d) = Σ 1/(k+rank_i(d))`,k通常取60
3. **学习融合**: 将各路分数作为特征,训练融合模型

RRF优势:
- **无需归一化**: 只用排名,不依赖原始分数(不同召回路的分数不可比)
- **鲁棒性好**: 对某一路的异常分数不敏感
- **无超参**: 除k外无需调参

```python
def rrf_fusion(result_lists, k=60):
    scores = {}
    for results in result_lists:
        for rank, doc_id in enumerate(results):
            scores[doc_id] = scores.get(doc_id, 0) + 1.0 / (k + rank + 1)
    return sorted(scores.items(), key=lambda x: -x[1])
```

**Q10: Hybrid Search (向量+关键词混合) 的最佳实践?**

A: 最佳实践:

```go
func (hs *HybridSearcher) Search(query string) []Result {

    ch1 := make(chan []Result)
    ch2 := make(chan []Result)
    go func() { ch1 <- hs.bm25.Search(query) }()
    go func() { ch2 <- hs.vector.Search(encode(query)) }()

    return rrfFusion(<-ch1, <-ch2)
```

**Q11: 如何设计搜索系统的降级策略?**

A: 分级降级方案:

| 级别 | 触发条件 | 降级措施 |
|------|---------|---------|
| L1 | 向量召回超时/异常 | 仅用倒排召回 |
| L2 | 精排模型超时 | 使用粗排分数/BM25分数 |
| L3 | 主索引异常 | 切换到备用索引 |
| L4 | 全量不可用 | 返回缓存/热门结果 |

```go
func (e *Engine) searchWithDegradation(ctx context.Context, req *Request) (*Response, error) {
    // 设置超时
    ctx, cancel := context.WithTimeout(ctx, 200*time.Millisecond)
    defer cancel()

    resp, err := e.fullSearch(ctx, req)
    if err == nil {
        return resp, nil
    }

    // L1降级: 仅倒排
    resp, err = e.invertedOnlySearch(ctx, req)
    if err == nil {
        resp.Degraded = true
        return resp, nil
    }

    // L4降级: 缓存兜底
    if cached := e.cache.GetFallback(req.Query); cached != nil {
        cached.Degraded = true
        return cached, nil
    }

    return nil, fmt.Errorf("all search paths failed")
}
```

**Q15: HyDE (Hypothetical Document Embeddings) 的原理和适用场景?**

3. 用这个假设答案的embedding去做向量检索
4. 假设答案和真实文档在语义空间更接近(都是陈述式,比问题→文档的匹配更准)

```python
            f"请写一段约100字的文字来回答: {question}"
        )
        # 2. 用假设文档的embedding检索
        hypo_embedding = self.encoder.encode(hypothetical_doc)
        results = self.vector_store.search(hypo_embedding, top_k)
```

**Q16: 搜索系统中如何保证结果多样性?**

A: 多样性策略:

1. **MMR (Maximal Marginal Relevance)**:
```python
def mmr_rerank(query_vec, docs, lambda_param=0.5, top_k=10):
    """MMR: 平衡相关性和多样性"""
    selected = []
    remaining = list(range(len(docs)))

    while len(selected) < top_k and remaining:
        best_idx = None
        best_score = -float('inf')
        for idx in remaining:
            relevance = cosine_sim(query_vec, docs[idx].vec)
            redundancy = max(
                cosine_sim(docs[idx].vec, docs[s].vec)
                for s in selected
            ) if selected else 0
            score = lambda_param * relevance - (1-lambda_param) * redundancy
            if score > best_score:
                best_score = score
                best_idx = idx
        selected.append(best_idx)
        remaining.remove(best_idx)
    return [docs[i] for i in selected]
```

2. **类目打散**: 连续不超过N个同类目商品
3. **品牌打散**: 同品牌间隔展示
4. **价格区间打散**: 不同价位穿插展示

**Q17: 如何设计一个高性能的Query Cache?**

A: Query Cache设计:

```go
package cache

import (
    "sync"
    "time"
)

// QueryCache LRU + TTL 的Query缓存
type QueryCache struct {
    mu       sync.RWMutex
    capacity int
    ttl      time.Duration
    cache    map[string]*cacheEntry
    lruList  *LRUList
}

type cacheEntry struct {
    key       string
    value     interface{}
    expireAt  time.Time
    lruNode   *LRUNode
}

func (qc *QueryCache) Get(key string) (interface{}, bool) {
    qc.mu.RLock()
    entry, ok := qc.cache[key]
    qc.mu.RUnlock()

    if !ok {
        return nil, false
    }
    if time.Now().After(entry.expireAt) {
        qc.mu.Lock()
        delete(qc.cache, key)
        qc.mu.Unlock()
        return nil, false
    }

    // 更新LRU位置
    qc.mu.Lock()
    qc.lruList.MoveToFront(entry.lruNode)
    qc.mu.Unlock()

    return entry.value, true
}

func (qc *QueryCache) Set(key string, value interface{}) {
    qc.mu.Lock()
    defer qc.mu.Unlock()

    if entry, ok := qc.cache[key]; ok {
        entry.value = value
        entry.expireAt = time.Now().Add(qc.ttl)
        qc.lruList.MoveToFront(entry.lruNode)
        return
    }

    // 容量满时淘汰
    if len(qc.cache) >= qc.capacity {
        oldest := qc.lruList.RemoveLast()
        delete(qc.cache, oldest.key)
    }

    node := qc.lruList.AddToFront(key)
    qc.cache[key] = &cacheEntry{
        key:      key,
        value:    value,
        expireAt: time.Now().Add(qc.ttl),
        lruNode:  node,
    }
}
```

缓存策略:
- 热门Query(Top 1000): 长TTL(5-10min)
- 普通Query: 短TTL(1-2min)
- 带用户个性化的Query: 不缓存或per-user cache
- 缓存Key: `hash(query + filters + sort + page)` 去除不影响结果的参数

**Q18: 搜索排序中如何处理位置偏差(Position Bias)?**

A: 位置偏差: 用户倾向于点击排名靠前的结果,即使靠后的更相关。

处理方法:
1. **点击模型**: 用Examination Hypothesis建模 `P(click) = P(examine) * P(relevant)`
2. **IPW (Inverse Propensity Weighting)**: 对低位点击给更高权重
3. **随机化实验**: 随机打乱部分位置,收集无偏点击数据
4. **Unbiased LTR**: 在训练时加入位置偏差矫正

```python
# IPW示例: 矫正位置偏差的点击标签
def correct_position_bias(clicks, positions):
    """
    clicks: [1, 0, 1, 0, ...]  点击/未点击
    positions: [1, 2, 3, 4, ...]  展示位置
    """
    # 位置的被观察概率(examination probability)
    exam_prob = {1: 1.0, 2: 0.8, 3: 0.6, 4: 0.4, 5: 0.3,
                 6: 0.2, 7: 0.15, 8: 0.1, 9: 0.08, 10: 0.05}

    corrected_labels = []
    for click, pos in zip(clicks, positions):
        if click == 1:
            # 低位点击给更高权重
            weight = 1.0 / exam_prob.get(pos, 0.05)
            corrected_labels.append(min(weight, 10.0))  # 截断
        else:
            corrected_labels.append(0.0)
    return corrected_labels
```
